# 移动端架构模式

## 概述

良好的架构是可维护、可测试、可扩展应用的基础。移动端架构经历了从 MVC → MVP → MVVM → Clean Architecture 的演进，并逐渐融合模块化与依赖注入等工程实践。本指南涵盖 MVVM、Clean Architecture、模块化拆分和依赖注入四大主题，适用于 Flutter、React Native、iOS、Android 等所有移动平台。

---

## 一、MVVM 模式

### 1.1 架构分层

```
┌─────────────────────────────────────────┐
│           View (UI Layer)               │
│  SwiftUI / Compose / Flutter Widget     │
│  - 负责渲染 UI                           │
│  - 绑定 ViewModel 数据                   │
│  - 处理用户交互 → 委托 ViewModel          │
├─────────────────────────────────────────┤
│         ViewModel (Logic Layer)         │
│  - 持有 UI 状态（State / Observable）    │
│  - 处理业务逻辑                          │
│  - 调用 Repository/UseCase              │
│  - 不引用任何 UI 框架类型                │
├─────────────────────────────────────────┤
│        Model (Data Layer)               │
│  - Repository / UseCase / DataSource    │
│  - 数据模型 Entity                       │
│  - 网络/本地数据访问                      │
└─────────────────────────────────────────┘
```

**数据流向：** View → Action → ViewModel → Repository → 返回数据 → ViewModel 更新 State → View 自动刷新

### 1.2 Flutter MVVM 实现

```dart
// ViewModel（不引用 Flutter UI 库）
class UserViewModel extends ChangeNotifier {
  final UserRepository _repository;

  UserViewModel(this._repository);

  // --- State ---
  User? _user;
  bool _isLoading = false;
  String? _errorMessage;

  User? get user => _user;
  bool get isLoading => _isLoading;
  String? get errorMessage => _errorMessage;

  // --- Actions ---
  Future<void> loadUser(String id) async {
    _isLoading = true;
    _errorMessage = null;
    notifyListeners();

    try {
      _user = await _repository.getUser(id);
    } catch (e) {
      _errorMessage = e.toString();
    } finally {
      _isLoading = false;
      notifyListeners();
    }
  }

  Future<void> updateProfile({String? name, String? avatar}) async {
    if (_user == null) return;
    _isLoading = true;
    notifyListeners();

    try {
      _user = await _repository.updateUser(
        _user!.id,
        name: name,
        avatar: avatar,
      );
    } catch (e) {
      _errorMessage = e.toString();
    } finally {
      _isLoading = false;
      notifyListeners();
    }
  }
}

// View（纯 UI）
class UserProfilePage extends StatelessWidget {
  final String userId;

  const UserProfilePage({super.key, required this.userId});

  @override
  Widget build(BuildContext context) {
    return ChangeNotifierProvider(
      create: (_) => UserViewModel(UserRepository())..loadUser(userId),
      child: Consumer<UserViewModel>(
        builder: (context, vm, _) {
          return Scaffold(
            appBar: AppBar(title: const Text('个人资料')),
            body: _buildBody(vm),
          );
        },
      ),
    );
  }

  Widget _buildBody(UserViewModel vm) {
    if (vm.isLoading) return const Center(child: CircularProgressIndicator());
    if (vm.errorMessage != null) return Center(child: Text(vm.errorMessage!));
    if (vm.user == null) return const Center(child: Text('无数据'));

    final user = vm.user!;
    return Column(
      children: [
        CircleAvatar(radius: 50, backgroundImage: NetworkImage(user.avatar)),
        const SizedBox(height: 16),
        Text(user.name, style: const TextStyle(fontSize: 24)),
      ],
    );
  }
}
```

### 1.3 Android (Compose) MVVM 实现

```kotlin
// UI State（不可变）
data class UserUiState(
    val user: User? = null,
    val isLoading: Boolean = false,
    val error: String? = null

// ViewModel
@HiltViewModel
class UserViewModel @Inject constructor(
    private val getUserUseCase: GetUserUseCase,
    private val updateUserUseCase: UpdateUserUseCase
) : ViewModel() {

    private val _uiState = MutableStateFlow(UserUiState())
    val uiState: StateFlow<UserUiState> = _uiState.asStateFlow()

    fun loadUser(id: String) {
        viewModelScope.launch {
            _uiState.update { it.copy(isLoading = true, error = null) }
            getUserUseCase(id)
                .onSuccess { user ->
                    _uiState.update { it.copy(user = user, isLoading = false) }
                }
                .onFailure { error ->
                    _uiState.update { it.copy(isLoading = false, error = error.message) }
                }
        }
    }
}

// View
@Composable
fun UserProfileScreen(
    userId: String,
    viewModel: UserViewModel = hiltViewModel()
) {
    val uiState by viewModel.uiState.collectAsState()

    LaunchedEffect(userId) {
        viewModel.loadUser(userId)
    }

    when {
        uiState.isLoading -> LoadingIndicator()
        uiState.error != null -> ErrorView(uiState.error!!) { viewModel.loadUser(userId) }
        uiState.user != null -> UserContent(uiState.user!!)
    }
}
```

### 1.4 iOS (SwiftUI) MVVM 实现

```swift
@MainActor
class UserViewModel: ObservableObject {
    @Published private(set) var state: LoadState<User> = .idle

    private let getUserUseCase: GetUserUseCase

    init(getUserUseCase: GetUserUseCase) {
        self.getUserUseCase = getUserUseCase
    }

    func loadUser(id: String) async {
        state = .loading
        do {
            let user = try await getUserUseCase.execute(id: id)
            state = .loaded(user)
        } catch {
            state = .error(error.localizedDescription)
        }
    }
}

enum LoadState<T> {
    case idle
    case loading
    case loaded(T)
    case error(String)
}

struct UserProfileView: View {
    let userId: String
    @StateObject private var viewModel: UserViewModel

    init(userId: String) {
        self.userId = userId
        _viewModel = StateObject(wrappedValue: UserViewModel(
            getUserUseCase: GetUserUseCase(repository: UserRepository())
        ))
    }

    var body: some View {
        switch viewModel.state {
        case .idle:
            Color.clear.task { await viewModel.loadUser(id: userId) }
        case .loading:
            ProgressView()
        case .loaded(let user):
            UserContentView(user: user)
        case .error(let message):
            ErrorView(message: message) {
                Task { await viewModel.loadUser(id: userId) }
            }
        }
    }
}
```

### 1.5 MVVM 最佳实践

| 实践 | 说明 |
|------|------|
| ViewModel 无 UI 依赖 | 不导入 Flutter/Compose/SwiftUI 类型 |
| 单一数据源 | UI 状态集中在 ViewModel 一个 State 对象 |
| 不可变状态 | 使用 `data class` / 不可变对象描述状态 |
| 单向数据流 | View → Intent → ViewModel → State → View |
| ViewModel 可测试 | Mock Repository 即可单元测试 ViewModel |

## 二、Clean Architecture

### 2.1 架构分层

```
┌──────────────────────────────────────────────────┐
│                  Frameworks & Drivers             │
│  UI (SwiftUI/Compose/Flutter) │ DB │ Network │    │
├──────────────────────────────────────────────────┤
│                   Interface Adapters              │
│     Presenters │ ViewModels │ Controllers         │
│     Repositories Impl │ Data Mappers              │
├──────────────────────────────────────────────────┤
│                      Use Cases                     │
│         Application Business Rules (Interactors)   │
├──────────────────────────────────────────────────┤
│                      Entities                      │
│           Enterprise Business Rules (Domain)       │
└──────────────────────────────────────────────────┘

依赖方向：外层 → 内层（内层不知道外层的存在）
```

### 2.2 Domain 层（Entity + UseCase）

```dart
// === Domain Layer ===

// Entity（纯领域模型，无框架依赖）
class User {
  final String id;
  final String name;
  final String email;
  final String? avatarUrl;

  const User({
    required this.id,
    required this.name,
    required this.email,
    this.avatarUrl,
  });
}

// Repository 接口（由 Domain 定义，Data 层实现）
abstract class UserRepository {
  Future<User> getUser(String id);
  Future<User> updateUser(String id, {String? name, String? avatarUrl});
  Future<void> deleteUser(String id);
}

// UseCase（单一职责）
class GetUserUseCase {
  final UserRepository repository;

  GetUserUseCase(this.repository);

  Future<User> call(String id) async {
    // 可以在此添加业务规则校验
    if (id.isEmpty) {
      throw ArgumentError('用户ID不能为空');
    }
    return await repository.getUser(id);
  }
}

class UpdateUserUseCase {
  final UserRepository repository;

  UpdateUserUseCase(this.repository);

  Future<User> call({
    required String id,
    String? name,
    String? avatarUrl,
  }) async {
    return await repository.updateUser(id, name: name, avatarUrl: avatarUrl);
  }
}
```

### 2.3 Data 层（Repository 实现 + DataSource）

```dart
// === Data Layer ===

// DTO（数据传输对象，与 API 格式一致）
class UserDto {
  final String id;
  final String name;
  final String email;
  final String? avatar;

  const UserDto({
    required this.id,
    required this.name,
    required this.email,
    this.avatar,
  });

  factory UserDto.fromJson(Map<String, dynamic> json) => UserDto(
        id: json['id'] as String,
        name: json['name'] as String,
        email: json['email'] as String,
        avatar: json['avatar'] as String?,
      );

  Map<String, dynamic> toJson() => {
        'id': id,
        'name': name,
        'email': email,
        if (avatar != null) 'avatar': avatar,
      };
}

// Mapper（DTO ↔ Entity 转换）
extension UserDtoMapper on UserDto {
  User toEntity() => User(
        id: id,
        name: name,
        email: email,
        avatarUrl: avatar,
      );
}

// Remote Data Source
class UserRemoteDataSource {
  final Dio _dio;

  UserRemoteDataSource(this._dio);

  Future<UserDto> fetchUser(String id) async {
    final response = await _dio.get('/users/$id');
    return UserDto.fromJson(response.data);
  }

  Future<UserDto> updateUser(String id, Map<String, dynamic> data) async {
    final response = await _dio.put('/users/$id', data: data);
    return UserDto.fromJson(response.data);
  }
}

// Repository 实现
class UserRepositoryImpl implements UserRepository {
  final UserRemoteDataSource remoteDataSource;

  UserRepositoryImpl(this.remoteDataSource);

  @override
  Future<User> getUser(String id) async {
    final dto = await remoteDataSource.fetchUser(id);
    return dto.toEntity();
  }

  @override
  Future<User> updateUser(String id, {String? name, String? avatarUrl}) async {
    final data = <String, dynamic>{};
    if (name != null) data['name'] = name;
    if (avatarUrl != null) data['avatar'] = avatarUrl;

    final dto = await remoteDataSource.updateUser(id, data);
    return dto.toEntity();
  }

  @override
  Future<void> deleteUser(String id) async {
    await remoteDataSource.deleteUser(id);
  }
}
```

### 2.4 Presentation 层（ViewModel）

```dart
// === Presentation Layer ===

class UserViewModel extends ChangeNotifier {
  final GetUserUseCase _getUserUseCase;
  final UpdateUserUseCase _updateUserUseCase;

  UserViewModel(this._getUserUseCase, this._updateUserUseCase);

  User? _user;
  bool _isLoading = false;
  String? _error;

  User? get user => _user;
  bool get isLoading => _isLoading;
  String? get error => _error;

  Future<void> loadUser(String id) async {
    _isLoading = true;
    _error = null;
    notifyListeners();

    final result = await _getUserUseCase(id).then((_) => null).catchError((e) => e.toString());
    // 或使用 Either 类型进行更优雅的错误处理
    try {
      _user = await _getUserUseCase(id);
    } catch (e) {
      _error = e.toString();
    } finally {
      _isLoading = false;
      notifyListeners();
    }
  }
}

// === 依赖组装（Composition Root）===
UserViewModel createUserViewModel() {
  final dio = Dio(BaseOptions(baseUrl: 'https://api.example.com'));
  final remoteDataSource = UserRemoteDataSource(dio);
  final repository = UserRepositoryImpl(remoteDataSource);

  return UserViewModel(
    GetUserUseCase(repository),
    UpdateUserUseCase(repository),
  );
}
```

### 2.5 使用 Either 进行错误处理

```dart
// 通用 Result 类型
sealed class Result<T> {
  const Result();
}

class Success<T> extends Result<T> {
  final T data;
  const Success(this.data);
}

class Failure<T> extends Result<T> {
  final String message;
  final int? code;
  const Failure(this.message, [this.code]);
}

// UseCase 返回 Result
class GetUserUseCase {
  final UserRepository repository;
  GetUserUseCase(this.repository);

  Future<Result<User>> call(String id) async {
    try {
      final user = await repository.getUser(id);
      return Success(user);
    } on ApiException catch (e) {
      return Failure(e.message, e.statusCode);
    } catch (e) {
      return Failure('未知错误: $e');
    }
  }
}

// ViewModel 处理
Future<void> loadUser(String id) async {
  _isLoading = true;
  notifyListeners();

  final result = await _getUserUseCase(id);
  switch (result) {
    case Success(:final data):
      _user = data;
      _error = null;
    case Failure(:final message):
      _error = message;
  }

  _isLoading = false;
  notifyListeners();
}
```

### 2.6 Clean Architecture 最佳实践

| 原则 | 说明 |
|------|------|
| 依赖倒置 | 内层定义接口，外层实现 |
| 单一职责 | 每个 UseCase 只做一件事 |
| 可测试 | 每层可独立测试，Mock 依赖 |
| 框架无关 | Domain 层不依赖任何框架 |
| 数据映射 | 使用 Mapper 隔离 DTO 与 Entity |

### 4.4 iOS - Resolver / 手动 DI Container

```swift
// DI Container
class DIContainer {
    static let shared = DIContainer()

    private var factories: [String: () -> Any] = [:]
    private var singletons: [String: Any] = [:]

    func register<T>(_ type: T.Type, factory: @escaping () -> T) {
        factories[String(describing: type)] = factory
    }

    func registerSingleton<T>(_ type: T.Type, factory: @escaping () -> T) {
        let key = String(describing: type)
        singletons[key] = factory()
    }

    func resolve<T>(_ type: T.Type) -> T {
        let key = String(describing: type)
        if let singleton = singletons[key] as? T {
            return singleton
        }
        guard let factory = factories[key], let instance = factory() as? T else {
            fatalError("未注册: \(type)")
        }
        return instance
    }
}

// 注册
DIContainer.shared.registerSingleton(APIClient.self) {
    APIClient(baseUrl: "https://api.example.com")
}

DIContainer.shared.register(UserRepository.self) {
    UserRepositoryImpl(apiClient: DIContainer.shared.resolve(APIClient.self))
}

DIContainer.shared.register(GetUserUseCase.self) {
    GetUserUseCase(repository: DIContainer.shared.resolve(UserRepository.self))
}

// 使用
let useCase = DIContainer.shared.resolve(GetUserUseCase.self)
let viewModel = UserViewModel(getUserUseCase: useCase)
```

## 五、架构决策矩阵

### 5.1 何时选择哪种架构

| 项目规模 | 推荐架构 | 复杂度 |
|----------|---------|--------|
| 原型/Demo | MVC / 简单 MVVM | ⭐ |
| 小型应用 | MVVM | ⭐⭐ |
| 中型应用 | MVVM + Repository | ⭐⭐⭐ |
| 大型应用 | Clean Architecture + MVVM | ⭐⭐⭐⭐ |
| 超大型/团队 | Clean Architecture + 模块化 + DI | ⭐⭐⭐⭐⭐ |

### 5.2 架构对比总结

| 维度 | MVC | MVP | MVVM | Clean Architecture |
|------|-----|-----|------|-------------------|
| 关注点分离 | 弱 | 中 | 强 | 最强 |
| 可测试性 | 低 | 中 | 高 | 最高 |
| 学习成本 | 低 | 中 | 中 | 高 |
| 代码量 | 少 | 中 | 中 | 多 |
| 适合规模 | 小 | 中小 | 中大 | 大 |
| 数据绑定 | 无 | 无 | 有 | 可选 |
| 依赖方向 | 混乱 | 单向 | 单向 | 严格单向 |

### 5.3 跨平台框架架构推荐

| 框架 | 状态管理 | 推荐架构 | DI 方案 |
|------|---------|---------|---------|
| Flutter | Riverpod / Bloc | Clean + MVVM | get_it / Riverpod |
| React Native | Zustand / Redux | MVVM + Repository | tsyringe |
| iOS (SwiftUI) | @Observable / Combine | Clean + MVVM | Resolver / 手动 |
| Android (Compose) | ViewModel + StateFlow | Clean + MVVM | Hilt |
