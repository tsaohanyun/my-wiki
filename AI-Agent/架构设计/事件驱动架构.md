# 事件驱动架构

## 概述

事件驱动架构（Event-Driven Architecture, EDA）是一种以**事件**的产生、检测、消费和响应为核心的软件架构模式。系统中的组件通过发布和订阅事件进行松耦合通信，而非直接调用。

## 核心概念

| 概念 | 说明 |
|------|------|
| **事件（Event）** | 系统中发生的状态变化或动作的不可变记录 |
| **事件生产者** | 检测并发布事件的组件 |
| **事件消费者** | 订阅并处理事件的组件 |
| **事件通道** | 连接生产者和消费者的中间件（消息队列/事件总线） |
| **事件存储** | 持久化事件序列的存储系统 |

## 1. 事件溯源（Event Sourcing）

### 原理

事件溯源不存储实体的当前状态，而是存储导致状态变化的所有事件序列。当前状态通过**重放事件**来重建。

```
事件流：
  OrderCreated → ItemAdded → ItemAdded → PaymentReceived → OrderShipped
      │              │           │            │               │
      ▼              ▼           ▼            ▼               ▼
  State: created  +item1     +item2       paid=true        shipped=true

当前状态 = fold(initial_state, [e1, e2, e3, e4, e5])
```

### 事件存储设计

```python
from dataclasses import dataclass, field
from datetime import datetime
from typing import List, Dict, Any
import json
import uuid

@dataclass(frozen=True)
class Event:
    """不可变事件基类"""
    event_id: str = field(default_factory=lambda: str(uuid.uuid4()))
    aggregate_id: str = ""
    event_type: str = ""
    timestamp: datetime = field(default_factory=datetime.utcnow)
    version: int = 0
    data: Dict[str, Any] = field(default_factory=dict)

    def to_dict(self) -> dict:
        return {
            "event_id": self.event_id,
            "aggregate_id": self.aggregate_id,
            "event_type": self.event_type,
            "timestamp": self.timestamp.isoformat(),
            "version": self.version,
            "data": self.data,
        }


class EventStore:
    """事件存储 - Append-Only Log"""

    def __init__(self):
        self._events: Dict[str, List[Event]] = {}  # aggregate_id -> events

    def append(self, event: Event, expected_version: int = -1) -> None:
        """追加事件（乐观并发控制）"""
        events = self._events.get(event.aggregate_id, [])
        current_version = len(events)

        if expected_version >= 0 and current_version != expected_version:
            raise ConcurrencyError(
                f"Expected version {expected_version}, got {current_version}"
            )

        events.append(event)
        self._events[event.aggregate_id] = events

    def get_events(self, aggregate_id: str) -> List[Event]:
        """获取聚合的所有事件"""
        return self._events.get(aggregate_id, [])

    def get_events_from(self, aggregate_id: str, version: int) -> List[Event]:
        """从指定版本获取事件"""
        events = self.get_events(aggregate_id)
        return [e for e in events if e.version >= version]


class ConcurrencyError(Exception):
    pass
```

### 聚合根与事件重放

```python
class OrderAggregate:
    """订单聚合根 - 通过事件溯源重建状态"""

    def __init__(self, order_id: str):
        self.order_id = order_id
        self.items: List[dict] = []
        self.status = "draft"
        self.total = 0.0
        self.version = 0

    def apply(self, event: Event) -> None:
        """应用事件到当前状态"""
        handler = getattr(self, f"_on_{event.event_type}", None)
        if handler:
            handler(event.data)
        self.version += 1

    def _on_order_created(self, data: dict):
        self.status = "created"

    def _on_item_added(self, data: dict):
        self.items.append(data)
        self.total += data.get("price", 0) * data.get("quantity", 1)

    def _on_order_paid(self, data: dict):
        self.status = "paid"

    def _on_order_shipped(self, data: dict):
        self.status = "shipped"

    @classmethod
    def rebuild(cls, order_id: str, events: List[Event]) -> "OrderAggregate":
        """从事件流重建聚合"""
        aggregate = cls(order_id)
        for event in events:
            aggregate.apply(event)
        return aggregate
```

### CQRS 实现

```python
from abc import ABC, abstractmethod
from typing import Optional

# ============ 命令端 ============

class Command(ABC):
    pass

@dataclass
class CreateOrderCommand(Command):
    customer_id: str
    items: List[dict]

@dataclass
class AddItemCommand(Command):
    order_id: str
    product_id: str
    quantity: int
    price: float

class CommandHandler:
    """命令处理器 - 处理写操作"""

    def __init__(self, event_store: EventStore):
        self.event_store = event_store

    def handle_create_order(self, cmd: CreateOrderCommand) -> str:
        order_id = str(uuid.uuid4())
        event = Event(
            aggregate_id=order_id,
            event_type="order_created",
            data={"customer_id": cmd.customer_id, "items": cmd.items},
        )
        self.event_store.append(event)
        return order_id

    def handle_add_item(self, cmd: AddItemCommand) -> None:
        events = self.event_store.get_events(cmd.order_id)
        order = OrderAggregate.rebuild(cmd.order_id, events)

        if order.status != "created":
            raise ValueError("Cannot add items to a non-created order")

        event = Event(
            aggregate_id=cmd.order_id,
            event_type="item_added",
            version=order.version + 1,
            data={
                "product_id": cmd.product_id,
                "quantity": cmd.quantity,
                "price": cmd.price,
            },
        )
        self.event_store.append(event, expected_version=order.version)


# ============ 查询端 ============

class OrderReadModel:
    """订单读模型 - 为查询优化"""

    def __init__(self):
        self._views: Dict[str, dict] = {}

    def get_order_summary(self, order_id: str) -> Optional[dict]:
        return self._views.get(order_id)

    def get_customer_orders(self, customer_id: str) -> List[dict]:
        return [
            v for v in self._views.values()
            if v.get("customer_id") == customer_id
        ]

    def update_from_event(self, event: Event) -> None:
        """从事件更新读模型（由投影器调用）"""
        if event.event_type == "order_created":
            self._views[event.aggregate_id] = {
                "order_id": event.aggregate_id,
                "customer_id": event.data["customer_id"],
                "status": "created",
                "items": [],
                "total": 0,
            }
        elif event.event_type == "item_added":
            view = self._views.get(event.aggregate_id)
            if view:
                view["items"].append(event.data)
                view["total"] += event.data["price"] * event.data["quantity"]


class EventProjector:
    """事件投影器 - 将写模型事件投影到读模型"""

    def __init__(self, read_models: List[Any]):
        self.read_models = read_models

    def project(self, event: Event):
        for model in self.read_models:
            if hasattr(model, "update_from_event"):
                model.update_from_event(event)
```

## 3. 事件总线（Event Bus）

```python
from typing import Callable, Dict, List
from collections import defaultdict
import asyncio

class EventBus:
    """事件总线 - 发布/订阅模式"""

    def __init__(self):
        self._handlers: Dict[str, List[Callable]] = defaultdict(list)
        self._middleware: List[Callable] = []

    def subscribe(self, event_type: str, handler: Callable) -> None:
        self._handlers[event_type].append(handler)

    def add_middleware(self, middleware: Callable) -> None:
        """添加中间件（日志、重试、死信等）"""
        self._middleware.append(middleware)

    async def publish(self, event: Event) -> None:
        """发布事件给所有订阅者"""
        handlers = self._handlers.get(event.event_type, [])
        tasks = []
        for handler in handlers:
            wrapped = handler
            for mw in reversed(self._middleware):
                wrapped = mw(wrapped)
            tasks.append(self._safe_call(wrapped, event))
        await asyncio.gather(*tasks)

    async def _safe_call(self, handler, event):
        try:
            result = handler(event)
            if asyncio.iscoroutine(result):
                await result
        except Exception as e:
            print(f"Handler error: {e}")
            # 发送到死信队列
            await self._send_to_dlq(event, e)

    async def _send_to_dlq(self, event: Event, error: Exception):
        print(f"Event {event.event_id} sent to DLQ: {error}")


# 中间件示例
def logging_middleware(handler):
    def wrapper(event):
        print(f"[LOG] Processing {event.event_type} -> {event.event_id}")
        return handler(event)
    return wrapper

def retry_middleware(handler, max_retries=3):
    def wrapper(event):
        for attempt in range(max_retries):
            try:
                return handler(event)
            except Exception as e:
                if attempt == max_retries - 1:
                    raise
                print(f"Retry {attempt + 1}/{max_retries}")
    return wrapper
```

class EventConsumer:
    """事件消费者基类"""

    def __init__(self, group_id: str):
        self.group_id = group_id
        self._handlers: Dict[str, Callable] = {}

    def on(self, event_type: str):
        """装饰器：注册事件处理器"""
        def decorator(func):
            self._handlers[event_type] = func
            return func
        return decorator

    async def handle(self, raw_message: dict) -> None:
        event_type = raw_message.get("event_type")
        handler = self._handlers.get(event_type)
        if handler:
            event = Event(**raw_message)
            await handler(event)


# 使用示例

## 最佳实践

### ✅ 推荐

1. **事件不可变** - 一旦发布，事件不能修改或删除
2. **事件版本化** - 使用 schema registry 管理事件版本
3. **幂等消费** - 消费者必须处理重复消息
4. **最终一致性** - 接受读写模型之间的短暂延迟
5. **事件快照** - 定期创建快照避免重放过多事件
6. **死信队列** - 处理失败的事件进入 DLQ 后续分析
7. **事件溯源 + CQRS 结合** - 写端事件溯源，读端物化视图

### ❌ 避免

1. **避免同步调用** - 事件驱动的核心是异步解耦
2. **避免大事件** - 事件应只包含必要数据
3. **避免循环依赖** - 事件流应为有向无环图（DAG）
4. **避免忽略幂等** - 消息至少投递一次，重复消费不可避免
