# 高可用架构设计

## 概述

高可用（High Availability, HA）指系统在面对各种故障时仍能持续提供服务的能力。可用性通常用"几个9"来衡量：

| 可用性 | 年停机时间 | 等级 |
|--------|-----------|------|
| 99%（2个9） | 3.65 天 | 基本 |
| 99.9%（3个9） | 8.76 小时 | 较高 |
| 99.99%（4个9） | 52.6 分钟 | 高 |
| 99.999%（5个9） | 5.26 分钟 | 极高 |

```python
class ServiceRegistry:
    """服务注册与发现"""

    def __init__(self):
        self._services: Dict[str, List[ServiceInstance]] = {}
        self._health_checker = HealthChecker()

    def register(self, service_name: str, instance: ServiceInstance):
        if service_name not in self._services:
            self._services[service_name] = []
        self._services[service_name].append(instance)
        self._health_checker.add_instance(instance)

    def deregister(self, service_name: str, instance_id: str):
        instances = self._services.get(service_name, [])
        self._services[service_name] = [
            i for i in instances if i.instance_id != instance_id
        ]

    def get_healthy_instances(self, service_name: str) -> List[ServiceInstance]:
        return [
            i for i in self._services.get(service_name, [])
            if i.status == "healthy"
        ]

class HealthChecker:
    """健康检查"""

    def __init__(self, check_interval: int = 10, timeout: int = 5,
                 unhealthy_threshold: int = 3, healthy_threshold: int = 2):
        self.check_interval = check_interval
        self.timeout = timeout
        self.unhealthy_threshold = unhealthy_threshold
        self.healthy_threshold = healthy_threshold
        self._failure_counts: Dict[str, int] = {}
        self._success_counts: Dict[str, int] = {}

    async def check(self, instance: ServiceInstance) -> bool:
        try:
            # TCP 或 HTTP 健康检查
            reader, writer = await asyncio.wait_for(
                asyncio.open_connection(instance.host, instance.port),
                timeout=self.timeout
            )
            writer.close()
            await writer.wait_closed()
            self._on_success(instance)
            return True
        except Exception:
            self._on_failure(instance)
            return False

    def _on_success(self, instance: ServiceInstance):
        self._failure_counts[instance.instance_id] = 0
        count = self._success_counts.get(instance.instance_id, 0) + 1
        self._success_counts[instance.instance_id] = count
        if count >= self.healthy_threshold:
            instance.status = "healthy"

    def _on_failure(self, instance: ServiceInstance):
        self._success_counts[instance.instance_id] = 0
        count = self._failure_counts.get(instance.instance_id, 0) + 1
        self._failure_counts[instance.instance_id] = count
        if count >= self.unhealthy_threshold:
            instance.status = "unhealthy"
```

```python
class ReplicationManager:
    """数据复制管理器"""

    def __init__(self, primary: "Database", replicas: List["Database"]):
        self.primary = primary
        self.replicas = replicas
        self.replication_lag_threshold_ms = 1000

    async def write(self, operation: dict) -> bool:
        """写入主节点，异步复制到从节点"""
        # 写入主节点
        result = await self.primary.execute(operation)

        # 异步复制到所有从节点
        replication_tasks = [
            self._replicate_to(replica, operation)
            for replica in self.replicas
        ]
        await asyncio.gather(*replication_tasks, return_exceptions=True)
        return result

    async def read(self, operation: dict, consistency: str = "eventual") -> Any:
        """读取数据"""
        if consistency == "strong":
            # 强一致性读：从主节点读
            return await self.primary.execute(operation)
        else:
            # 最终一致性读：从从节点读（负载均衡）
            replica = self._select_replica()
            return await replica.execute(operation)

    def _select_replica(self) -> "Database":
        """选择延迟最低的从节点"""
        best = min(self.replicas, key=lambda r: r.replication_lag_ms)
        return best

    async def _replicate_to(self, replica: "Database", operation: dict):
        try:
            await replica.execute(operation)
        except Exception as e:
            print(f"Replication failed to {replica.id}: {e}")
```

### 自动故障转移流程

```
┌──────────────────────────────────────────────────────┐
│                 自动故障转移流程                        │
├──────────────────────────────────────────────────────┤
│                                                      │
│  ┌──────────┐    心跳检测     ┌──────────┐           │
│  │  主节点   │ ──────────────→│  从节点   │           │
│  │ (Active) │               │(Standby) │           │
│  └──────────┘               └──────────┘           │
│       │                           │                  │
│       │ 故障检测（心跳超时）          │                  │
│       ✗                           │                  │
│       │                           │                  │
│  ┌────▼─────────────────────────────▼──┐             │
│  │           故障转移控制器              │             │
│  │  1. 确认主节点故障（多次检测）         │             │
│  │  2. 选举新主节点                     │             │
│  │  3. 更新 DNS/VIP                    │             │
│  │  4. 通知客户端                       │             │
│  │  5. 旧主节点恢复后作为从节点           │             │
│  └────────────────────────────────────┘             │
│                                                      │
│  故障转移时间 = 检测时间 + 选举时间 + 切换时间           │
│            =  10s      +   5s      +   2s             │
│            ≈  17s（典型值）                            │
└──────────────────────────────────────────────────────┘
```

```python
class FailoverManager:
    """故障转移管理器"""

    def __init__(self, primary: ServiceInstance,
                 standbys: List[ServiceInstance],
                 health_checker: HealthChecker):
        self.primary = primary
        self.standbys = standbys
        self.health_checker = health_checker
        self._current_primary = primary
        self._lock = asyncio.Lock()

    async def monitor(self):
        """持续监控主节点状态"""
        while True:
            is_healthy = await self.health_checker.check(self._current_primary)
            if not is_healthy:
                await self._do_failover()
            await asyncio.sleep(self.health_checker.check_interval)

    async def _do_failover(self):
        async with self._lock:
            # 双重检查
            if self._current_primary.status == "healthy":
                return

            print("[Failover] Primary is down, initiating failover...")

            # 选择最佳 standby
            new_primary = self._elect_new_primary()
            if not new_primary:
                print("[Failover] No healthy standby available!")
                # 告警
                return

            # 执行切换
            old_primary = self._current_primary
            self._current_primary = new_primary
            new_primary.status = "primary"
            self.standbys.remove(new_primary)
            self.standbys.append(old_primary)

            # 更新 VIP / DNS
            await self._update_routing(new_primary)

            print(f"[Failover] Switched from {old_primary.instance_id} "
                  f"to {new_primary.instance_id}")

    def _elect_new_primary(self) -> Optional[ServiceInstance]:
        """选举新主节点（选择优先级最高的健康 standby）"""
        healthy_standbys = [
            s for s in self.standbys
            if s.status == "healthy"
        ]
        if not healthy_standbys:
            return None
        return max(healthy_standbys, key=lambda s: s.weight)

    async def _update_routing(self, new_primary: ServiceInstance):
        """更新路由表 / VIP / DNS"""
        print(f"[Failover] Updating VIP to {new_primary.host}:{new_primary.port}")
```

## 3. 负载均衡

### 算法对比

```python
import random
import hashlib
from typing import List
from collections import defaultdict

class LoadBalancer:
    """负载均衡器"""

    @staticmethod
    def round_robin(instances: List[ServiceInstance],
                    state: dict) -> ServiceInstance:
        """轮询"""
        idx = state.get("rr_index", 0) % len(instances)
        state["rr_index"] = idx + 1
        return instances[idx]

    @staticmethod
    def weighted_round_robin(instances: List[ServiceInstance],
                             state: dict) -> ServiceInstance:
        """加权轮询"""
        current_weight = state.get("cw", 0)
        current_idx = state.get("ci", -1)

        while True:
            current_idx = (current_idx + 1) % len(instances)
            if current_idx == 0:
                current_weight = current_weight - 1
                if current_weight <= 0:
                    current_weight = max(i.weight for i in instances)
                    if current_weight == 0:
                        return instances[0]
            if instances[current_idx].weight >= current_weight:
                state["cw"] = current_weight
                state["ci"] = current_idx
                return instances[current_idx]

    @staticmethod
    def least_connections(instances: List[ServiceInstance],
                          connection_counts: dict) -> ServiceInstance:
        """最少连接"""
        return min(instances,
                   key=lambda i: connection_counts.get(i.instance_id, 0))

    @staticmethod
    def consistent_hash(instances: List[ServiceInstance],
                        key: str, hash_ring: ConsistentHash) -> ServiceInstance:
        """一致性哈希"""
        node_id = hash_ring.get_node(key)
        return next(i for i in instances if i.instance_id == node_id)

    @staticmethod
    def ip_hash(instances: List[ServiceInstance],
                client_ip: str) -> ServiceInstance:
        """IP 哈希"""
        h = int(hashlib.md5(client_ip.encode()).hexdigest(), 16)
        return instances[h % len(instances)]

    @staticmethod
    def random(instances: List[ServiceInstance]) -> ServiceInstance:
        """随机"""
        return random.choice(instances)
```

```python
class DisasterRecovery:
    """容灾管理器"""

    def __init__(self, regions: dict):
        self.regions = regions  # region_name -> RegionConfig
        self.active_region: str = "primary"
        self.dr_state = "normal"  # normal, degraded, failover

    async def check_regions(self):
        """检查所有区域健康状态"""
        results = {}
        for name, config in self.regions.items():
            results[name] = await self._check_region_health(config)
        return results

    async def _check_region_health(self, config: dict) -> dict:
        checks = {
            "api": await self._ping_endpoint(config["api_endpoint"]),
            "database": await self._check_db(config["db_url"]),
            "cache": await self._check_cache(config["cache_url"]),
        }
        all_healthy = all(checks.values())
        return {"healthy": all_healthy, "details": checks}

    async def failover_to(self, target_region: str):
        """执行灾备切换"""
        if target_region not in self.regions:
            raise ValueError(f"Unknown region: {target_region}")

        print(f"[DR] Starting failover to {target_region}...")

        # 1. 停止主区域流量
        await self._stop_traffic(self.active_region)

        # 2. 等待数据同步完成
        await self._wait_for_sync(target_region)

        # 3. 提升目标区域为主
        await self._promote_region(target_region)

        # 4. 切换 DNS
        await self._switch_dns(target_region)

        # 5. 验证新主区域
        health = await self._check_region_health(self.regions[target_region])
        if not health["healthy"]:
            raise RuntimeError("Target region unhealthy after failover")

        self.active_region = target_region
        self.dr_state = "failover"
        print(f"[DR] Failover to {target_region} completed")

    async def failback(self):
        """灾备回切"""
        print("[DR] Starting failback to primary...")
        # 类似 failover 但需要额外的数据同步验证
        primary = "primary"
        await self._sync_data(self.active_region, primary)
        await self.failover_to(primary)
        self.dr_state = "normal"
```

### 限流算法

```python
import time
from threading import Lock

class TokenBucket:
    """令牌桶限流器"""

    def __init__(self, rate: float, capacity: int):
        self.rate = rate          # 令牌生成速率（个/秒）
        self.capacity = capacity  # 桶容量
        self.tokens = capacity    # 当前令牌数
        self.last_time = time.monotonic()
        self._lock = Lock()

    def allow(self) -> bool:
        with self._lock:
            now = time.monotonic()
            # 补充令牌
            elapsed = now - self.last_time
            self.tokens = min(self.capacity, self.tokens + elapsed * self.rate)
            self.last_time = now

            if self.tokens >= 1:
                self.tokens -= 1
                return True
            return False


class SlidingWindowCounter:
    """滑动窗口计数限流器"""

    def __init__(self, limit: int, window_seconds: int):
        self.limit = limit
        self.window_seconds = window_seconds
        self._counters: Dict[int, int] = {}
        self._lock = Lock()

    def allow(self) -> bool:
        with self._lock:
            now = time.time()
            current_window = int(now)
            prev_window = current_window - 1
            elapsed_in_window = now - current_window

            # 计算滑动窗口内的总请求数
            prev_count = self._counters.get(prev_window, 0)
            curr_count = self._counters.get(current_window, 0)
            # 加权计算
            total = prev_count * (1 - elapsed_in_window) + curr_count

            if total < self.limit:
                self._counters[current_window] = curr_count + 1
                return True
            return False


class CircuitBreaker:
    """熔断器"""

    def __init__(self, failure_threshold: int = 5,
                 recovery_timeout: int = 30,
                 half_open_max: int = 3):
        self.failure_threshold = failure_threshold
        self.recovery_timeout = recovery_timeout
        self.half_open_max = half_open_max
        self.state = "closed"  # closed, open, half_open
        self.failure_count = 0
        self.success_count = 0
        self.last_failure_time = 0
        self._lock = Lock()

    def allow_request(self) -> bool:
        with self._lock:
            if self.state == "closed":
                return True
            elif self.state == "open":
                if time.time() - self.last_failure_time > self.recovery_timeout:
                    self.state = "half_open"
                    self.success_count = 0
                    return True
                return False
            elif self.state == "half_open":
                return self.success_count < self.half_open_max
        return False

    def record_success(self):
        with self._lock:
            if self.state == "half_open":
                self.success_count += 1
                if self.success_count >= self.half_open_max:
                    self.state = "closed"
                    self.failure_count = 0

    def record_failure(self):
        with self._lock:
            self.failure_count += 1
            self.last_failure_time = time.time()
            if self.failure_count >= self.failure_threshold:
                self.state = "open"
```

## 最佳实践

### ✅ 推荐

1. **无状态设计** - 应用层无状态，状态存储在外部（数据库/缓存）
2. **多可用区部署** - 至少跨 2 个可用区
3. **自动扩缩容** - 基于指标（CPU/内存/QPS）自动伸缩
4. **优雅关闭** - 接收 SIGTERM 后先排空连接再关闭
5. **重试 + 熔断 + 降级** - 三板斧组合使用
6. **定期演练** - Chaos Engineering，主动注入故障
7. **监控告警** - 关键指标（延迟、错误率、饱和度）实时监控

### ❌ 避免

1. **避免单点故障** - 每层都要有冗余
2. **避免同步强依赖** - 超时和降级是必须的
3. **避免忽略数据备份** - 定期验证备份可恢复性
4. **避免过度设计** - 根据业务 SLA 选择合适的可用性等级
