# 测试驱动开发指南

## 概述

本指南提供测试驱动开发（TDD）的方法论和最佳实践。

## 1. TDD流程

### 红-绿-重构

```
1. 红：编写失败的测试
2. 绿：编写最少代码使测试通过
3. 重构：改进代码结构
```

### 示例流程

```python
# 1. 红：编写测试
def test_add():
    assert add(1, 2) == 3

# 2. 绿：编写实现
def add(a, b):
    return a + b

# 3. 重构：改进代码
def add(a: int, b: int) -> int:
    """两数相加"""
    return a + b
```

## 2. 单元测试

### Python测试

```python
import pytest

# 测试函数
def test_add():
    assert add(1, 2) == 3
    assert add(-1, 1) == 0
    assert add(0, 0) == 0

# 测试类
class TestCalculator:
    def test_add(self):
        assert add(1, 2) == 3
    
    def test_subtract(self):
        assert subtract(3, 1) == 2
    
    def test_multiply(self):
        assert multiply(2, 3) == 6
    
    def test_divide(self):
        assert divide(6, 2) == 3
        
        with pytest.raises(ZeroDivisionError):
            divide(1, 0)
```

### 测试夹具


## 3. 集成测试

### 数据库测试

```python
import pytest
from sqlalchemy import create_engine
from sqlalchemy.orm import sessionmaker

@pytest.fixture
def db_session():
    engine = create_engine('sqlite:///:memory:')
    Session = sessionmaker(bind=engine)
    session = Session()
    Base.metadata.create_all(engine)
    yield session
    session.close()

def test_create_user(db_session):
    user = User(name="John", email="john@example.com")
    db_session.add(user)
    db_session.commit()
    
    saved_user = db_session.query(User).filter_by(email="john@example.com").first()
    assert saved_user is not None
    assert saved_user.name == "John"
```

### API测试

```python
import pytest
from app import create_app

@pytest.fixture
def client():
    app = create_app()
    app.config['TESTING'] = True
    with app.test_client() as client:
        yield client

def test_get_users(client):
    response = client.get('/api/users')
    assert response.status_code == 200
    assert isinstance(response.json['data'], list)

def test_create_user(client):
    response = client.post('/api/users', json={
        'name': 'John',
        'email': 'john@example.com'
    })
    assert response.status_code == 201
    assert response.json['data']['name'] == 'John'
```

## 4. Mock和Stub

### Mock对象

```python
from unittest.mock import Mock, patch

# Mock对象
def test_send_email():
    mock_smtp = Mock()
    mock_smtp.sendmail.return_value = True
    
    with patch('smtplib.SMTP', return_value=mock_smtp):
        result = send_email('test@example.com', 'Subject', 'Body')
    
    assert result is True
    mock_smtp.sendmail.assert_called_once()

# Mock返回值
def test_get_user():
    mock_db = Mock()
    mock_db.query.return_value.filter_by.return_value.first.return_value = User(id=1, name='John')
    
    user = get_user(1, mock_db)
    assert user.name == 'John'
```

### Stub

```python
class StubEmailService:
    def __init__(self):
        self.sent_emails = []
    
    def send(self, to, subject, body):
        self.sent_emails.append({'to': to, 'subject': subject, 'body': body})
        return True

def test_welcome_email():
    stub_email = StubEmailService()
    service = UserService(email_service=stub_email)
    
    service.register('John', 'john@example.com')
    
    assert len(stub_email.sent_emails) == 1
    assert stub_email.sent_emails[0]['to'] == 'john@example.com'
```

## 7. 测试最佳实践

### 测试命名

```python
# ❌ 错误（命名不清晰）
def test_1():
    pass

# ✅ 正确（描述性命名）
def test_calculate_tax_with_valid_amount():
    pass

def test_calculate_tax_raises_error_for_negative_amount():
    pass
```

### 测试结构

```python
def test_create_user():
    # Arrange（准备）
    user_data = {'name': 'John', 'email': 'john@example.com'}
    
    # Act（执行）
    user = create_user(user_data)
    
    # Assert（断言）
    assert user.name == 'John'
    assert user.email == 'john@example.com'
    assert user.id is not None
```

## 8. 持续集成

### CI配置

```yaml
# .github/workflows/test.yml
name: Tests

on: [push, pull_request]

jobs:
  test:
    runs-on: ubuntu-latest
    
    steps:
    - uses: actions/checkout@v2
    
    - name: Set up Python
      uses: actions/setup-python@v2
      with:
        python-version: '3.9'
    
    - name: Install dependencies
      run: |
        pip install -r requirements.txt
        pip install pytest pytest-cov
    
    - name: Run tests
      run: pytest --cov=app --cov-report=xml
    
    - name: Upload coverage
      uses: codecov/codecov-action@v1
```
