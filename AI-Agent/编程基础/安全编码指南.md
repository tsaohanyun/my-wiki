# 安全编码指南

## 概述

本指南提供安全编码的最佳实践。

## 1. 输入验证

### SQL注入防护

```python
# ❌ 错误（字符串拼接）
query = f"SELECT * FROM users WHERE name = '{name}'"

# ✅ 正确（参数化查询）
query = "SELECT * FROM users WHERE name = %s"
cursor.execute(query, (name,))

# ✅ 正确（ORM）
User.objects.filter(name=name)
```

### XSS防护

```python
# ❌ 错误（直接输出用户输入）
html = f"<div>{user_input}</div>"

# ✅ 正确（转义HTML）
import html
safe_input = html.escape(user_input)
html = f"<div>{safe_input}</div>"

# ✅ 正确（使用模板引擎自动转义）
# Jinja2默认自动转义
{{ user_input }}
```

### 命令注入防护

```python
# ❌ 错误（字符串拼接）
import os
os.system(f"ping {user_input}")

# ✅ 正确（使用列表参数）
import subprocess
subprocess.run(["ping", "-c", "1", user_input], capture_output=True)

# ✅ 正确（验证输入）
import re
if re.match(r'^[\w\.\-]+$', user_input):
    subprocess.run(["ping", "-c", "1", user_input])
```

### 路径遍历防护

```python
# ❌ 错误（直接拼接路径）
file_path = os.path.join(base_dir, user_input)

# ✅ 正确（验证路径）
import os
real_path = os.path.realpath(os.path.join(base_dir, user_input))
if real_path.startswith(os.path.realpath(base_dir)):
    # 路径安全
    pass
```

## 2. 认证安全

### 密码存储

```python
import bcrypt

# 哈希密码
def hash_password(password):
    salt = bcrypt.gensalt()
    hashed = bcrypt.hashpw(password.encode(), salt)
    return hashed

# 验证密码
def verify_password(password, hashed):
    return bcrypt.checkpw(password.encode(), hashed)
```

### JWT安全

```python
import jwt
from datetime import datetime, timedelta

# 生成Token
def create_token(user_id):
    payload = {
        'user_id': user_id,
        'exp': datetime.utcnow() + timedelta(hours=1),
        'iat': datetime.utcnow()
    }
    return jwt.encode(payload, SECRET_KEY, algorithm='HS256')

# 验证Token
def verify_token(token):
    try:
        payload = jwt.decode(token, SECRET_KEY, algorithms=['HS256'])
        return payload['user_id']
    except jwt.ExpiredSignatureError:
        # Token过期
        return None
    except jwt.InvalidTokenError:
        # Token无效
        return None
```

## 3. 授权安全

### RBAC实现

```python
class RBAC:
    def __init__(self):
        self.roles = {}
    
    def add_role(self, role, permissions):
        self.roles[role] = permissions
    
    def check_permission(self, user_role, permission):
        if user_role in self.roles:
            return permission in self.roles[user_role]
        return False

# 使用
rbac = RBAC()
rbac.add_role('admin', ['read', 'write', 'delete'])
rbac.add_role('user', ['read'])

if rbac.check_permission(user_role, 'write'):
    # 允许写操作
    pass
```

### API授权

```python
from functools import wraps

def require_permission(permission):
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            user = get_current_user()
            if not has_permission(user, permission):
                return {'error': 'Forbidden'}, 403
            return func(*args, **kwargs)
        return wrapper
    return decorator

@require_permission('admin')
def delete_user(user_id):
    # 删除用户
    pass
```

## 4. 数据安全

### 敏感数据加密

```python
from cryptography.fernet import Fernet

# 生成密钥
key = Fernet.generate_key()
cipher = Fernet(key)

# 加密数据
def encrypt_data(data):
    return cipher.encrypt(data.encode())

# 解密数据
def decrypt_data(encrypted_data):
    return cipher.decrypt(encrypted_data).decode()
```

### 数据脱敏

```python
def mask_phone(phone):
    """手机号脱敏"""
    return phone[:3] + '****' + phone[7:]

def mask_email(email):
    """邮箱脱敏"""
    parts = email.split('@')
    return parts[0][:2] + '***@' + parts[1]

def mask_id_card(id_card):
    """身份证脱敏"""
    return id_card[:6] + '********' + id_card[14:]
```

## 6. 日志安全

### 安全日志记录

```python
import logging

logger = logging.getLogger(__name__)

def log_security_event(event_type, user_id, details):
    """记录安全事件"""
    logger.info(f"Security Event: {event_type}", extra={
        'user_id': user_id,
        'event_type': event_type,

        'ip_address': request.remote_addr,

```

### 日志脱敏

```python
def sanitize_log_data(data):
    """脱敏日志数据"""
    sensitive_fields = ['password', 'token', 'secret', 'credit_card']
    sanitized = data.copy()
    for field in sensitive_fields:
        if field in sanitized:
            sanitized[field] = '***'
    return sanitized
```

## 7. 错误处理

### 安全错误信息

```python
# ❌ 错误（暴露系统信息）
except Exception as e:
    return {'error': str(e)}, 500

# ✅ 正确（通用错误信息）
except Exception as e:
    logger.error(f"Internal error: {str(e)}")
    return {'error': 'Internal server error'}, 500
```
