本文档涵盖常用数据结构和算法的 TypeScript 实现。

## 1. 排序算法

### 快速排序 (Quick Sort)

平均 O(n log n)，最坏 O(n²)

```typescript
function quickSort<T>(arr: T[], compare: (a: T, b: T) => number): T[] {
  if (arr.length <= 1) return arr;

  const pivot = arr[Math.floor(arr.length / 2)];
  const left = arr.filter((x) => compare(x, pivot) < 0);
  const middle = arr.filter((x) => compare(x, pivot) === 0);
  const right = arr.filter((x) => compare(x, pivot) > 0);

  return [...quickSort(left, compare), ...middle, ...quickSort(right, compare)];
}

// 使用
const nums = [38, 27, 43, 3, 9, 82, 10];
quickSort(nums, (a, b) => a - b); // [3, 9, 10, 27, 38, 43, 82]
```

### 归并排序 (Merge Sort)

稳定排序，O(n log n)

```typescript
function mergeSort<T>(arr: T[], compare: (a: T, b: T) => number): T[] {
  if (arr.length <= 1) return arr;

  const mid = Math.floor(arr.length / 2);
  const left = mergeSort(arr.slice(0, mid), compare);
  const right = mergeSort(arr.slice(mid), compare);

  return merge(left, right, compare);
}

function merge<T>(left: T[], right: T[], compare: (a: T, b: T) => number): T[] {
  const result: T[] = [];
  let i = 0, j = 0;

  while (i < left.length && j < right.length) {
    if (compare(left[i], right[j]) <= 0) {
      result.push(left[i++]);
    } else {
      result.push(right[j++]);
    }
  }

  return result.concat(left.slice(i), right.slice(j));
}
```

### 堆排序 (Heap Sort)

原地排序，O(n log n)

```typescript
function heapSort<T>(arr: T[], compare: (a: T, b: T) => number): T[] {
  const result = [...arr];
  const n = result.length;

  // 建堆
  for (let i = Math.floor(n / 2) - 1; i >= 0; i--) {
    heapify(result, n, i, compare);
  }

  // 逐个提取最大值
  for (let i = n - 1; i > 0; i--) {
    [result[0], result[i]] = [result[i], result[0]];
    heapify(result, i, 0, compare);
  }

  return result;
}

function heapify<T>(arr: T[], size: number, root: number, compare: (a: T, b: T) => number) {
  let largest = root;
  const left = 2 * root + 1;
  const right = 2 * root + 2;

  if (left < size && compare(arr[left], arr[largest]) > 0) largest = left;
  if (right < size && compare(arr[right], arr[largest]) > 0) largest = right;

  if (largest !== root) {
    [arr[root], arr[largest]] = [arr[largest], arr[root]];
    heapify(arr, size, largest, compare);
  }
}
```

## 2. 搜索算法

### 二分搜索

O(log n)，要求有序数组

```typescript
function binarySearch<T>(arr: T[], target: T, compare: (a: T, b: T) => number): number {
  let left = 0;
  let right = arr.length - 1;

  while (left <= right) {
    const mid = Math.floor((left + right) / 2);
    const cmp = compare(arr[mid], target);

    if (cmp === 0) return mid;
    if (cmp < 0) left = mid + 1;
    else right = mid - 1;
  }

  return -1; // 未找到
}

// 使用
const sorted = [1, 3, 5, 7, 9, 11, 13];
binarySearch(sorted, 7, (a, b) => a - b); // 3
```

## 3. 链表

```typescript
class ListNode<T> {
  constructor(
    public value: T,
    public next: ListNode<T> | null = null
  ) {}
}

class LinkedList<T> {
  head: ListNode<T> | null = null;
  private size = 0;

  append(value: T): void {
    const node = new ListNode(value);
    if (!this.head) {
      this.head = node;
    } else {
      let curr = this.head;
      while (curr.next) curr = curr.next;
      curr.next = node;
    }
    this.size++;
  }

  prepend(value: T): void {
    const node = new ListNode(value, this.head);
    this.head = node;
    this.size++;
  }

  delete(value: T): boolean {
    if (!this.head) return false;
    if (this.head.value === value) {
      this.head = this.head.next;
      this.size--;
      return true;
    }
    let curr = this.head;
    while (curr.next) {
      if (curr.next.value === value) {
        curr.next = curr.next.next;
        this.size--;
        return true;
      }
      curr = curr.next;
    }
    return false;
  }

  find(value: T): ListNode<T> | null {
    let curr = this.head;
    while (curr) {
      if (curr.value === value) return curr;
      curr = curr.next;
    }
    return null;
  }

  toArray(): T[] {
    const result: T[] = [];
    let curr = this.head;
    while (curr) {
      result.push(curr.value);
      curr = curr.next;
    }
    return result;
  }

  get length(): number { return this.size; }
}
```

## 4. 树

### 二叉搜索树 (BST)

```typescript
class TreeNode<T> {
  left: TreeNode<T> | null = null;
  right: TreeNode<T> | null = null;

  constructor(public value: T) {}
}

class BinarySearchTree<T> {
  root: TreeNode<T> | null = null;

  constructor(private compare: (a: T, b: T) => number) {}

  insert(value: T): void {
    this.root = this.insertNode(this.root, value);
  }

  private insertNode(node: TreeNode<T> | null, value: T): TreeNode<T> {
    if (!node) return new TreeNode(value);
    if (this.compare(value, node.value) < 0) {
      node.left = this.insertNode(node.left, value);
    } else {
      node.right = this.insertNode(node.right, value);
    }
    return node;
  }

  search(value: T): boolean {
    return this.searchNode(this.root, value);
  }

  private searchNode(node: TreeNode<T> | null, value: T): boolean {
    if (!node) return false;
    const cmp = this.compare(value, node.value);
    if (cmp === 0) return true;
    return cmp < 0
      ? this.searchNode(node.left, value)
      : this.searchNode(node.right, value);
  }

  // 中序遍历（有序输出）
  inorder(): T[] {
    const result: T[] = [];
    const traverse = (node: TreeNode<T> | null) => {
      if (!node) return;
      traverse(node.left);
      result.push(node.value);
      traverse(node.right);
    };
    traverse(this.root);
    return result;
  }

  // 层序遍历
  levelOrder(): T[] {
    if (!this.root) return [];
    const result: T[] = [];
    const queue: TreeNode<T>[] = [this.root];

    while (queue.length > 0) {
      const node = queue.shift()!;
      result.push(node.value);
      if (node.left) queue.push(node.left);
      if (node.right) queue.push(node.right);
    }

    return result;
  }
}

// 使用
const bst = new BinarySearchTree<number>((a, b) => a - b);
[5, 3, 7, 1, 4, 6, 8].forEach((v) => bst.insert(v));
bst.inorder();    // [1, 3, 4, 5, 6, 7, 8]
bst.search(4);    // true
bst.search(10);   // false
```

## 5. 图

```typescript
class Graph<T> {
  private adjacencyList = new Map<T, Set<T>>();

  addVertex(vertex: T): void {
    if (!this.adjacencyList.has(vertex)) {
      this.adjacencyList.set(vertex, new Set());
    }
  }

  addEdge(from: T, to: T): void {
    this.addVertex(from);
    this.addVertex(to);
    this.adjacencyList.get(from)!.add(to);
    this.adjacencyList.get(to)!.add(from); // 无向图
  }

  // 广度优先搜索 (BFS)
  bfs(start: T): T[] {
    const visited = new Set<T>();
    const result: T[] = [];
    const queue: T[] = [start];
    visited.add(start);

    while (queue.length > 0) {
      const vertex = queue.shift()!;
      result.push(vertex);

      for (const neighbor of this.adjacencyList.get(vertex) ?? []) {
        if (!visited.has(neighbor)) {
          visited.add(neighbor);
          queue.push(neighbor);
        }
      }
    }

    return result;
  }

  // 深度优先搜索 (DFS)
  dfs(start: T): T[] {
    const visited = new Set<T>();
    const result: T[] = [];

    const traverse = (vertex: T) => {
      visited.add(vertex);
      result.push(vertex);
      for (const neighbor of this.adjacencyList.get(vertex) ?? []) {
        if (!visited.has(neighbor)) traverse(neighbor);
      }
    };

    traverse(start);
    return result;
  }

  // 最短路径 (BFS-based, 无权图)
  shortestPath(start: T, end: T): T[] | null {
    const visited = new Set<T>();
    const parent = new Map<T, T | null>();
    const queue: T[] = [start];
    visited.add(start);
    parent.set(start, null);

    while (queue.length > 0) {
      const vertex = queue.shift()!;
      if (vertex === end) {
        const path: T[] = [];
        let curr: T | null = end;
        while (curr !== null) {
          path.unshift(curr);
          curr = parent.get(curr) ?? null;
        }
        return path;
      }

      for (const neighbor of this.adjacencyList.get(vertex) ?? []) {
        if (!visited.has(neighbor)) {
          visited.add(neighbor);
          parent.set(neighbor, vertex);
          queue.push(neighbor);
        }
      }
    }

    return null; // 不可达
  }
}

// 使用
const graph = new Graph<string>();
graph.addEdge("A", "B");
graph.addEdge("A", "C");
graph.addEdge("B", "D");
graph.addEdge("C", "D");
graph.addEdge("D", "E");

graph.bfs("A"); // ["A", "B", "C", "D", "E"]
graph.dfs("A"); // ["A", "B", "D", "E", "C"] (可能)
graph.shortestPath("A", "E"); // ["A", "B", "D", "E"]
```

## 6. 常用数据结构对比

| 数据结构 | 访问 | 搜索 | 插入 | 删除 | 空间 |
|---------|------|------|------|------|------|
| 数组 | O(1) | O(n) | O(n) | O(n) | O(n) |
| 链表 | O(n) | O(n) | O(1) | O(1) | O(n) |
| 哈希表 | - | O(1) | O(1) | O(1) | O(n) |
| BST（平衡） | O(log n) | O(log n) | O(log n) | O(log n) | O(n) |
| 堆 | - | O(n) | O(log n) | O(log n) | O(n) |

## 7. 算法复杂度速查

| 算法 | 最好 | 平均 | 最坏 | 稳定 |
|------|------|------|------|------|
| 冒泡排序 | O(n) | O(n²) | O(n²) | ✅ |
| 快速排序 | O(n log n) | O(n log n) | O(n²) | ❌ |
| 归并排序 | O(n log n) | O(n log n) | O(n log n) | ✅ |
| 堆排序 | O(n log n) | O(n log n) | O(n log n) | ❌ |
| 二分搜索 | O(1) | O(log n) | O(log n) | - |

## 最佳实践

- 选择数据结构时，优先考虑操作频率最高的操作类型
- 排序算法：小数组用插入排序，大数组用快速排序/归并排序
- 搜索：有序数据用二分，否则用哈希表
- 图遍历：BFS 求最短路径，DFS 求连通性
- 面试常见：双指针、滑动窗口、递归分治
