# 数据库性能调优

## 概述

数据库性能调优是一个系统工程，涵盖查询优化、索引调优、配置优化和监控诊断等多个方面。本指南提供实用的调优方法和最佳实践。

## 1. 查询优化

### 1.1 EXPLAIN分析

```sql
-- PostgreSQL EXPLAIN
EXPLAIN ANALYZE
SELECT u.name, COUNT(o.id) as order_count
FROM users u
LEFT JOIN orders o ON u.id = o.user_id
WHERE u.created_at > '2024-01-01'
GROUP BY u.id, u.name
ORDER BY order_count DESC
LIMIT 10;

-- 输出解读
-- Seq Scan on users (cost=0.00..1234.56 rows=1000 width=48)
-- -> Hash Join (cost=100.00..5678.90 rows=5000 width=56)
-- -> Index Scan using idx_orders_user on orders

-- MySQL EXPLAIN
EXPLAIN FORMAT=JSON
SELECT * FROM users WHERE email = 'test@example.com';

-- 查看实际执行计划
EXPLAIN ANALYZE
SELECT * FROM orders WHERE user_id = 123 AND status = 'pending';
```

### 1.2 常见查询问题

```sql
-- 问题1：SELECT * 全表扫描
-- 错误
SELECT * FROM users WHERE status = 'active';

-- 正确：只选择需要的列
SELECT id, name, email FROM users WHERE status = 'active';

-- 问题2：隐式类型转换
-- 错误：phone是VARCHAR，但传入数字
SELECT * FROM users WHERE phone = 13800138000;

-- 正确：使用正确类型
SELECT * FROM users WHERE phone = '13800138000';

-- 问题3：函数导致索引失效
-- 错误
SELECT * FROM users WHERE YEAR(created_at) = 2024;

-- 正确：使用范围查询
SELECT * FROM users 
WHERE created_at >= '2024-01-01' AND created_at < '2025-01-01';

-- 问题4：OR导致全表扫描
-- 错误
SELECT * FROM users WHERE status = 'active' OR status = 'pending';

-- 正确：使用IN
SELECT * FROM users WHERE status IN ('active', 'pending');

-- 问题5：子查询效率低
-- 错误
SELECT * FROM users WHERE id IN (SELECT user_id FROM orders);

-- 正确：使用JOIN
SELECT DISTINCT u.* FROM users u
JOIN orders o ON u.id = o.user_id;
```

### 1.3 分页优化

```sql
-- 问题：OFFSET大偏移量性能差
SELECT * FROM products ORDER BY id LIMIT 10 OFFSET 100000;

-- 方案1：使用游标分页（Keyset Pagination）
SELECT * FROM products 
WHERE id > 100000 
ORDER BY id 
LIMIT 10;

-- 方案2：延迟关联
SELECT p.* FROM products p
INNER JOIN (
  SELECT id FROM products ORDER BY id LIMIT 10 OFFSET 100000
) AS tmp ON p.id = tmp.id;

-- 方案3：覆盖索引分页
SELECT id, name, price FROM products 
ORDER BY id 
LIMIT 10 OFFSET 100000;
-- 确保有覆盖索引：CREATE INDEX idx_products_cover ON products(id, name, price);
```

### 1.4 批量操作优化

```sql
-- 批量插入
-- 错误：逐条插入
INSERT INTO logs (message, level) VALUES ('msg1', 'INFO');
INSERT INTO logs (message, level) VALUES ('msg2', 'INFO');

-- 正确：批量插入
INSERT INTO logs (message, level) VALUES 
('msg1', 'INFO'),
('msg2', 'INFO'),
('msg3', 'INFO'),
('msg4', 'INFO');

-- 批量更新
-- 使用CASE WHEN
UPDATE products SET price = CASE
  WHEN id = 1 THEN 99.99
  WHEN id = 2 THEN 149.99
  WHEN id = 3 THEN 199.99
END
WHERE id IN (1, 2, 3);

-- 批量删除
-- 分批删除避免锁表
DELETE FROM logs WHERE created_at < '2023-01-01' LIMIT 10000;
-- 循环执行直到没有更多记录
```

## 2. 索引调优

### 2.1 索引设计原则

```sql
-- 原则1：最左前缀匹配
-- 复合索引 (a, b, c)
-- 可以使用：a | a,b | a,b,c
-- 不能使用：b | c | b,c
CREATE INDEX idx_orders_user_status_date 
ON orders(user_id, status, created_at);

-- 可以利用索引的查询
SELECT * FROM orders WHERE user_id = 1;
SELECT * FROM orders WHERE user_id = 1 AND status = 'pending';
SELECT * FROM orders WHERE user_id = 1 AND status = 'pending' AND created_at > '2024-01-01';

-- 原则2：选择性高的列放前面
-- user_id 选择性高，status 选择性低
CREATE INDEX idx_orders_user_status ON orders(user_id, status);

-- 原则3：覆盖索引减少回表
CREATE INDEX idx_users_email_covering 
ON users(email) INCLUDE (name, phone);

-- 查询直接从索引获取数据
SELECT name, phone FROM users WHERE email = 'test@example.com';
```

### 2.2 索引监控

```sql
-- PostgreSQL：查看索引使用情况
SELECT 
    schemaname,
    tablename,
    indexname,
    idx_scan as scans,
    idx_tup_read as tuples_read,
    idx_tup_fetch as tuples_fetched,
    pg_size_pretty(pg_relation_size(indexrelid)) as index_size
FROM pg_stat_user_indexes
ORDER BY idx_scan DESC;

-- 查找未使用的索引
SELECT 
    schemaname,
    tablename,
    indexname,
    idx_scan,
    pg_size_pretty(pg_relation_size(indexrelid)) as size
FROM pg_stat_user_indexes
WHERE idx_scan = 0
    AND indexrelid NOT IN (
        SELECT conindid FROM pg_constraint WHERE contype IN ('p', 'u')
    )
ORDER BY pg_relation_size(indexrelid) DESC;

-- MySQL：查看索引使用情况
SELECT * FROM sys.schema_unused_indexes;
SELECT * FROM sys.schema_redundant_indexes;

-- 查看索引基数
SHOW INDEX FROM users;
```

### 2.3 索引优化策略

```sql
-- 策略1：删除重复索引
-- 如果有 (a, b) 索引，则 (a) 索引通常是多余的
-- 检查重复索引
SELECT * FROM sys.schema_redundant_indexes;

-- 策略2：前缀索引优化长字符串
-- 错误：全列索引
CREATE INDEX idx_users_address ON users(address);

-- 正确：前缀索引
CREATE INDEX idx_users_address ON users(address(20));

-- 策略3：使用函数索引替代计算列
-- PostgreSQL
CREATE INDEX idx_users_lower_email ON users(LOWER(email));

-- 策略4：部分索引（PostgreSQL）
CREATE INDEX idx_orders_pending ON orders(created_at) 
WHERE status = 'pending';

-- 策略5：JSON字段索引（PostgreSQL）
CREATE INDEX idx_data_metadata ON data USING GIN(metadata);
CREATE INDEX idx_data_status ON data ((metadata->>'status'));
```

## 4. 监控诊断

### 4.1 性能监控指标

```sql
-- PostgreSQL：查看当前活动
SELECT 
    pid,
    usename,
    application_name,
    client_addr,
    backend_start,
    xact_start,
    query_start,
    state,
    query
FROM pg_stat_activity
WHERE state = 'active';

-- 查看锁信息
SELECT 
    blocked_locks.pid AS blocked_pid,
    blocked_activity.usename AS blocked_user,
    blocking_locks.pid AS blocking_pid,
    blocking_activity.usename AS blocking_user,
    blocked_activity.query AS blocked_statement,
    blocking_activity.query AS blocking_statement
FROM pg_catalog.pg_locks blocked_locks
JOIN pg_catalog.pg_stat_activity blocked_activity ON blocked_activity.pid = blocked_locks.pid
JOIN pg_catalog.pg_locks blocking_locks 
    ON blocking_locks.locktype = blocked_locks.locktype
    AND blocking_locks.relation = blocked_locks.relation
    AND blocking_locks.pid != blocked_locks.pid
JOIN pg_catalog.pg_stat_activity blocking_activity ON blocking_activity.pid = blocking_locks.pid
WHERE NOT blocked_locks.granted;

-- 表统计信息
SELECT 
    schemaname,
    relname,
    seq_scan,
    seq_tup_read,
    idx_scan,
    idx_tup_fetch,
    n_tup_ins,
    n_tup_upd,
    n_tup_del,
    n_live_tup,
    n_dead_tup
FROM pg_stat_user_tables
ORDER BY n_dead_tup DESC;
```

### 4.4 性能基线

```sql
-- 建立性能基线
CREATE TABLE performance_baseline (
    id SERIAL PRIMARY KEY,
    metric_name VARCHAR(100),
    metric_value DECIMAL(20,4),
    collected_at TIMESTAMP DEFAULT NOW()
);

-- 收集基线数据
INSERT INTO performance_baseline (metric_name, metric_value)
SELECT 'qps', 
       (SELECT SUM(xact_commit + xact_rollback) FROM pg_stat_database);

INSERT INTO performance_baseline (metric_name, metric_value)
SELECT 'active_connections',
       (SELECT COUNT(*) FROM pg_stat_activity WHERE state = 'active');

INSERT INTO performance_baseline (metric_name, metric_value)
SELECT 'cache_hit_ratio',
       (SELECT SUM(blks_hit) * 100.0 / SUM(blks_hit + blks_read) 
        FROM pg_stat_database WHERE blks_hit + blks_read > 0);
```

## 5. 高级调优技巧

### 5.1 连接池优化

```python
# Python SQLAlchemy连接池配置
from sqlalchemy import create_engine

engine = create_engine(
    'postgresql://user:pass@localhost/mydb',
    pool_size=20,           # 连接池大小
    max_overflow=10,        # 最大溢出连接
    pool_timeout=30,        # 获取连接超时
    pool_recycle=1800,      # 连接回收时间
    pool_pre_ping=True      # 连接健康检查
)

# 连接池监控
from sqlalchemy import event

@event.listens_for(engine, "checkout")
def on_checkout(dbapi_conn, connection_rec, connection_proxy):
    print(f"Connection checked out: {connection_rec}")

@event.listens_for(engine, "checkin")
def on_checkin(dbapi_conn, connection_rec):
    print(f"Connection checked in: {connection_rec}")
```

### 5.2 查询缓存策略

```python
# 应用层缓存（Redis）
import redis
import json

r = redis.Redis(host='localhost', port=6379, db=0)

def get_user_with_cache(user_id):
    cache_key = f"user:{user_id}"
    
    # 尝试从缓存获取
    cached = r.get(cache_key)
    if cached:
        return json.loads(cached)
    
    # 从数据库查询
    user = db.query("SELECT * FROM users WHERE id = %s", user_id)
    
    # 写入缓存（过期时间30分钟）
    r.setex(cache_key, 1800, json.dumps(user))
    
    return user

# 缓存失效策略
def update_user(user_id, data):
    # 更新数据库
    db.execute("UPDATE users SET ... WHERE id = %s", user_id)
    
    # 删除缓存（而非更新）
    r.delete(f"user:{user_id}")
```

### 5.3 读写分离

```python
# SQLAlchemy读写分离
from sqlalchemy import create_engine
from sqlalchemy.orm import sessionmaker

# 主库（写）
master_engine = create_engine('postgresql://user:pass@master/mydb')

# 从库（读）
slave_engine = create_engine('postgresql://user:pass@slave/mydb')

MasterSession = sessionmaker(bind=master_engine)
SlaveSession = sessionmaker(bind=slave_engine)

# 使用装饰器自动路由
def use_master(func):
    def wrapper(*args, **kwargs):
        session = MasterSession()
        try:
            result = func(session, *args, **kwargs)
            session.commit()
            return result
        except:
            session.rollback()
            raise
        finally:
            session.close()
    return wrapper

def use_slave(func):
    def wrapper(*args, **kwargs):
        session = SlaveSession()
        try:
            return func(session, *args, **kwargs)
        finally:
            session.close()
    return wrapper

@use_master
def create_user(session, user_data):
    user = User(**user_data)
    session.add(user)
    return user

@use_slave
def get_users(session):
    return session.query(User).all()
```
