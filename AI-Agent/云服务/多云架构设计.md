# 多云架构设计

## 什么是多云架构

多云架构是指使用两个或更多云服务提供商来运行业务工作负载的策略。这种架构可以避免供应商锁定、提高可用性、优化成本。

## 多云策略

### 1. 多云部署模式

```
┌─────────────────────────────────────────────────────────────┐
│                      多云架构模式                              │
├──────────────┬──────────────┬──────────────┬────────────────┤
│  Active-Active │  Active-Passive │  Best-of-Breed │  Cloud Bursting │
│   双活部署     │   主备部署       │   最优服务选择   │   弹性扩展       │
└──────────────┴──────────────┴──────────────┴────────────────┘
```

### 2. 多云策略选择

| 策略 | 优点 | 缺点 | 适用场景 |
|------|------|------|----------|
| Active-Active | 高可用、低延迟 | 复杂度高、成本高 | 核心业务、全球化应用 |
| Active-Passive | 成本适中、容灾能力强 | 切换延迟、资源闲置 | 非核心业务、合规要求 |
| Best-of-Breed | 利用各云最优服务 | 集成复杂、运维成本高 | 特定功能需求 |
| Cloud Bursting | 弹性扩展、成本优化 | 数据同步复杂 | 流量波动大的业务 |

## 数据同步

### 1. 数据同步架构

```python
import boto3
from aliyun_oss_sdk import OssClient
import asyncio
from datetime import datetime

class MultiCloudDataSync:
    """多云数据同步管理器"""

    def __init__(self):
        # AWS客户端
        self.s3_client = boto3.client('s3')
        self.dynamodb = boto3.resource('dynamodb')

        # 阿里云客户端
        self.oss_client = OssClient(
            access_key_id='your-key-id',
            access_key_secret='your-key-secret',
            endpoint='oss-cn-hangzhou.aliyuncs.com'
        )

    async def sync_s3_to_oss(self, s3_bucket, oss_bucket, prefix=''):
        """S3 -> OSS 数据同步"""
        try:
            # 列出S3对象
            response = self.s3_client.list_objects_v2(
                Bucket=s3_bucket,
                Prefix=prefix
            )

            for obj in response.get('Contents', []):
                key = obj['Key']

                # 检查是否需要同步
                if await self._need_sync(s3_bucket, oss_bucket, key):
                    # 从S3下载
                    s3_response = self.s3_client.get_object(
                        Bucket=s3_bucket,
                        Key=key
                    )
                    data = s3_response['Body'].read()

                    # 上传到OSS
                    self.oss_client.put_object(
                        bucket=oss_bucket,
                        key=key,
                        data=data
                    )

                    # 记录同步状态
                    await self._log_sync_status(
                        s3_bucket, oss_bucket, key, 'success'
                    )

                    print(f"Synced: {key}")

        except Exception as e:
            print(f"Sync failed: {e}")
            await self._log_sync_status(
                s3_bucket, oss_bucket, key, 'failed', str(e)
            )

    async def _need_sync(self, source_bucket, target_bucket, key):
        """检查是否需要同步"""
        try:
            # 获取源对象元数据
            source_head = self.s3_client.head_object(
                Bucket=source_bucket, Key=key
            )

            # 获取目标对象元数据
            target_head = self.oss_client.head_object(
                bucket=target_bucket, key=key
            )

            # 比较最后修改时间
            source_modified = source_head['LastModified']
            target_modified = target_head.last_modified

            return source_modified > target_modified

        except:
            # 目标不存在，需要同步
            return True

    async def _log_sync_status(self, source, target, key, status, error=None):
        """记录同步状态"""
        table = self.dynamodb.Table('sync_log')
        table.put_item(Item={
            'source': source,
            'target': target,
            'key': key,
            'status': status,
            'error': error,
            'timestamp': datetime.utcnow().isoformat()
        })
```

```python
import pymysql
import psycopg2
from sqlalchemy import create_engine

class DatabaseSyncManager:
    """跨云数据库同步"""

    def __init__(self):
        # AWS RDS连接
        self.aws_engine = create_engine(
            'mysql+pymysql://admin:password@aws-rds-endpoint:3306/mydb'
        )

        # 阿里云RDS连接
        self.ali_engine = create_engine(
            'mysql+pymysql://admin:password@aliyun-rds-endpoint:3306/mydb'
        )

    async def sync_table(self, table_name, strategy='incremental'):
        """同步表数据"""
        if strategy == 'incremental':
            await self._incremental_sync(table_name)
        elif strategy == 'full':
            await self._full_sync(table_name)

    async def _incremental_sync(self, table_name):
        """增量同步"""
        # 获取上次同步时间
        last_sync = await self._get_last_sync_time(table_name)

        # 查询增量数据
        query = f"""
        SELECT * FROM {table_name}
        WHERE updated_at > '{last_sync}'
        ORDER BY updated_at
        """

        with self.aws_engine.connect() as aws_conn:
            result = aws_conn.execute(query)
            records = result.fetchall()

        # 写入目标数据库
        if records:
            with self.ali_engine.connect() as ali_conn:
                for record in records:
                    # UPSERT操作
                    upsert_sql = f"""
                    INSERT INTO {table_name} ({','.join(record.keys())})
                    VALUES ({','.join(['%s'] * len(record))})
                    ON DUPLICATE KEY UPDATE
                    {','.join([f'{k}=VALUES({k})' for k in record.keys()])}
                    """
                    ali_conn.execute(upsert_sql, tuple(record.values()))

        # 更新同步时间
        await self._update_last_sync_time(table_name)
```

## 成本优化

### 1. 多云成本分析

```python
from dataclasses import dataclass
from typing import Dict, List
from datetime import datetime, timedelta

@dataclass
class CloudCost:
    provider: str
    service: str
    cost: float
    period: str
    resource_id: str

class MultiCloudCostAnalyzer:
    """多云成本分析器"""

    def __init__(self):
        self.costs: List[CloudCost] = []

    def collect_costs(self):
        """收集各云平台成本"""
        # AWS成本
        aws_costs = self._get_aws_costs()
        self.costs.extend(aws_costs)

        # 阿里云成本
        ali_costs = self._get_aliyun_costs()
        self.costs.extend(ali_costs)

    def analyze(self) -> Dict:
        """分析成本分布"""
        analysis = {
            'by_provider': {},
            'by_service': {},
            'by_month': {},
            'recommendations': []
        }

        # 按提供商统计
        for cost in self.costs:
            if cost.provider not in analysis['by_provider']:
                analysis['by_provider'][cost.provider] = 0
            analysis['by_provider'][cost.provider] += cost.cost

        # 按服务统计
        for cost in self.costs:
            service_key = f"{cost.provider}/{cost.service}"
            if service_key not in analysis['by_service']:
                analysis['by_service'][service_key] = 0
            analysis['by_service'][service_key] += cost.cost

        # 生成优化建议
        analysis['recommendations'] = self._generate_recommendations()

        return analysis

    def _generate_recommendations(self) -> List[str]:
        """生成成本优化建议"""
        recommendations = []

        # 分析计算资源利用率
        compute_costs = [
            c for c in self.costs
            if c.service in ['EC2', 'ECS', 'Lambda', '函数计算']
        ]

        total_compute = sum(c.cost for c in compute_costs)

        if total_compute > 10000:  # 月计算成本超过1万
            recommendations.append(
                "考虑使用预留实例或包年包月降低计算成本"
            )

        # 分析存储成本
        storage_costs = [
            c for c in self.costs
            if c.service in ['S3', 'OSS', 'EBS']
        ]

        total_storage = sum(c.cost for c in storage_costs)

        if total_storage > 5000:
            recommendations.append(
                "检查存储生命周期策略，将冷数据迁移到低成本存储"
            )

        return recommendations

    def compare_prices(self, service_type, specs):
        """跨云价格比较"""
        prices = {}

        # AWS价格
        aws_price = self._get_aws_price(service_type, specs)
        prices['AWS'] = aws_price

        # 阿里云价格
        ali_price = self._get_aliyun_price(service_type, specs)
        prices['阿里云'] = ali_price

        # 找出最优价格
        best_provider = min(prices, key=prices.get)
        best_price = prices[best_provider]

        return {
            'prices': prices,
            'best_provider': best_provider,
            'best_price': best_price,
            'savings': {
                p: ((price - best_price) / price * 100)
                for p, price in prices.items()
                if p != best_provider
            }
        }
```

### 2. 成本优化策略

| 策略 | AWS实现 | 阿里云实现 | 节省比例 |
|------|---------|-----------|----------|
| 预留实例 | Reserved Instances | 包年包月 | 30-60% |
| 竞价实例 | Spot Instances | 抢占式实例 | 60-90% |
| 存储分层 | S3 Lifecycle | OSS生命周期 | 40-60% |

### 2. 灾备切换流程

```python
import asyncio
from enum import Enum
from typing import Optional

class SiteStatus(Enum):
    ACTIVE = "active"
    STANDBY = "standby"
    FAILED = "failed"
    MAINTENANCE = "maintenance"

class DisasterRecoveryManager:
    """灾备管理器"""

    def __init__(self):
        self.sites = {
            'primary': {
                'provider': 'AWS',
                'region': 'ap-northeast-1',
                'status': SiteStatus.ACTIVE,
                'endpoint': 'https://primary.example.com'
            },
            'dr': {
                'provider': '阿里云',
                'region': 'cn-hangzhou',
                'status': SiteStatus.STANDBY,
                'endpoint': 'https://dr.example.com'
            }
        }

    async def health_check(self):
        """健康检查"""
        for site_name, site in self.sites.items():
            try:
                # 执行健康检查
                is_healthy = await self._check_endpoint(site['endpoint'])

                if not is_healthy:
                    site['status'] = SiteStatus.FAILED
                    await self._alert(f"{site_name} site is unhealthy!")

                    # 如果主站点故障，触发切换
                    if site_name == 'primary':
                        await self.failover()

            except Exception as e:
                print(f"Health check failed for {site_name}: {e}")

    async def failover(self):
        """故障切换"""
        print("Starting failover process...")

        primary = self.sites['primary']
        dr = self.sites['dr']

        if dr['status'] != SiteStatus.STANDBY:
            raise Exception("DR site is not in standby mode")

        # 步骤1: 更新DNS
        await self._update_dns(
            primary_endpoint=primary['endpoint'],
            dr_endpoint=dr['endpoint']
        )

        # 步骤2: 启动DR站点应用
        await self._start_dr_applications()

        # 步骤3: 切换数据库为主
        await self._promote_dr_database()

        # 步骤4: 更新状态
        primary['status'] = SiteStatus.FAILED
        dr['status'] = SiteStatus.ACTIVE

        print("Failover completed!")

    async def failback(self):
        """故障恢复"""
        print("Starting failback process...")

        # 恢复主站点
        await self._restore_primary()

        # 同步数据
        await self._sync_data()

        # 切换回主站点
        await self.failover()

    async def _start_dr_applications(self):
        """启动DR站点应用"""

        pass
```

## 最佳实践

### 1. 架构设计原则

- ✅ **服务抽象化**：使用Terraform/Pulumi等IaC工具统一管理
- ✅ **数据一致性**：选择合适的数据同步策略
- ✅ **网络互通**：使用VPN或专线连接多云环境
- ✅ **安全统一**：统一身份认证和权限管理
- ✅ **监控统一**：使用统一监控平台（如Datadog、Grafana）

### 2. 常见陷阱

- ❌ 过度依赖单一云厂商的专有服务
- ❌ 忽视数据传输成本
- ❌ 缺乏统一的运维工具链
- ❌ 低估多云管理的复杂度
