# 云成本优化

## 什么是云成本优化

云成本优化是通过策略性地管理云资源使用，在保证性能和可靠性的同时，最大化云投入的价值。这是一种持续的过程，而非一次性任务。

### 2. 资源右调整（Right-Sizing）

```python
from dataclasses import dataclass
from typing import List, Dict
from datetime import datetime, timedelta

@dataclass
class ResourceMetrics:
    resource_id: str
    resource_type: str
    instance_type: str
    cpu_utilization: float  # 百分比
    memory_utilization: float  # 百分比
    network_in: float  # GB
    network_out: float  # GB
    period_days: int

class RightSizingAnalyzer:
    """资源右调整分析器"""

    # 推荐阈值
    CPU_THRESHOLD_LOW = 20  # CPU利用率低于20%
    CPU_THRESHOLD_HIGH = 80  # CPU利用率高于80%
    MEMORY_THRESHOLD_LOW = 30  # 内存利用率低于30%
    MEMORY_THRESHOLD_HIGH = 80  # 内存利用率高于80%

    def analyze(self, metrics: List[ResourceMetrics]) -> Dict:
        """分析资源使用情况并生成推荐"""
        recommendations = []

        for metric in metrics:
            recommendation = self._analyze_single(metric)
            if recommendation:
                recommendations.append(recommendation)

        return {
            'total_resources': len(metrics),
            'resources_to_downsize': len([r for r in recommendations if r['action'] == 'downsize']),
            'resources_to_upsize': len([r for r in recommendations if r['action'] == 'upsize']),
            'recommendations': recommendations,
            'estimated_savings': self._calculate_savings(recommendations)
        }

    def _analyze_single(self, metrics: ResourceMetrics) -> Dict:
        """分析单个资源"""
        issues = []

        # CPU分析
        if metrics.cpu_utilization < self.CPU_THRESHOLD_LOW:
            issues.append({
                'metric': 'CPU',
                'current': f"{metrics.cpu_utilization}%",
                'threshold': f"{self.CPU_THRESHOLD_LOW}%",
                'action': 'underutilized'
            })
        elif metrics.cpu_utilization > self.CPU_THRESHOLD_HIGH:
            issues.append({
                'metric': 'CPU',
                'current': f"{metrics.cpu_utilization}%",
                'threshold': f"{self.CPU_THRESHOLD_HIGH}%",
                'action': 'overutilized'
            })

        # 内存分析
        if metrics.memory_utilization < self.MEMORY_THRESHOLD_LOW:
            issues.append({
                'metric': 'Memory',
                'current': f"{metrics.memory_utilization}%",
                'threshold': f"{self.MEMORY_THRESHOLD_LOW}%",
                'action': 'underutilized'
            })

        if not issues:
            return None

        # 确定操作
        is_underutilized = any(i['action'] == 'underutilized' for i in issues)
        action = 'downsize' if is_underutilized else 'upsize'

        # 推荐新实例类型
        recommended_type = self._recommend_instance_type(
            metrics.instance_type, action, metrics
        )

        return {
            'resource_id': metrics.resource_id,
            'resource_type': metrics.resource_type,
            'current_type': metrics.instance_type,
            'recommended_type': recommended_type,
            'action': action,
            'issues': issues,
            'confidence': self._calculate_confidence(metrics)
        }

    def _recommend_instance_type(self, current_type: str, action: str, metrics: ResourceMetrics) -> str:
        """推荐新的实例类型"""
        # 简化的推荐逻辑，实际应用中需要更复杂的映射表
        instance_families = {
            't3': ['t3.micro', 't3.small', 't3.medium', 't3.large', 't3.xlarge'],
            'm6i': ['m6i.large', 'm6i.xlarge', 'm6i.2xlarge', 'm6i.4xlarge'],
            'c6i': ['c6i.large', 'c6i.xlarge', 'c6i.2xlarge', 'c6i.4xlarge'],
        }

        # 解析当前实例类型
        parts = current_type.split('.')
        family = parts[0]
        size = parts[1] if len(parts) > 1 else 'medium'

        size_order = ['micro', 'small', 'medium', 'large', 'xlarge', '2xlarge', '4xlarge']

        if family in instance_families:
            sizes = instance_families[family]
            current_idx = sizes.index(current_type) if current_type in sizes else 2

            if action == 'downsize':
                new_idx = max(0, current_idx - 1)
            else:
                new_idx = min(len(sizes) - 1, current_idx + 1)

            return sizes[new_idx]

        return current_type

    def _calculate_confidence(self, metrics: ResourceMetrics) -> str:
        """计算推荐置信度"""
        if metrics.period_days >= 30:
            return 'high'
        elif metrics.period_days >= 14:
            return 'medium'
        return 'low'

    def _calculate_savings(self, recommendations: List[Dict]) -> float:
        """估算节省金额"""
        # 简化的节省计算，实际需要查询具体价格
        savings_per_recommendation = {
            't3.micro -> t3.small': -5,
            't3.small -> t3.micro': 5,
            't3.medium -> t3.small': 15,
            't3.large -> t3.medium': 30,
            'm6i.xlarge -> m6i.large': 50,
            'm6i.2xlarge -> m6i.xlarge': 100,
        }

        total_savings = 0
        for rec in recommendations:
            key = f"{rec['current_type']} -> {rec['recommended_type']}"
            if key in savings_per_recommendation:
                total_savings += savings_per_recommendation[key]

        return total_savings
```

### 1. 预留实例策略

| 购买选项 | 折扣率 | 灵活性 | 适用场景 |
|----------|--------|--------|----------|
| 全额预付 | 最高（~60%） | 低 | 稳定工作负载 |
| 部分预付 | 中等（~40%） | 中 | 可预测工作负载 |
| 无预付 | 较低（~30%） | 中 | 预算有限 |
| Savings Plans | 中等（~40%） | 高 | 多种实例类型 |

### 2. AWS Reserved Instances vs Savings Plans

```python
from dataclasses import dataclass
from typing import List
from datetime import datetime, timedelta
import numpy as np

@dataclass
class UsagePattern:
    instance_type: str
    hours_per_month: float
    region: str
    platform: str

class ReservedInstanceCalculator:
    """预留实例计算器"""

    # 按需价格（每小时）
    ON_DEMAND_PRICES = {
        'm6i.large': 0.096,
        'm6i.xlarge': 0.192,
        'm6i.2xlarge': 0.384,
        'c6i.large': 0.085,
        'c6i.xlarge': 0.17,
        'r6i.large': 0.126,
        'r6i.xlarge': 0.252,
    }

    # 预留实例折扣（1年全预付）
    RI_DISCOUNTS = {
        'm6i.large': 0.40,
        'm6i.xlarge': 0.42,
        'm6i.2xlarge': 0.44,
        'c6i.large': 0.38,
        'c6i.xlarge': 0.40,
        'r6i.large': 0.41,
        'r6i.xlarge': 0.43,
    }

    def calculate_roi(self, usage: UsagePattern, term_years: int = 1) -> dict:
        """计算预留实例投资回报"""
        # 按需成本
        on_demand_monthly = (
            self.ON_DEMAND_PRICES.get(usage.instance_type, 0.1)
            * usage.hours_per_month
        )
        on_demand_annual = on_demand_monthly * 12

        # 预留实例成本
        ri_discount = self.RI_DISCOUNTS.get(usage.instance_type, 0.40)
        ri_annual = on_demand_annual * (1 - ri_discount)

        # 计算节省
        annual_savings = on_demand_annual - ri_annual
        total_savings = annual_savings * term_years

        return {
            'instance_type': usage.instance_type,
            'term_years': term_years,
            'on_demand_annual': f"${on_demand_annual:.2f}",
            'ri_annual': f"${ri_annual:.2f}",
            'annual_savings': f"${annual_savings:.2f}",
            'total_savings': f"${total_savings:.2f}",
            'roi_percentage': f"{ri_discount * 100:.1f}%",
            'breakeven_months': self._calculate_breakeven(
                on_demand_monthly, ri_annual / 12
            )
        }

    def _calculate_breakeven(self, on_demand_monthly: float, ri_monthly: float) -> int:
        """计算回本周期"""
        # 简化计算：假设全预付
        upfront = ri_monthly * 12  # 年付金额
        monthly_savings = on_demand_monthly - ri_monthly

        if monthly_savings <= 0:
            return -1

        return int(upfront / monthly_savings)

    def recommend_strategy(self, usage_patterns: List[UsagePattern]) -> dict:
        """推荐购买策略"""
        recommendations = []

        for usage in usage_patterns:
            # 计算使用率
            utilization = usage.hours_per_month / (24 * 30)  # 假设30天

            if utilization >= 0.75:
                # 高使用率，推荐预留实例
                roi = self.calculate_roi(usage)
                recommendations.append({
                    'instance_type': usage.instance_type,
                    'recommendation': 'Reserved Instance',
                    'reason': f'高使用率({utilization:.0%})',
                    'estimated_savings': roi['annual_savings']
                })
            elif utilization >= 0.40:
                # 中等使用率，推荐Savings Plans
                recommendations.append({
                    'instance_type': usage.instance_type,
                    'recommendation': 'Savings Plans',
                    'reason': f'中等使用率({utilization:.0%})，灵活性更好',
                    'estimated_savings': '10-20%'
                })
            else:
                # 低使用率，推荐按需或Spot
                recommendations.append({
                    'instance_type': usage.instance_type,
                    'recommendation': 'On-Demand or Spot',
                    'reason': f'低使用率({utilization:.0%})',
                    'estimated_savings': 'N/A'
                })

        return {
            'recommendations': recommendations,
            'total_potential_savings': self._calculate_total_savings(recommendations)
        }

    def _calculate_total_savings(self, recommendations: List[dict]) -> str:
        """计算总潜在节省"""
        # 简化计算
        return "15-25%"
```

### 1. Spot实例策略

```python
from dataclasses import dataclass
from typing import List, Optional
from datetime import datetime
import boto3

@dataclass
class SpotInstanceConfig:
    instance_type: str
    max_price: float
    availability_zone: str
    ami_id: str
    key_name: str
    security_group_ids: List[str]

class SpotInstanceManager:
    """Spot实例管理器"""

    def __init__(self):
        self.ec2 = boto3.client('ec2')

    def get_spot_price_history(self, instance_types: List[str], hours: int = 24) -> dict:
        """获取Spot价格历史"""
        start_time = datetime.utcnow() - timedelta(hours=hours)

        response = self.ec2.describe_spot_price_history(
            InstanceTypes=instance_types,
            StartTime=start_time,
            ProductDescriptions=['Linux/UNIX'],
            MaxResults=100
        )

        prices = {}
        for item in response['SpotPriceHistory']:
            key = f"{item['InstanceType']}-{item['AvailabilityZone']}"
            if key not in prices:
                prices[key] = []
            prices[key].append({
                'timestamp': item['Timestamp'].isoformat(),
                'price': float(item['SpotPrice'])
            })

        return prices

    def find_cheapest_az(self, instance_type: str) -> dict:
        """查找最便宜的可用区"""
        prices = self.get_spot_price_history([instance_type])
        cheapest = None

        for key, price_list in prices.items():
            if instance_type in key:
                avg_price = sum(p['price'] for p in price_list) / len(price_list)
                if cheapest is None or avg_price < cheapest['avg_price']:
                    cheapest = {
                        'availability_zone': key.split('-')[-1],
                        'avg_price': avg_price,
                        'current_price': price_list[-1]['price'] if price_list else 0
                    }

        return cheapest

    def create_spot_fleet(self, configs: List[SpotInstanceConfig], target_capacity: int) -> str:
        """创建Spot Fleet"""
        launch_specs = []

        for config in configs:
            launch_specs.append({
                'InstanceType': config.instance_type,
                'ImageId': config.ami_id,
                'KeyName': config.key_name,
                'SecurityGroupIds': config.security_group_ids,
                'SpotPrice': str(config.max_price),
                'SubnetId': f"subnet-{config.availability_zone}",
                'WeightedCapacity': 1.0
            })

        response = self.ec2.request_spot_fleet(
            SpotFleetRequestConfig={
                'IamFleetRole': 'arn:aws:iam::123456789012:role/aws-ec2-spot-fleet-role',
                'TargetCapacity': target_capacity,
                'AllocationStrategy': 'lowestPrice',
                'LaunchSpecifications': launch_specs,
                'TerminateInstancesWithExpiration': True,
                'Type': 'maintain'
            }
        )

        return response['SpotFleetRequestId']

    def handle_interruption(self, instance_id: str):
        """处理Spot中断"""
        # 1. 保存状态
        self._save_instance_state(instance_id)

        # 2. 启动新实例
        new_instance = self._launch_replacement(instance_id)

        # 3. 切换流量
        self._switch_traffic(instance_id, new_instance)

        # 4. 终止旧实例
        self.ec2.terminate_instances(InstanceIds=[instance_id])

        return new_instance
```

```python
from dataclasses import dataclass
from typing import Dict, List
from datetime import datetime, timedelta
import boto3

@dataclass
class CostAlert:
    threshold: float
    comparison: str  # 'greater_than', 'less_than'
    period: str  # 'daily', 'weekly', 'monthly'
    notification_channels: List[str]

class CostMonitor:
    """云成本监控器"""

    def __init__(self):
        self.ce = boto3.client('ce')  # Cost Explorer
        self.cloudwatch = boto3.client('cloudwatch')
        self.sns = boto3.client('sns')

    def get_cost_and_usage(self, start_date: str, end_date: str, granularity: str = 'MONTHLY') -> dict:
        """获取成本和使用数据"""
        response = self.ce.get_cost_and_usage(
            TimePeriod={
                'Start': start_date,
                'End': end_date
            },
            Granularity=granularity,
            Metrics=['UnblendedCost', 'UsageQuantity'],
            GroupBy=[
                {'Type': 'DIMENSION', 'Key': 'SERVICE'},
                {'Type': 'DIMENSION', 'Key': 'REGION'}
            ]
        )

        return response['ResultsByTime']

    def get_cost_forecast(self, days_ahead: int = 30) -> dict:
        """成本预测"""
        start_date = datetime.utcnow().date()
        end_date = start_date + timedelta(days=days_ahead)

        response = self.ce.get_cost_forecast(
            TimePeriod={
                'Start': start_date.isoformat(),
                'End': end_date.isoformat()
            },
            Metric='UNBLENDED_COST',
            Granularity='MONTHLY'
        )

        return {
            'forecast_period': f"{start_date} to {end_date}",
            'estimated_cost': response['Total']['Amount'],
            'currency': response['Total']['Unit']
        }

    def check_budget_alerts(self, budget_name: str, current_spend: float) -> dict:
        """检查预算告警"""
        # 获取预算信息
        budgets = boto3.client('budgets')

        response = budgets.describe_budget(
            AccountId='123456789012',
            BudgetName=budget_name
        )

        budget = response['Budget']
        budget_limit = float(budget['BudgetLimit']['Amount'])

        utilization = (current_spend / budget_limit) * 100

        alerts = []
        if utilization >= 100:
            alerts.append({
                'level': 'critical',
                'message': f'预算已超支！当前支出: ${current_spend:.2f}, 预算: ${budget_limit:.2f}'
            })
        elif utilization >= 80:
            alerts.append({
                'level': 'warning',
                'message': f'预算使用已达{utilization:.0f}%，请注意控制成本'
            })

        return {
            'budget_name': budget_name,
            'budget_limit': budget_limit,
            'current_spend': current_spend,
            'utilization': f"{utilization:.1f}%",
            'alerts': alerts
        }

    def create_cost_dashboard(self):
        """创建成本监控仪表板"""
        dashboard_body = {
            'widgets': [
                {
                    'type': 'metric',
                    'properties': {
                        'metrics': [
                            ['AWS/Billing', 'EstimatedCharges', 'Currency', 'USD']
                        ],
                        'period': 86400,
                        'stat': 'Maximum',
                        'title': 'Daily Estimated Charges'
                    }
                },
                {
                    'type': 'metric',
                    'properties': {
                        'metrics': [
                            ['AWS/EC2', 'CPUUtilization', 'InstanceId', 'i-1234567890abcdef0']
                        ],
                        'period': 300,
                        'stat': 'Average',
                        'title': 'EC2 CPU Utilization'
                    }
                }
            ]
        }

        self.cloudwatch.put_dashboard(
            DashboardName='CostDashboard',
            DashboardBody=str(dashboard_body)
        )
```

## 月度成本优化检查清单

## 最佳实践

### 1. 成本优化原则

- ✅ **可见性优先**：建立完善的成本监控和报告机制
- ✅ **责任到人**：将成本分配到具体团队或项目
- ✅ **持续优化**：定期审查和优化资源使用
- ✅ **自动化**：使用工具自动化成本优化流程

### 2. 常见成本陷阱

- ❌ 忘记释放未使用的资源
- ❌ 过度配置资源
- ❌ 忽视数据传输成本
- ❌ 不合理的存储策略
- ❌ 缺乏成本预警机制
