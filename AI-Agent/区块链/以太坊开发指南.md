## 1. Solidity 语言核心

### 1.1 数据类型与变量

```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.24;

contract SolidityBasics {
    // ===== 状态变量 =====
    uint256 public counter;           // 默认 0
    address public owner;             // 部署者地址
    bool public isPaused;             // 默认 false
    string public name;               // 默认 ""
    bytes32 public hashValue;         // 默认 0x0...

    // ===== 常量与不可变 =====
    uint256 public constant MAX_SUPPLY = 10000;
    address public immutable deployer;

    // ===== 映射 =====
    mapping(address => uint256) public balances;
    mapping(address => mapping(address => uint256)) public allowances;

    // ===== 枚举 =====
    enum Status { Pending, Active, Inactive }
    Status public contractStatus;

    // ===== 结构体 =====
    struct User {
        address wallet;
        uint256 balance;
        bool isVerified;
        uint256 registeredAt;
    }
    mapping(address => User) public users;

    // ===== 数组 =====
    uint256[] public dynamicArray;
    uint256[5] public fixedArray;

    constructor() {
        owner = msg.sender;
        deployer = msg.sender;
        contractStatus = Status.Pending;
    }

    // ===== 函数修饰器 =====
    modifier onlyOwner() {
        require(msg.sender == owner, "Not authorized");
        _;
    }

    modifier whenNotPaused() {
        require(!isPaused, "Contract is paused");
        _;
    }

    // ===== 函数类型 =====
    // view: 只读状态，不修改
    function getCount() external view returns (uint256) {
        return counter;
    }

    // pure: 纯计算，不读写状态
    function add(uint256 a, uint256 b) external pure returns (uint256) {
        return a + b;
    }

    // payable: 可接收 ETH
    function deposit() external payable whenNotPaused {
        require(msg.value > 0, "Must send ETH");
        balances[msg.sender] += msg.value;
    }

    function withdraw(uint256 amount) external onlyOwner {
        require(balances[msg.sender] >= amount, "Insufficient balance");
        // Checks-Effects-Interactions 模式
        balances[msg.sender] -= amount;
        (bool success, ) = msg.sender.call{value: amount}("");
        require(success, "Transfer failed");
    }

    // ===== 事件 =====
    event Transfer(address indexed from, address indexed to, uint256 value);
    event CounterIncremented(address indexed by, uint256 newValue);

    function increment() external whenNotPaused {
        counter++;
        emit CounterIncremented(msg.sender, counter);
    }

    function transfer(address to, uint256 amount) external {
        require(balances[msg.sender] >= amount, "Insufficient");
        balances[msg.sender] -= amount;
        balances[to] += amount;
        emit Transfer(msg.sender, to, amount);
    }

    // ===== 自定义错误 (Gas 优化) =====
    error Unauthorized(address caller);
    error InsufficientBalance(uint256 available, uint256 required);

    function adminWithdraw(uint256 amount) external {
        if (msg.sender != owner) revert Unauthorized(msg.sender);
        if (balances[msg.sender] < amount)
            revert InsufficientBalance(balances[msg.sender], amount);
        balances[msg.sender] -= amount;
    }

    // ===== 接收 ETH =====
    receive() external payable {
        balances[msg.sender] += msg.value;
    }

    fallback() external payable {
        // 处理未知函数调用
    }
}
```

### 1.2 继承与接口

```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.24;

interface IERC20 {
    function totalSupply() external view returns (uint256);
    function balanceOf(address account) external view returns (uint256);
    function transfer(address to, uint256 amount) external returns (bool);
    event Transfer(address indexed from, address indexed to, uint256 value);
}

abstract contract Ownable {
    address public owner;
    constructor() { owner = msg.sender; }
    modifier onlyOwner() {
        require(msg.sender == owner, "Not owner");
        _;
    }
    function transferOwnership(address newOwner) external onlyOwner {
        owner = newOwner;
    }
}

contract MyToken is Ownable, IERC20 {
    string public name = "MyToken";
    string public symbol = "MTK";
    uint8 public decimals = 18;
    uint256 private _totalSupply;

    mapping(address => uint256) private _balances;

    constructor(uint256 initialSupply) {
        _totalSupply = initialSupply;
        _balances[msg.sender] = initialSupply;
        emit Transfer(address(0), msg.sender, initialSupply);
    }

    function totalSupply() external view returns (uint256) {
        return _totalSupply;
    }

    function balanceOf(address account) external view returns (uint256) {
        return _balances[account];
    }

    function transfer(address to, uint256 amount) external returns (bool) {
        require(_balances[msg.sender] >= amount, "Insufficient");
        _balances[msg.sender] -= amount;
        _balances[to] += amount;
        emit Transfer(msg.sender, to, amount);
        return true;
    }
}
```

### 2.3 部署脚本

```typescript
// scripts/deploy.ts
import { ethers } from "hardhat";

async function main() {
  const [deployer] = await ethers.getSigners();
  console.log("Deploying with account:", deployer.address);
  console.log("Account balance:", ethers.formatEther(await deployer.provider.getBalance(deployer.address)));

  // 部署 MyToken
  const MyToken = await ethers.getContractFactory("MyToken");
  const initialSupply = ethers.parseEther("1000000");
  const token = await MyToken.deploy(initialSupply);
  await token.waitForDeployment();

  const address = await token.getAddress();
  console.log("MyToken deployed to:", address);
  console.log("Token name:", await token.name());
  console.log("Token symbol:", await token.symbol());
}

main()
  .then(() => process.exit(0))
  .catch((error) => {
    console.error(error);
    process.exit(1);
  });
```

### 2.4 测试脚本

```typescript
// test/MyToken.test.ts
import { expect } from "chai";
import { ethers } from "hardhat";
import { MyToken } from "../typechain-types";

describe("MyToken", () => {
  let token: MyToken;
  let owner: any, addr1: any, addr2: any;

  beforeEach(async () => {
    [owner, addr1, addr2] = await ethers.getSigners();
    const MyTokenFactory = await ethers.getContractFactory("MyToken");
    token = await MyTokenFactory.deploy(ethers.parseEther("1000000"));
    await token.waitForDeployment();
  });

  describe("Deployment", () => {
    it("Should set the right owner", async () => {
      expect(await token.owner()).to.equal(owner.address);
    });

    it("Should assign total supply to owner", async () => {
      const balance = await token.balanceOf(owner.address);
      expect(balance).to.equal(ethers.parseEther("1000000"));
    });
  });

  describe("Transactions", () => {
    it("Should transfer tokens between accounts", async () => {
      await token.transfer(addr1.address, ethers.parseEther("50"));
      expect(await token.balanceOf(addr1.address)).to.equal(ethers.parseEther("50"));
    });

    it("Should emit Transfer event", async () => {
      await expect(token.transfer(addr1.address, ethers.parseEther("50")))
        .to.emit(token, "Transfer")
        .withArgs(owner.address, addr1.address, ethers.parseEther("50"));
    });

    it("Should fail if sender has insufficient balance", async () => {
      await expect(
        token.connect(addr1).transfer(addr2.address, ethers.parseEther("1"))
      ).to.be.revertedWith("Insufficient");
    });
  });
});
```

### 3.1 安装与连接钱包

```typescript
// src/lib/ethers.ts
import { ethers, Contract, JsonRpcProvider, BrowserProvider } from "ethers";

// === 连接到 MetaMask (浏览器端) ===
export async function connectWallet(): Promise<string> {
  if (!window.ethereum) throw new Error("MetaMask not installed");

  const provider = new BrowserProvider(window.ethereum);
  const accounts = await provider.send("eth_requestAccounts", []);
  const signer = await provider.getSigner();
  const network = await provider.getNetwork();

  console.log(`Connected: ${accounts[0]} on chain ${network.chainId}`);
  return accounts[0];
}

// === 连接到 RPC 节点 (服务端) ===
export function getProvider(network: string = "sepolia"): JsonRpcProvider {
  const rpcUrl = process.env.RPC_URL || "https://rpc.ankr.com/eth_sepolia";
  return new JsonRpcProvider(rpcUrl);
}
```

### 3.2 与合约交互

```typescript
// src/lib/contract.ts
import { ethers, Contract, BrowserProvider } from "ethers";

const TOKEN_ABI = [
  "function name() view returns (string)",
  "function symbol() view returns (string)",
  "function balanceOf(address) view returns (uint256)",
  "function transfer(address to, uint256 amount) returns (bool)",
  "event Transfer(address indexed from, address indexed to, uint256 value)",
];

const TOKEN_ADDRESS = "0x1234...abcd";

// 读取合约 (只读)
export async function getTokenBalance(address: string): Promise<string> {
  const provider = new BrowserProvider(window.ethereum);
  const contract = new Contract(TOKEN_ADDRESS, TOKEN_ABI, provider);
  const balance = await contract.balanceOf(address);
  return ethers.formatEther(balance);
}

// 写入合约 (需要签名者)
export async function transferTokens(to: string, amount: string): Promise<string> {
  const provider = new BrowserProvider(window.ethereum);
  const signer = await provider.getSigner();
  const contract = new Contract(TOKEN_ADDRESS, TOKEN_ABI, signer);

  const tx = await contract.transfer(to, ethers.parseEther(amount));
  console.log("Transaction sent:", tx.hash);

  const receipt = await tx.wait();
  console.log("Confirmed in block:", receipt.blockNumber);
  return tx.hash;
}

// 监听事件
export function listenToTransfers(callback: (from: string, to: string, amount: bigint) => void) {
  const provider = new BrowserProvider(window.ethereum);
  const contract = new Contract(TOKEN_ADDRESS, TOKEN_ABI, provider);

  contract.on("Transfer", (from, to, amount) => {
    console.log(`Transfer: ${ethers.formatEther(amount)} from ${from} to ${to}`);
    callback(from, to, amount);
  });
}
```

### 3.3 React 集成示例

```tsx
// src/components/WalletButton.tsx
import { useState, useEffect } from "react";
import { ethers } from "ethers";

export function WalletButton() {
  const [account, setAccount] = useState<string>("");
  const [balance, setBalance] = useState<string>("");

  const connect = async () => {
    if (!window.ethereum) {
      alert("Please install MetaMask");
      return;
    }
    const provider = new ethers.BrowserProvider(window.ethereum);
    const accounts = await provider.send("eth_requestAccounts", []);
    const signer = await provider.getSigner();
    const bal = await provider.getBalance(accounts[0]);
    setAccount(accounts[0]);
    setBalance(ethers.formatEther(bal));
  };

  // 监听账户切换
  useEffect(() => {
    if (window.ethereum) {
      window.ethereum.on("accountsChanged", (accounts: string[]) => {
        setAccount(accounts[0] || "");
      });
    }
  }, []);

  return (
    <div>
      {account ? (
        <div>
          <p>Address: {account.slice(0, 6)}...{account.slice(-4)}</p>
          <p>Balance: {parseFloat(balance).toFixed(4)} ETH</p>
        </div>
      ) : (
        <button onClick={connect}>Connect Wallet</button>
      )}
    </div>
  );
}
```

### 4.2 部署脚本（多合约 + 地址管理）

```typescript
// scripts/deploy_all.ts
import { ethers } from "hardhat";
import * as fs from "fs";

async function main() {
  const [deployer] = await ethers.getSigners();
  console.log("Deployer:", deployer.address);

  const deployments: Record<string, string> = {};

  // 1. 部署 Token
  const Token = await ethers.getContractFactory("MyToken");
  const token = await Token.deploy(ethers.parseEther("1000000"));
  await token.waitForDeployment();
  deployments.MyToken = await token.getAddress();
  console.log("MyToken:", deployments.MyToken);

  // 2. 部署 Marketplace (依赖 Token 地址)
  const Marketplace = await ethers.getContractFactory("Marketplace");
  const marketplace = await Marketplace.deploy(deployments.MyToken);
  await marketplace.waitForDeployment();
  deployments.Marketplace = await marketplace.getAddress();
  console.log("Marketplace:", deployments.Marketplace);

  // 保存部署地址
  const chainId = (await ethers.provider.getNetwork()).chainId.toString();
  const deployPath = `deployments/${chainId}.json`;
  fs.mkdirSync("deployments", { recursive: true });
  fs.writeFileSync(deployPath, JSON.stringify(deployments, null, 2));
  console.log(`Addresses saved to ${deployPath}`);
}

main().catch(console.error);
```

## 5. Gas 优化

### 5.1 Gas 消耗对比

| 操作 | Gas 消耗 (约) | 说明 |
|------|--------------|------|
| SSTORE (新值) | 20,000 | 写入新存储槽 |
| SSTORE (修改) | 5,000 | 修改现有存储 |
| SLOAD | 2,100 | 读取存储槽 |
| MLOAD/MSTORE | 3 | 内存操作 |
| ADD/MUL/SUB | 3 | 算术运算 |
| 外部函数调用 | ~2,500+ | CALL 指令 |
| 事件日志 | 375 + 375/byte | LOG 指令 |

### 5.2 优化技巧

```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.24;

contract GasOptimization {
    // ❌ 不好的做法: 每个值单独存储
    mapping(address => uint256) public balanceA;
    mapping(address => uint256) public balanceB;
    mapping(address => uint256) public balanceC;

    // ✅ 好的做法: 使用 struct 打包
    struct Balances {
        uint128 a;  // 16 bytes
        uint128 b;  // 16 bytes → 一个槽 = 32 bytes
    }
    mapping(address => Balances) public packedBalances;

    // ❌ 不好的做法: 循环中多次 SLOAD
    function badLoop(uint256[] memory ids) external view returns (uint256 total) {
        for (uint256 i = 0; i < ids.length; i++) {
            total += balanceA[msg.sender]; // 每次 SLOAD
        }
    }

    // ✅ 好的做法: 缓存到 memory
    function goodLoop(uint256[] memory ids) external view returns (uint256 total) {
        uint256 cached = balanceA[msg.sender]; // 仅一次 SLOAD
        for (uint256 i = 0; i < ids.length; i++) {
            total += cached;
        }
    }

    // ✅ 使用 calldata 而非 memory (只读参数)
    function processIds(uint256[] calldata ids) external pure returns (uint256 sum) {
        for (uint256 i = 0; i < ids.length; i++) {
            sum += ids[i];
        }
    }

    // ✅ 使用 unchecked 块 (确定不会溢出时)
    function incrementLoop(uint256 n) external pure returns (uint256 sum) {
        unchecked {
            for (uint256 i = 0; i < n; i++) {
                sum += i;
            }
        }
    }

    // ✅ 使用 custom errors 代替 require string
    error AmountTooLow(uint256 provided, uint256 minimum);

    function deposit(uint256 amount) external pure {
        if (amount < 100) revert AmountTooLow(amount, 100);
        // require(amount >= 100, "Amount too low"); // ❌ 更贵
    }

    // ✅ 使用 immutable / constant
    uint256 public constant FEE_BPS = 100; // 1%
    address public immutable FEE_RECEIVER;

    constructor() {
        FEE_RECEIVER = msg.sender;
    }

    // ✅ 批量操作减少交易次数
    function batchTransfer(address[] calldata recipients, uint256[] calldata amounts) external {
        require(recipients.length == amounts.length, "Length mismatch");
        for (uint256 i = 0; i < recipients.length; i++) {
            balanceA[recipients[i]] += amounts[i];
        }
    }
}
```

```typescript
// scripts/estimate_gas.ts
import { ethers } from "hardhat";

async function estimateGas() {
  const [signer] = await ethers.getSigners();
  const contract = await ethers.getContractAt("MyToken", "0x...");

  // 估算 Gas
  const gasEstimate = await contract.transfer.estimateGas(
```

## 6. 最佳实践

### 安全清单

| 类别 | 实践 |
|------|------|
| 重入防护 | 使用 `ReentrancyGuard`，遵循 CEI 模式 |
| 整数溢出 | Solidity 0.8+ 已内置，使用 `unchecked` 需谨慎 |
| 访问控制 | 使用 `Ownable` / `AccessControl` / 时间锁 |
| 升级安全 | 使用 UUPS 或透明代理模式 |
| 预言机 | 使用 Chainlink 等 TWAP 预言机，避免单一价格源 |
| 前端跑 | 对关键操作使用 `commit-reveal` 模式 |
| 审计 | 主网前必须经过专业审计 |

### 开发流程

```
本地开发 → 单元测试 → 测试网部署 → 安全审计 → 主网部署 → 监控
```
