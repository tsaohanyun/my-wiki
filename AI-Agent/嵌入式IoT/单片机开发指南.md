## 1. STM32 开发基础

### 1.1 STM32 系列对比

| 系列 | 内核 | 定位 | 典型应用 |
|------|------|------|----------|
| STM32F0 | Cortex-M0 | 入门级 | 简单控制、传感器节点 |
| STM32F1 | Cortex-M3 | 通用型 | 电机控制、消费电子 |
| STM32F4 | Cortex-M4F | 高性能 | 数字信号处理、无人机 |
| STM32F7 | Cortex-M7 | 超高性能 | GUI、音视频处理 |
| STM32G0 | Cortex-M0+ | 低功耗 | IoT 传感器节点 |
| STM32H7 | Cortex-M7+M4 | 双核旗舰 | 高端工业控制 |
| STM32L4 | Cortex-M4F | 超低功耗 | 穿戴设备、远程传感 |
| STM32U5 | Cortex-M33 | 超低功耗+安全 | 智能卡、医疗设备 |
| STM32WL | Cortex-M4+LoRa | 无线SoC | LoRaWAN 节点 |

### 1.4 Makefile 模板

```makefile
# Makefile — STM32 GCC 构建模板

TARGET    = firmware
BUILD_DIR = build

# 工具链
CC      = arm-none-eabi-gcc
OBJCOPY = arm-none-eabi-objcopy
SIZE    = arm-none-eabi-size

# CPU 配置 (STM32F407)
CPU = -mcpu=cortex-m4
FPU = -mfpu=fpv4-sp-d16 -mfloat-abi=hard
MCU = $(CPU) -mthumb $(FPU)

# 编译选项
CFLAGS  = $(MCU) -std=c11 -Wall -Wextra -Os -g3
CFLAGS += -DUSE_HAL_DRIVER -DSTM32F407xx
CFLAGS += -ICore/Inc -IDrivers/CMSIS/Device/ST/STM32F4xx/Include
CFLAGS += -IDrivers/STM32F4xx_HAL_Driver/Inc
CFLAGS += -IDrivers/CMSIS/Include

# 链接脚本
LDSCRIPT = STM32F407VGTx.ld
LDFLAGS  = $(MCU) -T$(LDSCRIPT) -Wl,--gc-sections -Wl,-Map=$(BUILD_DIR)/$(TARGET).map

# 源文件
C_SOURCES = \
    Core/Src/main.c \
    Core/Src/stm32f4xx_it.c \
    Core/Src/system_stm32f4xx.c \
    Drivers/STM32F4xx_HAL_Driver/Src/stm32f4xx_hal_gpio.c \
    Drivers/STM32F4xx_HAL_Driver/Src/stm32f4xx_hal_uart.c \
    Drivers/STM32F4xx_HAL_Driver/Src/stm32f4xx_hal_i2c.c \
    Drivers/STM32F4xx_HAL_Driver/Src/stm32f4xx_hal_spi.c \
    Drivers/STM32F4xx_HAL_Driver/Src/stm32f4xx_hal_tim.c \
    Drivers/STM32F4xx_HAL_Driver/Src/stm32f4xx_hal.c \
    Drivers/STM32F4xx_HAL_Driver/Src/stm32f4xx_hal_rcc.c \
    Drivers/STM32F4xx_HAL_Driver/Src/stm32f4xx_hal_cortex.c

all: $(BUILD_DIR)/$(TARGET).elf $(BUILD_DIR)/$(TARGET).hex $(BUILD_DIR)/$(TARGET).bin

$(BUILD_DIR)/$(TARGET).elf: $(C_SOURCES)
	mkdir -p $(BUILD_DIR)
	$(CC) $(CFLAGS) $(C_SOURCES) $(LDFLAGS) -o $@
	$(SIZE) $@

$(BUILD_DIR)/$(TARGET).hex: $(BUILD_DIR)/$(TARGET).elf
	$(OBJCOPY) -O ihex $< $@

$(BUILD_DIR)/$(TARGET).bin: $(BUILD_DIR)/$(TARGET).elf
	$(OBJCOPY) -O binary $< $@

flash:
	openocd -f interface/stlink.cfg -f target/stm32f4x.cfg \
		-c "program $(BUILD_DIR)/$(TARGET).elf verify reset exit"

clean:
	rm -rf $(BUILD_DIR)
```

## 2. HAL 库编程

### 2.1 GPIO 操作

```c
#include "main.h"

// GPIO 初始化（由 CubeMX 生成或手动配置）
void GPIO_Init(void) {
    __HAL_RCC_GPIOA_CLK_ENABLE();
    __HAL_RCC_GPIOB_CLK_ENABLE();

    // LED 输出
    GPIO_InitTypeDef gpio = {0};
    gpio.Pin   = GPIO_PIN_5;         // PA5
    gpio.Mode  = GPIO_MODE_OUTPUT_PP;
    gpio.Pull  = GPIO_NOPULL;
    gpio.Speed = GPIO_SPEED_FREQ_HIGH;
    HAL_GPIO_Init(GPIOA, &gpio);

    // 按键输入（带上拉）
    gpio.Pin  = GPIO_PIN_0;          // PB0
    gpio.Mode = GPIO_MODE_INPUT;
    gpio.Pull = GPIO_PULLUP;
    HAL_GPIO_Init(GPIOB, &gpio);
}

int main(void) {
    HAL_Init();
    SystemClock_Config();
    GPIO_Init();

    while (1) {
        // 读取按键
        if (HAL_GPIO_ReadPin(GPIOB, GPIO_PIN_0) == GPIO_PIN_RESET) {
            HAL_GPIO_WritePin(GPIOA, GPIO_PIN_5, GPIO_PIN_SET);   // LED ON
        } else {
            HAL_GPIO_TogglePin(GPIOA, GPIO_PIN_5);                // LED 闪烁
        }
        HAL_Delay(500);
    }
}
```

## 3. 外设驱动开发

### 3.1 I2C 传感器驱动（BME280）

```c
#include "bme280.h"

#define BME280_ADDR         0x76 << 1   // HAL 需要 8 位地址
#define BME280_REG_ID       0xD0
#define BME280_REG_CTRL_HUM 0xF2
#define BME280_REG_CTRL     0xF4
#define BME280_REG_DATA     0xF7

extern I2C_HandleTypeDef hi2c1;

// 读寄存器
HAL_StatusTypeDef BME280_ReadReg(uint8_t reg, uint8_t *data, uint16_t len) {
    return HAL_I2C_Mem_Read(&hi2c1, BME280_ADDR, reg,
                            I2C_MEMADD_SIZE_8BIT, data, len, 100);
}

// 写寄存器
HAL_StatusTypeDef BME280_WriteReg(uint8_t reg, uint8_t value) {
    return HAL_I2C_Mem_Write(&hi2c1, BME280_ADDR, reg,
                             I2C_MEMADD_SIZE_8BIT, &value, 1, 100);
}

// 校验设备 ID
bool BME280_Init(void) {
    uint8_t id = 0;
    if (BME280_ReadReg(BME280_REG_ID, &id, 1) != HAL_OK) return false;
    if (id != 0x60) return false;

    // 配置：湿度 1x 过采样
    BME280_WriteReg(BME280_REG_CTRL_HUM, 0x01);
    // 温度 2x, 气压 16x, 正常模式
    BME280_WriteReg(BME280_REG_CTRL, 0x54);
    HAL_Delay(10);
    return true;
}

// 读取数据
typedef struct {
    float temperature;
    float pressure;
    float humidity;
} BME280_Data;

bool BME280_Read(BME280_Data *data) {
    uint8_t raw[8];
    if (BME280_ReadReg(BME280_REG_DATA, raw, 8) != HAL_OK) return false;

    // 原始数据拼接
    int32_t adc_T = (raw[3] << 12) | (raw[4] << 4) | (raw[5] >> 4);
    int32_t adc_P = (raw[0] << 12) | (raw[1] << 4) | (raw[2] >> 4);
    int32_t adc_H = (raw[6] << 8)  | raw[7];

    // 补偿计算（简化版，实际需要校准系数）
    // 完整补偿算法请参考 BME280 数据手册
    data->temperature = (float)adc_T / 5120.0f;  // 简化
    data->pressure    = (float)adc_P / 256.0f;
    data->humidity    = (float)adc_H / 1024.0f;
    return true;
}
```

### 3.2 SPI 驱动（W25Q64 Flash）

```c
#include "w25q64.h"

#define W25Q64_CS_LOW()   HAL_GPIO_WritePin(GPIOA, GPIO_PIN_4, GPIO_PIN_RESET)
#define W25Q64_CS_HIGH()  HAL_GPIO_WritePin(GPIOA, GPIO_PIN_4, GPIO_PIN_SET)

extern SPI_HandleTypeDef hspi1;

#define CMD_READ  0x03
#define CMD_WRITE 0x02
#define CMD_WREN  0x06
#define CMD_RDID  0x9F

uint32_t W25Q64_ReadID(void) {
    uint8_t cmd[4] = {CMD_RDID, 0, 0, 0};
    uint8_t rx[4];

    W25Q64_CS_LOW();
    HAL_SPI_TransmitReceive(&hspi1, cmd, rx, 4, 100);
    W25Q64_CS_HIGH();

    return (rx[1] << 16) | (rx[2] << 8) | rx[3];
}

void W25Q64_WriteEnable(void) {
    uint8_t cmd = CMD_WREN;
    W25Q64_CS_LOW();
    HAL_SPI_Transmit(&hspi1, &cmd, 1, 100);
    W25Q64_CS_HIGH();
}

void W25Q64_Read(uint32_t addr, uint8_t *buf, uint16_t len) {
    uint8_t cmd[4] = {
        CMD_READ,
        (addr >> 16) & 0xFF,
        (addr >> 8)  & 0xFF,
        addr & 0xFF
    };
    W25Q64_CS_LOW();
    HAL_SPI_Transmit(&hspi1, cmd, 4, 100);
    HAL_SPI_Receive(&hspi1, buf, len, 1000);
    W25Q64_CS_HIGH();
}
```

### 3.3 PWM 输出（TIM 定时器）

```c
// TIM3 PWM 初始化 — 控制舵机
void PWM_Init(void) {
    __HAL_RCC_TIM3_CLK_ENABLE();

    TIM_HandleTypeDef htim3;
    htim3.Instance               = TIM3;
    htim3.Init.Prescaler         = 84 - 1;      // 84MHz / 84 = 1MHz
    htim3.Init.CounterMode       = TIM_COUNTERMODE_UP;
    htim3.Init.Period            = 20000 - 1;   // 1MHz / 20000 = 50Hz (舵机标准)
    htim3.Init.ClockDivision     = TIM_CLOCKDIVISION_DIV1;
    HAL_TIM_PWM_Init(&htim3);

    // PWM 通道配置
    TIM_OC_InitTypeDef ocConfig;
    ocConfig.OCMode     = TIM_OCMODE_PWM1;
    ocConfig.Pulse      = 1500;   // 1.5ms → 中间位置
    ocConfig.OCPolarity = TIM_OCPOLARITY_HIGH;
    HAL_TIM_PWM_ConfigChannel(&htim3, &ocConfig, TIM_CHANNEL_1);

    HAL_TIM_PWM_Start(&htim3, TIM_CHANNEL_1);
}

// 设置舵机角度 (0~180度)
void Servo_SetAngle(TIM_HandleTypeDef *htim, float angle) {
    // 0度=500us, 180度=2500us
    uint16_t pulse = (uint16_t)(500 + (angle / 180.0f) * 2000);
    __HAL_TIM_SET_COMPARE(htim, TIM_CHANNEL_1, pulse);
}
```

```c
void EXTI_Init(void) {
    __HAL_RCC_GPIOB_CLK_ENABLE();
    __HAL_RCC_SYSCFG_CLK_ENABLE();

    // GPIO 配置
    GPIO_InitTypeDef gpio = {0};
    gpio.Pin  = GPIO_PIN_0;
```

### 4.2 DMA 传输

```c
// DMA 接收 UART 数据（高效，CPU 不参与）
#define RX_BUF_SIZE 256

uint8_t dma_rx_buf[RX_BUF_SIZE];

void UART_DMA_Init(void) {
    // 开启 UART 空闲中断
    __HAL_UART_ENABLE_IT(&huart1, UART_IT_IDLE);

    // 启动 DMA 接收
    HAL_UART_Receive_DMA(&huart1, dma_rx_buf, RX_BUF_SIZE);
}

// 在中断中处理空闲事件
void USART1_IRQHandler(void) {
    if (__HAL_UART_GET_FLAG(&huart1, UART_FLAG_IDLE)) {
        __HAL_UART_CLEAR_IDLEFLAG(&huart1);

        // 停止 DMA
        HAL_UART_DMAStop(&huart1);

        // 计算接收长度
        uint16_t len = RX_BUF_SIZE - __HAL_DMA_GET_COUNTER(huart1.hdmarx);

        // 处理数据
        ProcessData(dma_rx_buf, len);

        // 重启 DMA 接收
        HAL_UART_Receive_DMA(&huart1, dma_rx_buf, RX_BUF_SIZE);
    }
}
```

### 5.2 Arduino + WiFi (ESP32)

```cpp
#include <WiFi.h>
#include <PubSubClient.h>

const char* WIFI_SSID = "MyIoT-Network";
const char* WIFI_PASS = "WiFiPassword123";
const char* MQTT_BROKER = "broker.myiot.local";
const int   MQTT_PORT   = 1883;

WiFiClient esp_client;
PubSubClient mqtt_client(esp_client);

void mqtt_callback(char* topic, byte* payload, unsigned int length) {
    Serial.print("Message [");
    Serial.print(topic);
    Serial.print("]: ");
    for (int i = 0; i < length; i++) {
        Serial.print((char)payload[i]);
    }
    Serial.println();
}

void setup() {
    Serial.begin(115200);

    // 连接 WiFi
    WiFi.begin(WIFI_SSID, WIFI_PASS);
    while (WiFi.status() != WL_CONNECTED) {
        delay(500);
        Serial.print(".");
    }
    Serial.println("\nWiFi connected: " + WiFi.localIP().toString());

    // 连接 MQTT
    mqtt_client.setServer(MQTT_BROKER, MQTT_PORT);
    mqtt_client.setCallback(mqtt_callback);
}

void loop() {
    if (!mqtt_client.connected()) {
        String client_id = "esp32-" + String(WiFi.macAddress());
        if (mqtt_client.connect(client_id.c_str(),
                                "device_001", "DeviceP@ss")) {
            mqtt_client.subscribe("commands/device_001/#");
            Serial.println("MQTT connected");
        }
    }

    // 每隔 10 秒发布传感器数据
    static unsigned long last_pub = 0;
    if (millis() - last_pub >= 10000) {
        last_pub = millis();

        float temp = readTemperature();
        float humi = readHumidity();

        String payload = "{\"temp\":" + String(temp, 1) +
                         ",\"humi\":" + String(humi, 1) + "}";
        mqtt_client.publish("devices/device_001/telemetry",
                           payload.c_str());
    }

    mqtt_client.loop();
}

float readTemperature() {
    return 25.6 + (analogRead(34) * 0.001f);  // 示例
}

float readHumidity() {
    return 60.0 + (analogRead(35) * 0.001f);  // 示例
}
```

### 5.3 ESP32 Deep Sleep（低功耗）

```cpp
    Serial.println("Going to sleep...");
    esp_deep_sleep_start();
}

void loop() {
    // Deep Sleep 模式下 loop 不会执行
}

void collectAndSend() {
    // 采集传感器数据并通过 MQTT/LoRa 发送
    Serial.println("Data collected and sent");
```

## 6. FreeRTOS 实时操作系统

### 6.1 FreeRTOS 基础概念

| 概念 | 说明 |
|------|------|
| Task | 独立执行的线程 |
| Priority | 任务优先级（0~configMAX_PRIORITIES-1） |
| Queue | 任务间通信的 FIFO 队列 |
| Semaphore | 信号量（二值/计数） |
| Mutex | 互斥锁（带优先级继承） |
| Timer | 软件定时器 |

### 6.2 任务创建与管理

```c
#include "FreeRTOS.h"
#include "task.h"
#include "queue.h"
#include "semphr.h"

// 任务栈大小
#define TASK_STACK_SIZE 256

// 传感器数据队列
QueueHandle_t sensorQueue;

// 互斥锁（保护 UART）
SemaphoreHandle_t uartMutex;

// 传感器采集任务
void SensorTask(void *pvParameters) {
    (void)pvParameters;
    TickType_t lastWake = xTaskGetTickCount();

    while (1) {
        // 以固定周期运行（500ms）
        vTaskDelayUntil(&lastWake, pdMS_TO_TICKS(500));

        // 读取传感器
        BME280_Data data;
        if (BME280_Read(&data)) {
            // 发送到队列
            xQueueSend(sensorQueue, &data, pdMS_TO_TICKS(10));
        }
    }
}

// 数据处理任务
void ProcessTask(void *pvParameters) {
    (void)pvParameters;
    BME280_Data data;

    while (1) {
        // 等待队列数据（最多等待 1 秒）
        if (xQueueReceive(sensorQueue, &data, pdMS_TO_TICKS(1000)) == pdPASS) {
            // 加锁保护 UART
            xSemaphoreTake(uartMutex, portMAX_DELAY);
            printf("T=%.1f°C  H=%.1f%%  P=%.0fhPa\r\n",
                   data.temperature, data.humidity, data.pressure);
            xSemaphoreGive(uartMutex);
        }
    }
}

// MQTT 通信任务
void MQTTTask(void *pvParameters) {
    (void)pvParameters;
    BME280_Data data;

    while (1) {
        if (xQueueReceive(sensorQueue, &data, portMAX_DELAY) == pdPASS) {
            // 发送到 MQTT
            char payload[128];
            snprintf(payload, sizeof(payload),
                     "{\"t\":%.1f,\"h\":%.1f,\"p\":%.0f}",
                     data.temperature, data.humidity, data.pressure);
            mqtt_publish(payload);
        }
    }
}

int main(void) {
    HAL_Init();
    SystemClock_Config();

    // 创建队列和互斥锁
    sensorQueue = xQueueCreate(10, sizeof(BME280_Data));
    uartMutex   = xSemaphoreCreateMutex();

    // 创建任务
    xTaskCreate(SensorTask,  "Sensor",  TASK_STACK_SIZE, NULL, 3, NULL);
    xTaskCreate(ProcessTask, "Process", TASK_STACK_SIZE, NULL, 2, NULL);
    xTaskCreate(MQTTTask,    "MQTT",    TASK_STACK_SIZE, NULL, 2, NULL);

    // 启动调度器
    vTaskStartScheduler();

    while (1) {}  // 不应到达此处
}
```

### 6.4 任务间通信：事件组

```c
#include "event_groups.h"

#define BIT_SENSOR_READY  (1 << 0)
#define BIT_NETWORK_READY (1 << 1)
#define BIT_DATA_SENT     (1 << 2)

EventGroupHandle_t systemEvents;

// 等待多个条件同时满足
void CoordinatorTask(void *pv) {
    (void)pv;
    while (1) {
        // 等待传感器和网络都就绪
        EventBits_t bits = xEventGroupWaitBits(
            systemEvents,
            BIT_SENSOR_READY | BIT_NETWORK_READY,
            pdFALSE,   // 不清除位
            pdTRUE,    // 等待所有位
            portMAX_DELAY
        );

        if ((bits & (BIT_SENSOR_READY | BIT_NETWORK_READY)) ==
            (BIT_SENSOR_READY | BIT_NETWORK_READY)) {
            // 两个条件都满足，开始数据上传
            uploadData();
            xEventGroupSetBits(systemEvents, BIT_DATA_SENT);
        }
    }
}
```
