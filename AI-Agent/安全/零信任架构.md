# 零信任架构

## 概述

零信任架构（ZTA）是一种安全模型，核心理念是"永不信任，始终验证"。无论请求来自内部还是外部网络，都需要经过严格的身份验证和授权。

---

## 零信任核心原则

### NIST SP 800-207 三大原则

| 原则 | 描述 | 实施要点 |
|------|------|----------|
| **所有数据源和计算服务都是资源** | 将所有服务视为独立资源 | 资产清单、服务目录 |
| **所有通信都必须保护** | 不区分内外网，一律加密 | mTLS、端到端加密 |
| **按会话授予访问权限** | 每次访问都需验证 | 最小权限、即时撤销 |

### 零信任 vs 传统安全模型

```
传统模型:                          零信任模型:
┌────────────────────┐            ┌────────────────────┐
│    信任边界（防火墙） │            │   身份验证网关      │
│  ┌────────────────┐ │            │  ┌────────────────┐ │
│  │  内部网络（信任） │ │            │  │ 每个资源独立验证 │ │
│  │  ┌────┐  ┌────┐ │ │            │  │ ┌────┐  ┌────┐ │ │
│  │  │服务│←→│服务│ │ │            │  │ │服务│←→│服务│ │ │
│  │  └────┘  └────┘ │ │            │  │ └────┘  └────┘ │ │
│  └────────────────┘ │            │  └────────────────┘ │
└────────────────────┘            └────────────────────┘
    边界内信任一切                    每次访问都验证
```

## 身份验证

### 多因素认证（MFA）

```python
# TOTP实现（基于时间的一次性密码）
import pyotp
import qrcode

class MFAManager:
    def __init__(self):
        self.issuer = "MyApp"

    def generate_secret(self, user_id: str) -> dict:
        """生成MFA密钥"""
        secret = pyotp.random_base32()
        totp = pyotp.TOTP(secret)

        # 生成二维码URI
        provisioning_uri = totp.provisioning_uri(
            name=user_id,
            issuer_name=self.issuer
        )

        return {
            'secret': secret,
            'uri': provisioning_uri,
            'qr_code': self._generate_qr(provisioning_uri)
        }

    def verify_token(self, secret: str, token: str) -> bool:
        """验证TOTP令牌"""
        totp = pyotp.TOTP(secret)
        # 允许前后各30秒的窗口
        return totp.verify(token, valid_window=1)

    def _generate_qr(self, uri: str) -> str:
        """生成QR码图片"""
        qr = qrcode.QRCode(version=1, box_size=10, border=5)
        qr.add_data(uri)
        qr.make(fit=True)
        img = qr.make_image(fill_color="black", back_color="white")
        # 返回base64编码的图片
        import base64, io
        buffer = io.BytesIO()
        img.save(buffer, format='PNG')
        return base64.b64encode(buffer.getvalue()).decode()

# WebAuthn无密码认证
from webauthn import generate_registration_options, verify_registration_response

async def register_webauthn(user):
    options = generate_registration_options(
        rp_id="example.com",
        rp_name="My Application",
        user_id=user.id.encode(),
        user_name=user.username,
        authenticator_selection=AuthenticatorSelectionCriteria(
            authenticator_attachment=AuthenticatorAttachment.PLATFORM,
            resident_key=ResidentKeyRequirement.REQUIRED
        )
    )
    return options
```

### 设备信任评估

```python
from dataclasses import dataclass
from enum import Enum

class TrustLevel(Enum):
    HIGH = "high"
    MEDIUM = "medium"
    LOW = "low"
    UNTRUSTED = "untrusted"

@dataclass
class DeviceProfile:
    device_id: str
    os_version: str
    patch_level: str
    is_managed: bool
    is_encrypted: bool
    has_antivirus: bool
    last_compliance_check: str
    jailbroken: bool

class DeviceTrustEvaluator:
    def evaluate(self, device: DeviceProfile) -> TrustLevel:
        score = 0

        # 设备管理状态
        if device.is_managed:
            score += 30

        # 加密状态
        if device.is_encrypted:
            score += 25

        # 安全软件
        if device.has_antivirus:
            score += 15

        # 越狱/root检测
        if device.jailbroken:
            score -= 50

        # 补丁状态
        if self._is_patched(device.patch_level):
            score += 20

        # 评估信任等级
        if score >= 80:
            return TrustLevel.HIGH
        elif score >= 60:
            return TrustLevel.MEDIUM
        elif score >= 30:
            return TrustLevel.LOW
        return TrustLevel.UNTRUSTED

    def _is_patched(self, patch_level: str) -> bool:
        # 检查补丁是否在90天内
        from datetime import datetime, timedelta
        patch_date = datetime.fromisoformat(patch_level)
        return datetime.now() - patch_date < timedelta(days=90)
```

## 持续验证

### 持续自适应信任评估

```python
from datetime import datetime, timedelta
from dataclasses import dataclass

@dataclass
class AccessRequest:
    user_id: str
    device_id: str
    resource: str
    action: str
    ip_address: str
    timestamp: datetime
    geo_location: str

@dataclass
class TrustScore:
    user_trust: float      # 0-100
    device_trust: float    # 0-100
    network_trust: float   # 0-100
    behavior_trust: float  # 0-100
    overall: float         # 加权平均

class ContinuousVerifier:
    def __init__(self):
        self.user_history: dict[str, list] = {}
        self.risk_thresholds = {
            'high_risk_resource': 80,
            'medium_risk_resource': 60,
            'low_risk_resource': 40
        }

    def evaluate_trust(self, request: AccessRequest) -> TrustScore:
        user_trust = self._evaluate_user_trust(request.user_id)
        device_trust = self._evaluate_device_trust(request.device_id)
        network_trust = self._evaluate_network_trust(request.ip_address)
        behavior_trust = self._evaluate_behavior(request)

        # 加权计算
        overall = (
            user_trust * 0.3 +
            device_trust * 0.25 +
            network_trust * 0.2 +
            behavior_trust * 0.25
        )

        return TrustScore(
            user_trust=user_trust,
            device_trust=device_trust,
            network_trust=network_trust,
            behavior_trust=behavior_trust,
            overall=overall
        )

    def authorize(self, request: AccessRequest) -> dict:
        trust_score = self.evaluate_trust(request)
        threshold = self.risk_thresholds.get(
            self._classify_resource(request.resource), 60
        )

        if trust_score.overall >= threshold:
            # 检查是否需要额外验证
            if trust_score.overall < threshold + 20:
                return {
                    'decision': 'allow_with_mfa',
                    'trust_score': trust_score,
                    'message': 'Additional verification required'
                }
            return {'decision': 'allow', 'trust_score': trust_score}
        else:
            return {
                'decision': 'deny',
                'trust_score': trust_score,
                'message': 'Insufficient trust score'
            }

    def _evaluate_user_trust(self, user_id: str) -> float:
        # 检查用户认证状态、密码强度、MFA状态等
        return 85.0

    def _evaluate_device_trust(self, device_id: str) -> float:
        # 检查设备合规性、补丁状态等
        return 75.0

    def _evaluate_network_trust(self, ip_address: str) -> float:
        # 检查IP信誉、地理位置、VPN状态等
        return 80.0

    def _evaluate_behavior(self, request: AccessRequest) -> float:
        # 用户行为分析（UEBA）
        history = self.user_history.get(request.user_id, [])

        # 检查异常时间访问
        if request.timestamp.hour < 6 or request.timestamp.hour > 22:
            return 40.0

        # 检查异常地理位置
        if history and request.geo_location not in [h.geo_location for h in history[-10:]]:
            return 50.0

        return 90.0

    def _classify_resource(self, resource: str) -> str:
        if 'admin' in resource or 'payment' in resource:
            return 'high_risk_resource'
        elif 'api' in resource:
            return 'medium_risk_resource'
        return 'low_risk_resource'
```

### 策略引擎（OPA）

```rego
# OPA策略 - 零信任访问控制
package zerotrust.authz

default allow = false

# 允许条件：信任分数足够高
allow {
    input.trust_score.overall >= resource_threshold
    input.user.mfa_verified == true
    input.device.compliant == true
}

# 资源风险阈值
resource_threshold = threshold {
    input.resource.classification == "critical"
    threshold := 80
}

resource_threshold = threshold {
    input.resource.classification == "high"
    threshold := 70
}

resource_threshold = threshold {
    input.resource.classification == "medium"
    threshold := 60
}

resource_threshold = 50  # 默认

# 高风险操作需要额外验证
require_step_up {
    input.action == "delete"
    input.resource.classification == "critical"
}

require_step_up {
    input.trust_score.behavior_trust < 60
}

# 时间限制
deny_outside_hours {
    input.timestamp.hour < 6
    input.resource.classification == "critical"
}

deny_outside_hours {
    input.timestamp.hour > 22
    input.resource.classification == "critical"
}
```

## 最佳实践

| 实践 | 描述 | 优先级 |
|------|------|--------|
| 永不信任，始终验证 | 每次访问都需身份验证和授权 | 🔴 高 |
| 最小权限访问 | 仅授予完成任务所需的最小权限 | 🔴 高 |
| 假设已被入侵 | 设计时假设网络已被渗透 | 🔴 高 |
| 微分段 | 将网络和应用划分为小的安全区域 | 🟡 中 |
| 持续监控 | 实时监控所有访问和行为 | 🔴 高 |
| 加密一切 | 所有数据传输和存储都加密 | 🔴 高 |
| 自动化响应 | 自动化安全事件检测和响应 | 🟡 中 |
| 最小暴露面 | 减少攻击面，隐藏不必要的服务 | 🟡 中 |
