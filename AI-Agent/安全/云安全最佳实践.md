# 云安全最佳实践

## 概述

云安全涵盖身份访问管理（IAM）、网络安全、数据加密和合规审计等核心领域，确保云基础设施和工作负载的安全。

## 身份与访问管理（IAM）

### 最小权限原则


### AWS IAM最佳实践

```hcl
# Terraform - IAM用户配置
resource "aws_iam_user" "developer" {
  name = "developer-${each.key}"
  path = "/developers/"

  for_each = var.developer_names
}

resource "aws_iam_user_policy_attachment" "developer_policy" {
  for_each   = var.developer_names
  user       = aws_iam_user.developer[each.key].name
  policy_arn = aws_iam_policy.developer_access.arn
}

# 策略：仅允许特定操作
resource "aws_iam_policy" "developer_access" {
  name        = "DeveloperAccess"
  description = "Minimal developer access"

  policy = jsonencode({
    Version = "2012-10-17"
    Statement = [
      {
        Effect   = "Allow"
        Action   = ["ec2:Describe*", "s3:Get*", "s3:List*"]
        Resource = "*"
        Condition = {
          StringEquals = {
            "aws:RequestedRegion" = ["us-east-1", "us-west-2"]
          }
        }
      },
      {
        Effect   = "Deny"
        Action   = ["iam:*", "organizations:*"]
        Resource = "*"
      }
    ]
  })
}

# 角色信任策略
resource "aws_iam_role" "app_role" {
  name = "application-role"

  assume_role_policy = jsonencode({
    Version = "2012-10-17"
    Statement = [{
      Effect = "Allow"
      Principal = {
        Service = "ec2.amazonaws.com"
      }
      Action = "sts:AssumeRole"
      Condition = {
        StringEquals = {
          "aws:SourceAccount" = var.account_id
        }
      }
    }]
  })
}
```

## 网络安全

### VPC安全架构

```hcl
# Terraform - AWS VPC安全配置
resource "aws_vpc" "main" {
  cidr_block           = "10.0.0.0/16"
  enable_dns_support   = true
  enable_dns_hostnames = true

  tags = { Name = "secure-vpc" }
}

# 私有子网（应用层）
resource "aws_subnet" "private_app" {
  vpc_id            = aws_vpc.main.id
  cidr_block        = "10.0.1.0/24"
  availability_zone = "us-east-1a"

  tags = { Name = "private-app-subnet" }
}

# 私有子网（数据层）
resource "aws_subnet" "private_data" {
  vpc_id            = aws_vpc.main.id
  cidr_block        = "10.0.2.0/24"
  availability_zone = "us-east-1a"

  tags = { Name = "private-data-subnet" }
}

# 网络ACL - 数据层
resource "aws_network_acl" "data_layer" {
  vpc_id     = aws_vpc.main.id
  subnet_ids = [aws_subnet.private_data.id]

  # 仅允许应用层访问数据库端口
  ingress {
    protocol   = "tcp"
    rule_no    = 100
    action     = "allow"
    cidr_block = "10.0.1.0/24"
    from_port  = 5432
    to_port    = 5432
  }

  # 拒绝所有其他入站
  ingress {
    protocol   = "-1"
    rule_no    = 999
    action     = "deny"
    cidr_block = "0.0.0.0/0"
    from_port  = 0
    to_port    = 0
  }

  egress {
    protocol   = "-1"
    rule_no    = 100
    action     = "allow"
    cidr_block = "0.0.0.0/0"
    from_port  = 0
    to_port    = 0
  }
}

# 安全组 - Web层
resource "aws_security_group" "web" {
  name        = "web-sg"
  description = "Web tier security group"
  vpc_id      = aws_vpc.main.id

  ingress {
    from_port   = 443
    to_port     = 443
    protocol    = "tcp"
    cidr_blocks = ["0.0.0.0/0"]
    description = "HTTPS"
  }

  ingress {
    from_port   = 80
    to_port     = 80
    protocol    = "tcp"
    cidr_blocks = ["0.0.0.0/0"]
    description = "HTTP redirect"
  }

  egress {
    from_port   = 0
    to_port     = 0
    protocol    = "-1"
    cidr_blocks = ["10.0.1.0/24"]
    description = "To app tier"
  }
}
```

## 数据加密

### 静态加密

```hcl
# AWS KMS密钥
resource "aws_kms_key" "app_key" {
  description             = "Application encryption key"
  deletion_window_in_days = 30
  enable_key_rotation     = true

  policy = jsonencode({
    Version = "2012-10-17"
    Statement = [
      {
        Sid    = "AllowRootAccount"
        Effect = "Allow"
        Principal = {
          AWS = "arn:aws:iam::${var.account_id}:root"
        }
        Action   = "kms:*"
        Resource = "*"
      },
      {
        Sid    = "AllowAppAccess"
        Effect = "Allow"
        Principal = {
          AWS = aws_iam_role.app_role.arn
        }
        Action = [
          "kms:Encrypt",
          "kms:Decrypt",
          "kms:GenerateDataKey"
        ]
        Resource = "*"
      }
    ]
  })
}

# S3存储桶加密
resource "aws_s3_bucket_server_side_encryption_configuration" "main" {
  bucket = aws_s3_bucket.data.id

  rule {
    apply_server_side_encryption_by_default {
      sse_algorithm     = "aws:kms"
      kms_master_key_id = aws_kms_key.app_key.arn
    }
    bucket_key_enabled = true
  }
}

# RDS加密
resource "aws_db_instance" "main" {
  engine               = "postgres"
  instance_class       = "db.t3.medium"
  storage_encrypted    = true
  kms_key_id           = aws_kms_key.app_key.arn
  ca_cert_identifier   = "rds-ca-rsa2048-g1"
}
```

### 应用层加密

```python
# 字段级加密
from cryptography.fernet import Fernet
from cryptography.hazmat.primitives import hashes
from cryptography.hazmat.primitives.kdf.pbkdf2 import PBKDF2HMAC
import base64

class FieldEncryptor:
    def __init__(self, master_key: str, salt: bytes):
        kdf = PBKDF2HMAC(
            algorithm=hashes.SHA256(),
            length=32,
            salt=salt,
            iterations=100000,
        )
        key = base64.urlsafe_b64encode(kdf.derive(master_key.encode()))
        self.cipher = Fernet(key)

    def encrypt(self, plaintext: str) -> str:
        return self.cipher.encrypt(plaintext.encode()).decode()

    def decrypt(self, ciphertext: str) -> str:
        return self.cipher.decrypt(ciphertext.encode()).decode()

# 使用示例
encryptor = FieldEncryptor(
    master_key=os.environ['MASTER_KEY'],
    salt=os.environ['ENCRYPTION_SALT'].encode()
)

# 数据库存储加密字段
encrypted_ssn = encryptor.encrypt("123-45-6789")
# 查询时解密
original_ssn = encryptor.decrypt(encrypted_ssn)
```

## 合规与审计

### AWS CloudTrail配置

```hcl
resource "aws_cloudtrail" "main" {
  name                          = "security-trail"
  s3_bucket_name                = aws_s3_bucket.audit_logs.id
  include_global_service_events = true
  is_multi_region_trail         = true
  enable_log_file_validation    = true
  kms_key_id                    = aws_kms_key.audit_key.arn

  event_selector {
    read_write_type           = "All"
    include_management_events = true

    data_resource {
      type   = "AWS::S3::Object"
      values = ["${aws_s3_bucket.sensitive_data.arn}/"]
    }
  }

  insight_selector {
    insight_type = "ApiCallRateInsight"
  }
}

# S3存储桶策略 - 保护审计日志
resource "aws_s3_bucket_policy" "audit_logs" {
  bucket = aws_s3_bucket.audit_logs.id

  policy = jsonencode({
    Version = "2012-10-17"
    Statement = [
      {
        Sid       = "DenyDelete"
        Effect    = "Deny"
        Principal = "*"
        Action    = "s3:Delete*"
        Resource  = "${aws_s3_bucket.audit_logs.arn}/*"
      },
      {
        Sid       = "DenyUnencryptedUploads"
        Effect    = "Deny"
        Principal = "*"
        Action    = "s3:PutObject"
        Resource  = "${aws_s3_bucket.audit_logs.arn}/*"
        Condition = {
          StringNotEquals = {
            "s3:x-amz-server-side-encryption" = "aws:kms"
          }
        }
      }
    ]
  })
}
```

### 合规检查自动化

```python
# AWS Config规则检查
import boto3

def check_compliance():
    config = boto3.client('config')

    # 查询不合规资源
    response = config.get_compliance_details_by_config_rule(
        ConfigRuleName='encrypted-volumes',
        ComplianceTypes=['NON_COMPLIANT']
    )

    non_compliant = []
    for item in response['EvaluationResults']:
        non_compliant.append({
            'resource_type': item['EvaluationResultIdentifier']['EvaluationResultQualifier']['ResourceType'],
            'resource_id': item['EvaluationResultIdentifier']['EvaluationResultQualifier']['ResourceId'],
            'annotation': item.get('Annotation', 'No encryption')
        })

    return non_compliant
```

## 最佳实践总结

### 安全分层模型

```
┌─────────────────────────────────────┐
│        应用安全 (WAF/防护)          │
├─────────────────────────────────────┤
│        身份与访问管理 (IAM)         │
├─────────────────────────────────────┤
│        网络安全 (VPC/SG/NACL)       │
├─────────────────────────────────────┤
│        数据安全 (加密/备份)         │
├─────────────────────────────────────┤
│        基础设施安全 (补丁/硬化)     │
├─────────────────────────────────────┤
│        监控与审计 (CloudTrail/SIEM) │
└─────────────────────────────────────┘
```

| 领域 | 关键实践 | 工具 |
|------|----------|------|
| IAM | 最小权限、MFA、角色轮换 | AWS IAM, Azure AD |
| 网络 | VPC隔离、安全组、WAF | AWS VPC, Azure VNet |
| 加密 | 静态加密、传输加密、密钥轮换 | KMS, Vault |
| 审计 | 日志记录、合规检查、告警 | CloudTrail, Config |
| 备份 | 跨区域备份、加密备份、定期恢复测试 | S3, Azure Backup |
