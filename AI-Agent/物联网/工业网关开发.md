## 协议转换

### 常见工业协议

| 协议 | 层级 | 特点 | 典型设备 |
|------|------|------|----------|
| **Modbus RTU** | 串行 (RS485) | 简单、广泛使用 | 传感器、变频器 |
| **Modbus TCP** | 以太网 | Modbus 的网络版 | PLC、仪表 |
| **OPC-UA** | 以太网 | 工业标准、信息建模 | 高端PLC、DCS |
| **BACnet** | 以太网/串行 | 楼宇自动化 | 暖通空调 |
| **CANopen** | CAN总线 | 汽车工业 | 运动控制 |
| **EtherCAT** | 以太网 | 实时性高 | 伺服驱动 |
| **PROFINET** | 以太网 | 西门子生态 | 西门子设备 |

## Modbus 协议详解

### 核心概念

| 概念 | 说明 |
|------|------|
| **Slave/Server** | 从站设备地址 (1-247) |
| **Function Code** | 功能码 (读/写线圈/寄存器) |
| **Register** | 寄存器 (16位) |
| **Coil** | 线圈 (1位，开关量) |

### 常用功能码

| 功能码 | 操作 | 说明 |
|--------|------|------|
| 0x01 | Read Coils | 读线圈（开关状态） |
| 0x03 | Read Holding Registers | 读保持寄存器 |
| 0x04 | Read Input Registers | 读输入寄存器 |
| 0x05 | Write Single Coil | 写单个线圈 |
| 0x06 | Write Single Register | 写单个寄存器 |
| 0x10 | Write Multiple Registers | 写多个寄存器 |

### 寄存器地址映射示例

```
温度传感器 (Slave Address: 1)
┌────────────────┬──────────┬─────────────────────┐
│ 寄存器地址      │ 功能码   │ 说明                 │
├────────────────┼──────────┼─────────────────────┤
│ 40001 (0x0000) │ 0x03     │ 当前温度 (×0.1°C)   │
│ 40002 (0x0001) │ 0x03     │ 当前湿度 (×0.1%)    │
│ 40003 (0x0002) │ 0x03     │ 露点温度 (×0.1°C)   │
│ 40010 (0x0009) │ 0x03     │ 温度报警阈值         │
│ 40010 (0x0009) │ 0x06     │ 设置温度报警阈值     │
└────────────────┴──────────┴─────────────────────┘
```

### Python Modbus 示例

```python
"""
Modbus TCP/RTU 通信
pip install pymodbus
"""
from pymodbus.client import ModbusTcpClient, ModbusSerialClient
from pymodbus.exceptions import ModbusException
from datetime import datetime, timezone
import struct
import logging

logging.getLogger("pymodbus").setLevel(logging.WARNING)


# ─── Modbus TCP ───
class ModbusTCPDevice:
    def __init__(self, host: str, port: int = 502, unit_id: int = 1):
        self.client = ModbusTcpClient(host, port=port, timeout=3)
        self.unit_id = unit_id

    def read_temperature(self) -> float:
        """读取保持寄存器 40001 (地址 0)"""
        try:
            result = self.client.read_holding_registers(
                address=0, count=2, slave=self.unit_id
            )
            if result.isError():
                raise ModbusException(f"Read error: {result}")

            # 将两个16位寄存器合并为32位浮点数
            raw = struct.pack('>HH', result.registers[0], result.registers[1])
            value = struct.unpack('>f', raw)[0]
            return round(value, 2)
        except Exception as e:
            print(f"[ERROR] Modbus read failed: {e}")
            return None

    def write_threshold(self, value: float):
        """写入温度阈值到寄存器 40010 (地址 9)"""
        raw = struct.pack('>f', value)
        registers = struct.unpack('>HH', raw)

        result = self.client.write_registers(
            address=9, values=list(registers), slave=self.unit_id
        )
        if result.isError():
            raise ModbusException(f"Write error: {result}")

    def close(self):
        self.client.close()


# ─── Modbus RTU (串口) ───
class ModbusRTUDevice:
    def __init__(self, port: str = "/dev/ttyUSB0", baudrate: int = 9600,
                 unit_id: int = 1):
        self.client = ModbusSerialClient(
            port=port,
            baudrate=baudrate,
            bytesize=8,
            parity='N',
            stopbits=1,
            timeout=1
        )
        self.unit_id = unit_id

    def connect(self):
        if not self.client.connect():
            raise ConnectionError(f"Cannot open {self.client.port}")

    def read_holding_registers(self, start_addr: int, count: int) -> list:
        """读取保持寄存器"""
        result = self.client.read_holding_registers(
            address=start_addr, count=count, slave=self.unit_id
        )
        if result.isError():
            raise ModbusException(str(result))
        return result.registers

    def read_coils(self, start_addr: int, count: int) -> list:
        """读取线圈状态"""
        result = self.client.read_coils(
            address=start_addr, count=count, slave=self.unit_id
        )
        if result.isError():
            raise ModbusException(str(result))
        return result.bits[:count]

    def write_single_coil(self, addr: int, value: bool):
        """写单个线圈"""
        result = self.client.write_coil(
            address=addr, value=value, slave=self.unit_id
        )
        if result.isError():
            raise ModbusException(str(result))

    def close(self):
        self.client.close()


# ─── 使用示例 ───
# Modbus TCP
tcp_dev = ModbusTCPDevice("192.168.1.100", port=502, unit_id=1)
temp = tcp_dev.read_temperature()
print(f"Temperature: {temp}°C")
tcp_dev.write_threshold(35.0)
tcp_dev.close()

# Modbus RTU
rtu_dev = ModbusRTUDevice("/dev/ttyUSB0", baudrate=9600, unit_id=1)
rtu_dev.connect()
registers = rtu_dev.read_holding_registers(0, 4)
print(f"Registers: {registers}")
rtu_dev.close()
```

## 数据采集引擎

### 采集任务配置

```python
"""
数据采集引擎 - 基于 YAML 配置驱动
"""
import yaml
from dataclasses import dataclass, field
from typing import List, Optional, Callable
import threading
import time
from datetime import datetime, timezone
import json

# ─── 采集配置 ───
COLLECTION_CONFIG = """
devices:
  - name: "温度传感器-车间A"
    protocol: modbus_tcp
    connection:
      host: "192.168.1.100"
      port: 502
      unit_id: 1
    poll_interval: 5  # 秒
    tags:
      - name: temperature
        address: 0
        count: 2
        data_type: float32
        scale: 1.0
        unit: "°C"
      - name: humidity
        address: 2
        count: 2
        data_type: float32
        scale: 1.0
        unit: "%"

  - name: "电表-总表"
    protocol: modbus_tcp
    connection:
      host: "192.168.1.200"
      port: 502
      unit_id: 2
    poll_interval: 30
    tags:
      - name: voltage
        address: 0
        count: 2
        data_type: float32
        scale: 0.1
        unit: "V"
      - name: current
        address: 2
        count: 2
        data_type: float32
        scale: 0.01
        unit: "A"
      - name: power
        address: 4
        count: 2
        data_type: float32
        scale: 0.1
        unit: "kW"
      - name: energy
        address: 6
        count: 4
        data_type: float64
        scale: 1.0
        unit: "kWh"
"""


@dataclass
class TagConfig:
    name: str
    address: int
    count: int
    data_type: str
    scale: float = 1.0
    offset: float = 0.0
    unit: str = ""

@dataclass
class DeviceConfig:
    name: str
    protocol: str
    connection: dict
    poll_interval: float
    tags: List[TagConfig] = field(default_factory=list)


class DataCollectionEngine:
    def __init__(self, config_yaml: str):
        self.config = yaml.safe_load(config_yaml)
        self.devices: List[DeviceConfig] = []
        self.collected_callbacks: List[Callable] = []
        self._running = False
        self._threads: List[threading.Thread] = []

        self._parse_config()

    def _parse_config(self):
        for dev_cfg in self.config["devices"]:
            tags = [
                TagConfig(**tag) for tag in dev_cfg["tags"]
            ]
            dev = DeviceConfig(
                name=dev_cfg["name"],
                protocol=dev_cfg["protocol"],
                connection=dev_cfg["connection"],
                poll_interval=dev_cfg["poll_interval"],
                tags=tags
            )
            self.devices.append(dev)

    def on_data(self, callback: Callable):
        """注册数据回调"""
        self.collected_callbacks.append(callback)

    def _convert_value(self, registers: list, tag: TagConfig):
        """寄存器值转换为实际工程值"""
        import struct

        if tag.data_type == "float32":
            raw = struct.pack('>HH', registers[0], registers[1])
            value = struct.unpack('>f', raw)[0]
        elif tag.data_type == "float64":
            raw = struct.pack('>HHHH', *registers[:4])
            value = struct.unpack('>d', raw)[0]
        elif tag.data_type == "int16":
            value = registers[0]
            if value > 32767:
                value -= 65536
        elif tag.data_type == "uint16":
            value = registers[0]
        elif tag.data_type == "int32":
            raw = struct.pack('>HH', registers[0], registers[1])
            value = struct.unpack('>i', raw)[0]
        else:
            value = registers[0]

        # 应用缩放和偏移
        return round(value * tag.scale + tag.offset, 4)

    def _poll_device(self, device: DeviceConfig):
        """轮询单个设备"""
        # 创建协议客户端
        if device.protocol == "modbus_tcp":
            client = ModbusTCPDevice(
                device.connection["host"],
                device.connection.get("port", 502),
                device.connection.get("unit_id", 1)
            )
        else:
            print(f"Unsupported protocol: {device.protocol}")
            return

        while self._running:
            try:
                data = {"device_name": device.name}
                for tag in device.tags:
                    result = client.client.read_holding_registers(
                        address=tag.address,
                        count=tag.count,
                        slave=client.unit_id
                    )
                    if not result.isError():
                        value = self._convert_value(result.registers, tag)
                        data[tag.name] = {
                            "value": value,
                            "unit": tag.unit,
                            "quality": "good"
                        }
                    else:
                        data[tag.name] = {
                            "value": None,
                            "quality": "bad",
                            "error": str(result)
                        }

                data["timestamp"] = datetime.now(timezone.utc).isoformat()

                # 触发回调
                for cb in self.collected_callbacks:
                    cb(device, data)

            except Exception as e:
                print(f"[ERROR] Polling {device.name}: {e}")

            time.sleep(device.poll_interval)

    def start(self):
        """启动采集引擎"""
        self._running = True
        for device in self.devices:
            t = threading.Thread(
                target=self._poll_device,
                args=(device,),
                daemon=True,
                name=f"poll_{device.name}"
            )
            t.start()
            self._threads.append(t)
            print(f"[START] Polling {device.name} every {device.poll_interval}s")

    def stop(self):
        """停止采集"""
        self._running = False
        for t in self._threads:
            t.join(timeout=5)
        print("[STOP] Collection engine stopped")


# ─── 使用示例 ───
def data_handler(device, data):
    """采集数据回调"""
    print(f"[{data['timestamp']}] {device.name}:")
    for key, val in data.items():
        if key not in ("timestamp", "device_name"):
            if isinstance(val, dict):
                print(f"  {key}: {val.get('value')} {val.get('unit', '')} [{val.get('quality')}]")

engine = DataCollectionEngine(COLLECTION_CONFIG)
engine.on_data(data_handler)
engine.start()

try:
    while True:
        time.sleep(1)
except KeyboardInterrupt:
    engine.stop()
```

## 边缘计算

### 边缘计算能力

| 能力 | 说明 | 示例 |
|------|------|------|
| **数据过滤** | 过滤无效/重复数据 | 温度变化 < 0.1°C 不上报 |
| **数据聚合** | 多点数据合并 | 10个传感器取平均值 |
| **公式计算** | 边缘侧计算派生指标 | 功率 = 电压 × 电流 |
| **异常检测** | 本地实时告警 | 温度 > 阈值 → 立即报警 |
| **AI推理** | 边缘AI模型 | 轴承故障预测 |
| **规则引擎** | 条件触发动作 | 温度 > 80°C → 关闭设备 |

### 边缘计算引擎实现

```python
"""
边缘计算引擎 - 在网关本地执行数据处理
"""
from dataclasses import dataclass
from typing import Any, List, Callable
import re
import operator

@dataclass
class EdgeRule:
    """边缘计算规则"""
    name: str
    condition: str  # 条件表达式，如 "temperature > 80"
    action: str     # 动作：report, alarm, filter, aggregate
    params: dict    # 动作参数
    enabled: bool = True


class EdgeComputeEngine:
    def __init__(self):
        self.rules: List[EdgeRule] = []
        self.transformers: List[Callable] = []
        self.last_values: dict = {}  # 用于变化检测

    def add_rule(self, rule: EdgeRule):
        self.rules.append(rule)

    def add_transformer(self, func: Callable):
        """添加数据变换器"""
        self.transformers.append(func)

    def process(self, device_name: str, data: dict) -> dict:
        """处理采集数据"""
        result = {
            "device": device_name,
            "original": dict(data),
            "computed": {},
            "actions": [],
            "filtered": False
        }

        # 1. 数据变换
        for transformer in self.transformers:
            data = transformer(device_name, data)

        # 2. 规则评估
        for rule in self.rules:
            if not rule.enabled:
                continue

            if self._evaluate_condition(rule.condition, data):
                action_result = self._execute_action(rule, data)
                result["actions"].append({
                    "rule": rule.name,
                    "action": rule.action,
                    "result": action_result
                })

                if rule.action == "filter":
                    result["filtered"] = True

        # 3. 变化检测
        changes = self._detect_changes(device_name, data)
        result["computed"] = {**data, **changes}

        return result

    def _evaluate_condition(self, condition: str, data: dict) -> bool:
        """安全地评估条件表达式"""
        # 提取变量名
        try:
            # 创建安全上下文
            safe_vars = {k: v.get("value", v) if isinstance(v, dict) else v
                        for k, v in data.items()}
            # 安全的比较操作符替换
            expr = condition.replace("AND", "and").replace("OR", "or")
            return bool(eval(expr, {"__builtins__": {}}, safe_vars))
        except:
            return False

    def _execute_action(self, rule: EdgeRule, data: dict):
        """执行规则动作"""
        if rule.action == "alarm":
            return {"level": rule.params.get("level", "warning"),
                    "message": rule.params.get("message", "")}
        elif rule.action == "filter":
            return {"reason": rule.params.get("reason", "filtered by rule")}
        elif rule.action == "report":
            return {"topic": rule.params.get("topic")}
        return None

    def _detect_changes(self, device_name: str, data: dict) -> dict:
        """检测数据变化"""
        changes = {}
        key = device_name

        for field, value in data.items():
            if field in ("timestamp", "device_name"):
                continue

            actual_value = value.get("value", value) if isinstance(value, dict) else value
            last_key = f"{key}.{field}"
            last_value = self.last_values.get(last_key)

            if last_value is not None and actual_value is not None:
                if isinstance(actual_value, (int, float)):
                    change_rate = abs(actual_value - last_value) / max(abs(last_value), 0.001)
                    changes[f"{field}_change"] = round(change_rate, 4)

            if actual_value is not None:
                self.last_values[last_key] = actual_value

        return changes


# ─── 使用示例 ───
engine = EdgeComputeEngine()

# 添加边缘规则
engine.add_rule(EdgeRule(
    name="高温报警",
    condition="temperature > 80",
    action="alarm",
    params={"level": "critical", "message": "温度超过80°C！"}
))

engine.add_rule(EdgeRule(
    name="数据过滤-无变化",
    condition="temperature_change < 0.001",
    action="filter",
    params={"reason": "temperature change too small"}
))

engine.add_rule(EdgeRule(
    name="功率计算",
    condition="True",  # 总是执行
    action="report",
    params={"topic": "edge/power"}
))

# 添加变换器：计算派生指标
def compute_power(device_name, data):
    v = data.get("voltage", {})
    i = data.get("current", {})
    if isinstance(v, dict): v = v.get("value")
    if isinstance(i, dict): i = i.get("value")
    if v and i:
        data["computed_power"] = {"value": round(v * i, 2), "unit": "W"}
    return data

engine.add_transformer(compute_power)

# 处理数据
sample_data = {
    "temperature": {"value": 85.2, "unit": "°C", "quality": "good"},
    "humidity": {"value": 60.0, "unit": "%", "quality": "good"},
    "voltage": {"value": 220.0, "unit": "V", "quality": "good"},
    "current": {"value": 5.5, "unit": "A", "quality": "good"},
}

result = engine.process("设备A", sample_data)
print(f"Actions: {result['actions']}")
print(f"Computed: {result['computed']}")
```

## 本地缓存与断点续传

### 断点续传机制

```
网络正常时：
设备数据 → 采集引擎 → 边缘计算 → MQTT 上报 → 云端

网络断开时：
设备数据 → 采集引擎 → 边缘计算 → 本地缓存 (SQLite)
                                    ↓ (累积数据)
网络恢复时：
缓存数据 → 逐批上报 → 云端 → 清除已确认缓存
```

### 实现

```python
"""
本地缓存与断点续传
"""
import sqlite3
import json
import threading
from queue import Queue
from datetime import datetime, timezone
from collections import deque
import paho.mqtt.client as mqtt


class OfflineBuffer:
    """离线数据缓存（基于 SQLite）"""

    def __init__(self, db_path: str = "/data/gateway_buffer.db", max_size: int = 100000):
        self.db_path = db_path
        self.max_size = max_size
        self._init_db()

    def _init_db(self):
        conn = sqlite3.connect(self.db_path)
        conn.execute("""
            CREATE TABLE IF NOT EXISTS data_buffer (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                device_name TEXT NOT NULL,
                topic TEXT NOT NULL,
                payload TEXT NOT NULL,
                timestamp TEXT NOT NULL,
                synced INTEGER DEFAULT 0,
                created_at TEXT DEFAULT (datetime('now'))
            );
        """)
        conn.execute("CREATE INDEX IF NOT EXISTS idx_synced ON data_buffer(synced);")
        conn.commit()
        conn.close()

    def push(self, device_name: str, topic: str, payload: dict):
        """缓存数据"""
        conn = sqlite3.connect(self.db_path)
        conn.execute(
            "INSERT INTO data_buffer (device_name, topic, payload, timestamp) VALUES (?, ?, ?, ?)",
            (device_name, topic, json.dumps(payload),
             datetime.now(timezone.utc).isoformat())
        )
        conn.commit()
        conn.close()

    def get_unsynced(self, limit: int = 100) -> list:
        """获取未同步的数据"""
        conn = sqlite3.connect(self.db_path)
        cursor = conn.execute(
            "SELECT id, device_name, topic, payload, timestamp FROM data_buffer WHERE synced = 0 ORDER BY id LIMIT ?",
            (limit,)
        )
        rows = cursor.fetchall()
        conn.close()
        return [
            {"id": r[0], "device_name": r[1], "topic": r[2],
             "payload": json.loads(r[3]), "timestamp": r[4]}
            for r in rows
        ]

    def mark_synced(self, ids: list):
        """标记已同步"""
        if not ids:
            return
        conn = sqlite3.connect(self.db_path)
        placeholders = ",".join("?" * len(ids))
        conn.execute(f"UPDATE data_buffer SET synced = 1 WHERE id IN ({placeholders})", ids)
        conn.commit()
        conn.close()

    def cleanup(self):
        """清理已同步的旧数据"""
        conn = sqlite3.connect(self.db_path)
        conn.execute("DELETE FROM data_buffer WHERE synced = 1 AND created_at < datetime('now', '-7 days')")
        conn.commit()
        conn.close()

    def count(self) -> dict:
        conn = sqlite3.connect(self.db_path)
        total = conn.execute("SELECT COUNT(*) FROM data_buffer").fetchone()[0]
        unsynced = conn.execute("SELECT COUNT(*) FROM data_buffer WHERE synced = 0").fetchone()[0]
        conn.close()
        return {"total": total, "unsynced": unsynced}


class GatewayMQTTClient:
    """支持断点续传的 MQTT 客户端"""

    def __init__(self, broker: str, port: int, buffer: OfflineBuffer):
        self.buffer = buffer
        self.online = False
        self.mqtt = mqtt.Client(client_id="gateway_001")
        self.mqtt.on_connect = self._on_connect
        self.mqtt.on_disconnect = self._on_disconnect
        self.mqtt.on_publish = self._on_publish
        self._pending = {}  # mid → data_id

        self.broker = broker
        self.port = port

    def _on_connect(self, client, userdata, flags, rc):
        self.online = True
        print(f"[MQTT] Connected (rc={rc}), starting sync...")
        threading.Thread(target=self._sync_buffered, daemon=True).start()

    def _on_disconnect(self, client, userdata, rc):
        self.online = False
        print(f"[MQTT] Disconnected (rc={rc})")

    def _on_publish(self, client, userdata, mid):
        """消息发布成功"""
        if mid in self._pending:
            data_id = self._pending.pop(mid)
            self.buffer.mark_synced([data_id])

    def connect(self):
        self.mqtt.connect(self.broker, self.port, keepalive=60)
        self.mqtt.loop_start()

    def publish_data(self, device_name: str, topic: str, payload: dict):
        """发布数据（自动处理在线/离线）"""
        if self.online:
            info = self.mqtt.publish(topic, json.dumps(payload), qos=1)
            # 追踪未确认的消息
            self._pending[info.mid] = -1  # -1 表示直接发送（无缓存ID）
        else:
            # 离线：存入缓存
            self.buffer.push(device_name, topic, payload)

    def _sync_buffered(self):
        """同步缓存数据到云端"""
        while self.online:
            batch = self.buffer.get_unsynced(limit=50)
            if not batch:
                break

            for item in batch:
                if not self.online:
                    break
                info = self.mqtt.publish(
                    item["topic"], json.dumps(item["payload"]), qos=1
                )
                self._pending[info.mid] = item["id"]

            # 等待 ACK
            import time
            time.sleep(2)

        count = self.buffer.count()
        print(f"[SYNC] Buffer status: {count}")


# ─── 完整网关数据流 ───
buffer = OfflineBuffer("/tmp/gateway_buffer.db")
mqtt_client = GatewayMQTTClient("localhost", 1883, buffer)
mqtt_client.connect()

# 模拟采集数据上报

        "device": "温度传感器A",
        "temperature": 25.0 + i * 0.5,
        "timestamp": datetime.now(timezone.utc).isoformat()
    }
    mqtt_client.publish_data("温度传感器A", "gateway/data/temp_sensor_a", data)
    import time; time.sleep(0.5)

print(buffer.count())
```

```python
"""
完整的工业网关应用
集成：数据采集 + 边缘计算 + 本地缓存 + MQTT 上报
"""
import signal
import sys
import time
import json
from datetime import datetime, timezone


class IndustrialGateway:
    def __init__(self, config_path: str = "gateway_config.yaml"):
        # 加载配置
        with open(config_path) as f:
            self.config = yaml.safe_load(f)

        # 初始化组件
        self.collection = DataCollectionEngine(open(config_path).read())
        self.edge = EdgeComputeEngine()
        self.buffer = OfflineBuffer(self.config.get("buffer_path", "/tmp/gw.db"))
        self.mqtt = GatewayMQTTClient(
            self.config["mqtt"]["broker"],
            self.config["mqtt"]["port"],
            self.buffer
        )

        # 注册数据处理流水线
        self.collection.on_data(self._on_data_collected)
        self._setup_edge_rules()

        self._running = True

    def _setup_edge_rules(self):
        """配置边缘规则"""
        for rule_cfg in self.config.get("edge_rules", []):
            self.edge.add_rule(EdgeRule(**rule_cfg))

    def _on_data_collected(self, device, data):
        """数据采集 → 边缘计算 → 上报"""
        # 1. 边缘计算
        result = self.edge.process(device.name, data)

        if result["filtered"]:
            return  # 被过滤，不上报

        # 2. 构建上报消息
        payload = {
            "gateway_id": self.config["gateway_id"],
            "device": device.name,
            "data": result["computed"],
            "actions": result["actions"],
            "timestamp": datetime.now(timezone.utc).isoformat()
        }

        # 3. 发布到 MQTT（自动处理离线缓存）
        topic = f"gateway/{self.config['gateway_id']}/data"
        self.mqtt.publish_data(device.name, topic, payload)

    def start(self):
        """启动网关"""
        print("=" * 60)
        print(f"  Industrial Gateway Starting...")
        print(f"  Gateway ID: {self.config['gateway_id']}")
        print(f"  Devices: {len(self.collection.devices)}")
        print("=" * 60)

        # 启动 MQTT
        self.mqtt.connect()
        time.sleep(2)  # 等待连接

        # 启动采集
        self.collection.start()

        # 主循环
        try:
            while self._running:
                time.sleep(10)
                # 定期清理缓存
                self.buffer.cleanup()
                # 打印状态
                count = self.buffer.count()
                if count["unsynced"] > 0:
                    print(f"[STATUS] Buffer: {count}")
        except KeyboardInterrupt:
            self.stop()

    def stop(self):
        """停止网关"""
        print("\n[SHUTDOWN] Stopping gateway...")
        self._running = False
        self.collection.stop()
        time.sleep(2)
        print("[SHUTDOWN] Done.")


if __name__ == "__main__":
    gateway = IndustrialGateway("gateway_config.yaml")

    def signal_handler(sig, frame):
        gateway.stop()
        sys.exit(0)

    signal.signal(signal.SIGINT, signal_handler)
    signal.signal(signal.SIGTERM, signal_handler)

    gateway.start()
```

## 最佳实践

### 1. 协议适配设计
- 使用 **配置驱动** 而非硬编码设备地址
- 支持 **热加载** 配置（不重启网关即可添加设备）
- 为每种协议实现统一的 **驱动接口**

```python
# 统一驱动接口
class ProtocolDriver(ABC):
    @abstractmethod
    def connect(self): pass
    @abstractmethod
    def read(self, address: int, count: int) -> list: pass
    @abstractmethod
    def write(self, address: int, values: list): pass
    @abstractmethod
    def close(self): pass
```

### 2. 数据采集优化
- **批量读取**：一次性读取连续寄存器，减少通信次数
- **智能轮询**：高频变化的数据快采，低频数据慢采
- **连接池**：复用 TCP 连接，避免频繁建连

### 3. 可靠性保障
- **看门狗**：进程异常退出自动重启
- **心跳上报**：定期上报网关状态
- **配置备份**：配置变更自动备份
- **日志轮转**：防止日志撑满磁盘

### 4. 安全加固
```bash
# 网关安全检查清单
```

|------|--------|----------|
