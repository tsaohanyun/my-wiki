# 智能合约开发

## 概述

智能合约是运行在区块链上的自执行程序，其条款直接写入代码。本指南涵盖 Solidity 语法基础、Remix IDE 使用、Hardhat 开发框架以及 OpenZeppelin 安全库的集成。

## 一、Solidity 语法基础

### 1.1 基本结构

```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.20;

contract SimpleStorage {
    // 状态变量
    uint256 public storedValue;

    // 事件
    event ValueChanged(address indexed setter, uint256 oldValue, uint256 newValue);

    // 修饰器
    modifier nonZero(uint256 _value) {
        require(_value != 0, "Value cannot be zero");
        _;
    }

    // 构造函数
    constructor(uint256 _initialValue) {
        storedValue = _initialValue;
        emit ValueChanged(msg.sender, 0, _initialValue);
    }

    // 写入函数
    function setValue(uint256 _value) external nonZero(_value) {
        uint256 oldValue = storedValue;
        storedValue = _value;
        emit ValueChanged(msg.sender, oldValue, _value);
    }

    // 读取函数（view - 不修改状态）
    function getValue() external view returns (uint256) {
        return storedValue;
    }
}
```

```solidity
pragma solidity ^0.8.20;

contract DataTypes {
    // 值类型
    bool public isActive = true;
    uint256 public count = 42;
    int256 public balance = -100;
    address public owner = msg.sender;
    bytes32 public hash = keccak256(abi.encodePacked("hello"));
    string public name = "MyContract";
    enum Status { Pending, Active, Inactive }
    Status public status = Status.Pending;

    // 引用类型
    uint256[] public numbers;
    mapping(address => uint256) public balances;
    mapping(address => mapping(address => bool)) public isApproved;

    struct User {
        address wallet;
        string name;
        uint256 age;
        bool isVerified;
    }

    mapping(address => User) public users;
    User[] public allUsers;

    // storage vs memory vs calldata
    function processArray(uint256[] calldata _input) external pure returns (uint256 sum) {
        // calldata: 只读，不可修改，gas 成本最低
        for (uint256 i = 0; i < _input.length; i++) {
            sum += _input[i];
        }
    }

    function modifyInMemory(uint256[] memory _data) internal pure returns (uint256[] memory) {
        // memory: 临时数据，可修改，函数结束后销毁
        for (uint256 i = 0; i < _data.length; i++) {
            _data[i] = _data[i] * 2;
        }
        return _data;
    }
}
```

### 1.3 函数可见性与修饰器

```solidity
pragma solidity ^0.8.20;

contract AccessControl {
    address public owner;
    mapping(address => bool) public admins;

    modifier onlyOwner() {
        require(msg.sender == owner, "Only owner can call this");
        _;
    }

    modifier onlyAdmin() {
        require(admins[msg.sender], "Only admin can call this");
        _;
    }

    modifier nonReentrant() {
        bool _locked;
        require(!_locked, "Reentrant call");
        _locked = true;
        _;
        _locked = false;
    }

    constructor() {
        owner = msg.sender;
    }

    // external: 只能从外部调用
    function externalFunc() external pure returns (string memory) {
        return "external";
    }

    // public: 内部和外部都可以调用
    function publicFunc() public pure returns (string memory) {
        return "public";
    }

    // internal: 只能从当前合约及子合约调用
    function internalFunc() internal pure returns (string memory) {
        return "internal";
    }

    // private: 只能从当前合约调用
    function privateFunc() private pure returns (string memory) {
        return "private";
    }
}

// 继承示例
contract ChildContract is AccessControl {
    function testInternal() external pure returns (string memory) {
        // 可以调用父合约的 internal 函数
        // return internalFunc(); // 合法
        return "child";
    }
}
```

### 1.5 错误处理

```solidity
pragma solidity ^0.8.20;

contract ErrorHandling {
    // Solidity 0.8+ 自定义错误（省 gas）
    error InsufficientBalance(uint256 available, uint256 required);
    error Unauthorized(address caller);

    mapping(address => uint256) public balances;
    address public owner;

    constructor() {
        owner = msg.sender;
    }

    // 使用 require（旧方式）
    function withdrawOld(uint256 _amount) external {
        require(balances[msg.sender] >= _amount, "Insufficient balance");
        balances[msg.sender] -= _amount;
        (bool success, ) = msg.sender.call{value: _amount}("");
        require(success, "Transfer failed");
    }

    // 使用自定义错误（推荐，省 gas）
    function withdraw(uint256 _amount) external {
        if (balances[msg.sender] < _amount) {
            revert InsufficientBalance(balances[msg.sender], _amount);
        }
        balances[msg.sender] -= _amount;
        (bool success, ) = msg.sender.call{value: _amount}("");
        require(success, "Transfer failed");
    }

    // try / catch
    function interactWithExternal(address _contractAddr) external returns (bool) {
        try ExternalContract(_contractAddr).doSomething() {
            return true;
        } catch Error(string memory reason) {
            // require/revert 导致的错误
            emit LogError(reason);
            return false;
        } catch Panic(uint256 errorCode) {
            // assert 或算术溢出导致
            emit LogPanic(errorCode);
            return false;
        } catch (bytes memory lowLevelData) {
            // 其他底层错误
            return false;
        }
    }

    event LogError(string reason);
    event LogPanic(uint256 code);
}

interface ExternalContract {
    function doSomething() external;
}
```

### 1.6 合约继承与接口

```solidity
pragma solidity ^0.8.20;

// 接口
interface IERC20 {
    function totalSupply() external view returns (uint256);
    function balanceOf(address account) external view returns (uint256);
    function transfer(address to, uint256 amount) external returns (bool);
    function approve(address spender, uint256 amount) external returns (bool);
    function transferFrom(address from, address to, uint256 amount) external returns (bool);
    event Transfer(address indexed from, address indexed to, uint256 value);
    event Approval(address indexed owner, address indexed spender, uint256 value);
}

// 抽象合约
abstract contract Ownable {
    address public owner;

    constructor() {
        owner = msg.sender;
    }

    modifier onlyOwner() {
        require(msg.sender == owner, "Not owner");
        _;
    }

    function transferOwnership(address newOwner) external onlyOwner {
        require(newOwner != address(0), "Zero address");
        owner = newOwner;
    }
}

// 多重继承
contract ERC20Token is Ownable, IERC20 {
    string public name;
    string public symbol;
    uint8 public constant decimals = 18;
    uint256 private _totalSupply;

    mapping(address => uint256) private _balances;
    mapping(address => mapping(address => uint256)) private _allowances;

    constructor(string memory _name, string memory _symbol, uint256 _initialSupply) {
        name = _name;
        symbol = _symbol;
        _totalSupply = _initialSupply;
        _balances[msg.sender] = _initialSupply;
        emit Transfer(address(0), msg.sender, _initialSupply);
    }

    function totalSupply() external view returns (uint256) {
        return _totalSupply;
    }

    function balanceOf(address account) external view returns (uint256) {
        return _balances[account];
    }

    function transfer(address to, uint256 amount) external returns (bool) {
        _transfer(msg.sender, to, amount);
        return true;
    }

    function approve(address spender, uint256 amount) external returns (bool) {
        _allowances[msg.sender][spender] = amount;
        emit Approval(msg.sender, spender, amount);
        return true;
    }

    function transferFrom(address from, address to, uint256 amount) external returns (bool) {
        uint256 allowed = _allowances[from][msg.sender];
        require(allowed >= amount, "Allowance exceeded");
        _allowances[from][msg.sender] = allowed - amount;
        _transfer(from, to, amount);
        return true;
    }

    function _transfer(address from, address to, uint256 amount) internal {
        require(_balances[from] >= amount, "Insufficient balance");
        _balances[from] -= amount;
        _balances[to] += amount;
        emit Transfer(from, to, amount);
    }

    // 仅 owner 可铸造
    function mint(address to, uint256 amount) external onlyOwner {
        _totalSupply += amount;
        _balances[to] += amount;
        emit Transfer(address(0), to, amount);
    }
}
```

## 二、Remix IDE

[Remix](https://remix.ethereum.org/) 是以太坊官方推荐的在线智能合约 IDE，适合初学者快速上手和原型验证。

### 2.1 核心功能

| 功能 | 说明 |
|------|------|
| **Solidity 编译器** | 支持多版本一键切换 |
| **部署 & 运行** | 支持 JavaScript VM、注入 MetaMask、连接外部节点 |
| **静态分析** | 安全审计提示 |
| **测试** | 自动生成测试模板 |
| **Debugger** | 交易级调试器 |

### 2.2 Remix 工作流程

### 3.3 编写测试

```javascript
// test/Token.test.js
const { expect } = require("chai");
const { ethers } = require("hardhat");

describe("Token Contract", function () {
  let Token;
  let token;
  let owner;
  let addr1;
  let addr2;
  let addrs;

  beforeEach(async function () {
    Token = await ethers.getContractFactory("Token");
    [owner, addr1, addr2, ...addrs] = await ethers.getSigners();

    token = await Token.deploy("MyToken", "MTK", ethers.parseEther("1000000"));
    await token.waitForDeployment();
  });

  describe("Deployment", function () {
    it("Should set the right owner", async function () {
      expect(await token.owner()).to.equal(owner.address);
    });

    it("Should assign the total supply to owner", async function () {
      const ownerBalance = await token.balanceOf(owner.address);
      expect(await token.totalSupply()).to.equal(ownerBalance);
    });
  });

  describe("Transactions", function () {
    it("Should transfer tokens between accounts", async function () {
      await token.transfer(addr1.address, 50);
      const addr1Balance = await token.balanceOf(addr1.address);
      expect(addr1Balance).to.equal(50);
    });

    it("Should fail if sender doesn't have enough balance", async function () {
      const initialOwnerBalance = await token.balanceOf(owner.address);
      await expect(
        token.connect(addr1).transfer(owner.address, 1)
      ).to.be.revertedWith("Insufficient balance");
      expect(await token.balanceOf(owner.address)).to.equal(initialOwnerBalance);
    });
  });
});
```

```solidity
        _grantRole(MINTER_ROLE, minter);
```

### 4.3 安全工具

```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.20;

import "@openzeppelin/contracts/utils/ReentrancyGuard.sol";
import "@openzeppelin/contracts/access/Ownable.sol";
import "@openzeppelin/contracts/security/Pausable.sol";
import "@openzeppelin/contracts/token/ERC20/utils/SafeERC20.sol";
import "@openzeppelin/contracts/token/ERC20/IERC20.sol";
import "@openzeppelin/contracts/utils/cryptography/MerkleProof.sol";

contract SecureVault is ReentrancyGuard, Ownable, Pausable {
    using SafeERC20 for IERC20;

    mapping(address => uint256) public deposits;
    IERC20 public immutable token;

    // 白名单（Merkle Tree）
    bytes32 public merkleRoot;

    event Deposited(address indexed user, uint256 amount);
    event Withdrawn(address indexed user, uint256 amount);

    constructor(address _token, bytes32 _merkleRoot) Ownable(msg.sender) {
        token = IERC20(_token);
        merkleRoot = _merkleRoot;
    }

    // ReentrancyGuard 防止重入攻击
    function deposit(uint256 _amount) external nonReentrant whenNotPaused {
        require(_amount > 0, "Amount must be > 0");
        token.safeTransferFrom(msg.sender, address(this), _amount);
        deposits[msg.sender] += _amount;
        emit Deposited(msg.sender, _amount);
    }

    function withdraw(uint256 _amount) external nonReentrant whenNotPaused {
        require(deposits[msg.sender] >= _amount, "Insufficient balance");
        deposits[msg.sender] -= _amount;
        token.safeTransfer(msg.sender, _amount);
        emit Withdrawn(msg.sender, _amount);
    }

    // Merkle 白名单验证
    function claim(bytes32[] calldata _proof, uint256 _amount) external nonReentrant {
        bytes32 leaf = keccak256(abi.encode(msg.sender, _amount));
        require(MerkleProof.verify(_proof, merkleRoot, leaf), "Invalid proof");
        // ... claim logic
    }

    function pause() external onlyOwner {
        _pause();
    }

    function unpause() external onlyOwner {
        _unpause();
    }
}
```

## 五、最佳实践

### 5.1 安全最佳实践

```solidity
pragma solidity ^0.8.20;

contract BestPractices {
    // ✅ 使用自定义错误而非 require 字符串（省 gas）
    error InvalidAmount(uint256 provided, uint256 required);

    // ✅ 使用 immutable / constant 节省 gas
    address public immutable owner;
    uint256 public constant MAX_SUPPLY = 1_000_000 * 10**18;

    // ✅ 使用 checked math（Solidity 0.8+ 默认开启溢出检查）
    // 如需 unchecked，需明确标注
    function uncheckedMath(uint256 a, uint256 b) external pure returns (uint256) {
        unchecked {
            return a + b; // 不检查溢出，省 gas
        }
    }

    // ✅ 正确的 CEI 模式 (Checks-Effects-Interactions)
    function withdrawSecure(address _token) external {
        // 1. Checks
        uint256 amount = balances[msg.sender];
        require(amount > 0, "No balance");

        // 2. Effects
        balances[msg.sender] = 0;

        // 3. Interactions
        (bool success, ) = _token.call(abi.encodeWithSignature("transfer(address,uint256)", msg.sender, amount));
        require(success, "Transfer failed");
    }

    mapping(address => uint256) public balances;

    // ✅ 使用 events 便于链下追踪
    event ActionExecuted(address indexed user, uint256 amount, uint256 timestamp);

    function doAction(uint256 _amount) external {
        require(_amount > 0, "Amount must be positive");
        balances[msg.sender] += _amount;
        emit ActionExecuted(msg.sender, _amount, block.timestamp);
    }

    constructor() {
        owner = msg.sender;
    }

    // ✅ 添加紧急停止功能
    bool public stopped = false;

    modifier stopInEmergency { require(!stopped, "Contract stopped"); _; }
    modifier onlyInEmergency { require(stopped, "Contract not stopped"); _; }

    function toggleContractActive() external {
        require(msg.sender == owner, "Only owner");
        stopped = !stopped;
    }
}
```

### 5.2 Gas 优化清单

```solidity
pragma solidity ^0.8.20;

contract GasOptimization {
    // ✅ 打包状态变量以节省存储槽
    struct PackedUser {
        uint64 balance;    // 8 bytes
        uint64 timestamp;  // 8 bytes
        uint32 id;         // 4 bytes
        uint16 level;      // 2 bytes
        bool isActive;     // 1 byte
        // 总计: 23 bytes → 1 个存储槽 (32 bytes)
    }

    // ❌ 不好的做法：分开声明
    // uint64 public balance;
    // uint64 public timestamp;
    // uint32 public id;
    // uint16 public level;
    // bool public isActive;
    // → 占用 3 个存储槽

    // ✅ 使用 mapping 而非 array 进行查找
    mapping(address => uint256) public balances;          // O(1) 查找
    mapping(address => bool) public isWhitelisted;        // O(1) 检查

    // ✅ 使用 calldata 而非 memory 用于只读参数
    function processData(bytes calldata data) external pure returns (bytes32) {
        return keccak256(data);
    }

    // ✅ 使用 immutable 和 constant
    uint256 public constant FEE_PERCENTAGE = 3;  // 不变
    address public immutable factory;              // 构造函数中设置

    constructor() {
        factory = msg.sender;
    }

    // ✅ 短路求值
    function requireConditions(address addr) external view {
        require(
            addr != address(0) && balances[addr] > 0,
            "Invalid address or no balance"
        );
    }

    mapping(address => uint256) balances;

    // ✅ 缓存数组长度
    function sumArray(uint256[] calldata arr) external pure returns (uint256 sum) {
        uint256 len = arr.length; // 缓存到局部变量
        for (uint256 i = 0; i < len; i++) {
            sum += arr[i];
        }
    }

    // ✅ 使用 unchecked 在安全场景下
    function incrementLoop() external {
        for (uint256 i = 0; i < 10; i++) {
            unchecked { ++i; }
        }
    }
}
```

### 5.3 审计检查清单

| 类别 | 检查项 |
|------|--------|
| **重入攻击** | 使用 `nonReentrant`；遵循 CEI 模式 |
| **整数溢出** | Solidity 0.8+ 自动检查；如 unchecked 需谨慎 |
| **访问控制** | 所有状态修改函数加权限检查 |
| **闪电贷攻击** | 价格不依赖单一 DEX；使用 TWAP |
| **Front-running** | 使用 commit-reveal 方案；限制滑点 |
| **存储碰撞** | 升级合约时新增变量加 gap |
| **权限升级** | 多签管理 proxy admin |
| **外部调用** | 标记不可信合约；处理返回值 |
| **Delegatecall** | 谨慎使用，检查存储布局 |
| **拒绝服务** | 避免 for 循环操作外部地址 |
