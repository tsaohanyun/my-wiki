## 2. GPIO 编程

### 2.1 RPi.GPIO 基础

```python
import RPi.GPIO as GPIO
import time

# 引脚编号模式
GPIO.setmode(GPIO.BCM)  # 使用 BCM 编号（GPIO2, GPIO3...）
# GPIO.setmode(GPIO.BOARD)  # 使用物理引脚编号（1, 2, 3...）

# 设置警告（关闭重复设置警告）
GPIO.setwarnings(False)

# LED 引脚
LED_PIN = 18
# 按钮引脚
BUTTON_PIN = 17

# 配置引脚
GPIO.setup(LED_PIN, GPIO.OUT)
GPIO.setup(BUTTON_PIN, GPIO.IN, pull_up_down=GPIO.PUD_UP)

# --- 基本输出 ---
# 点亮 LED
GPIO.output(LED_PIN, GPIO.HIGH)
time.sleep(1)

# 熄灭 LED
GPIO.output(LED_PIN, GPIO.LOW)

# --- 基本输入 ---
# 读取按钮状态
button_state = GPIO.input(BUTTON_PIN)
if button_state == GPIO.LOW:
    print("Button pressed!")

# --- LED 闪烁 ---
try:
    while True:
        GPIO.output(LED_PIN, GPIO.HIGH)
        time.sleep(0.5)
        GPIO.output(LED_PIN, GPIO.LOW)
        time.sleep(0.5)
except KeyboardInterrupt:
    pass
finally:
    GPIO.cleanup()  # 清理所有 GPIO 设置
```

### 2.2 GPIO 中断（事件检测）

```python
import RPi.GPIO as GPIO
import time
from datetime import datetime

GPIO.setmode(GPIO.BCM)

LED_PIN = 18
BUTTON_PIN = 17

GPIO.setup(LED_PIN, GPIO.OUT)
GPIO.setup(BUTTON_PIN, GPIO.IN, pull_up_down=GPIO.PUD_UP)

press_count = 0

# 中断回调函数
def button_callback(channel):
    global press_count
    press_count += 1
    timestamp = datetime.now().strftime("%H:%M:%S.%f")[:-3]
    print(f"[{timestamp}] Button pressed! Count: {press_count}")

    # 切换 LED 状态
    GPIO.output(LED_PIN, not GPIO.input(LED_PIN))

# 添加事件检测（下降沿触发，消抖时间200ms）
GPIO.add_event_detect(
    BUTTON_PIN,
    GPIO.FALLING,
    callback=button_callback,
    bouncetime=200
)

# 使用线程方式处理事件（推荐）
# GPIO.add_event_detect(BUTTON_PIN, GPIO.FALLING, bouncetime=200)

print("Waiting for button press... (Ctrl+C to exit)")
try:
    while True:
        time.sleep(0.1)  # 主循环做其他事情
except KeyboardInterrupt:
    print("\nExiting...")
finally:
    GPIO.remove_event_detect(BUTTON_PIN)
    GPIO.cleanup()
```

### 2.4 多 LED 控制（流水灯）

```python
from gpiozero import LEDBoard
from time import sleep
import itertools

# LED 组
leds = LEDBoard(2, 3, 4, 17, 27, 22, 10, 9)

# 流水灯效果
def running_light(leds, delay=0.1, iterations=5):
    for _ in range(iterations):
        for led in leds:
            led.on()
            sleep(delay)
            led.off()

# 来回扫描
def scanner(leds, delay=0.1, iterations=5):
    for _ in range(iterations):
        for led in leds:
            led.on()
            sleep(delay)
            led.off()
        for led in reversed(leds):
            led.on()
            sleep(delay)
            led.off()

# 二进制计数器
def binary_counter(leds, max_val=256, delay=0.2):
    for i in range(max_val):
        for j, led in enumerate(leds):
            led.value = (i >> j) & 1
        sleep(delay)

try:
    running_light(leds)
    scanner(leds)
    binary_counter(leds)
except KeyboardInterrupt:
    leds.close()
```

### 3.2 Python I2C 读取（BME280 传感器）

```python
import smbus2
import time

# I2C 总线号（树莓派为 1）
bus = smbus2.SMBus(1)

# BME280 地址
BME280_ADDR = 0x76

# 寄存器地址
BME280_REG_ID = 0xD0
BME280_REG_RESET = 0xE0
BME280_REG_CTRL_HUM = 0xF2
BME280_REG_STATUS = 0xF3
BME280_REG_CTRL_MEAS = 0xF4
BME280_REG_CONFIG = 0xF5
BME280_REG_DATA = 0xF7
BME280_REG_CALIB00 = 0x88
BME280_REG_CALIB26 = 0xE1

class BME280:
    def __init__(self, bus_num=1, address=0x76):
        self.bus = smbus2.SMBus(bus_num)
        self.address = address
        self.calib = []
        self._read_calibration()
        self._configure()

    def _read_calibration(self):
        # 读取校准数据
        calib1 = self.bus.read_i2c_block_data(self.address, BME280_REG_CALIB00, 26)
        calib2 = self.bus.read_i2c_block_data(self.address, BME280_REG_CALIB26, 7)

        # 解析校准参数
        self.calib = []
        for i in range(0, 24, 2):
            self.calib.append(calib1[i] | (calib1[i+1] << 8))
        self.calib.append((calib1[24] << 8) | calib1[25])

        # 有符号转换
        self.dig_T1 = self.calib[0]
        self.dig_T2 = self._to_signed(self.calib[1])
        self.dig_T3 = self._to_signed(self.calib[2])
        self.dig_P1 = self.calib[3]
        self.dig_P2 = self._to_signed(self.calib[4])
        # ... 更多校准参数

    def _configure(self):
        # 湿度过采样 x1
        self.bus.write_byte_data(self.address, BME280_REG_CTRL_HUM, 0x01)
        # 温度/压力过采样 x1，正常模式
        self.bus.write_byte_data(self.address, BME280_REG_CTRL_MEAS, 0x27)
        # 待机时间 0.5ms
        self.bus.write_byte_data(self.address, BME280_REG_CONFIG, 0xA0)

    @staticmethod
    def _to_signed(val):
        return val - 65536 if val > 32767 else val

    def read_data(self):
        data = self.bus.read_i2c_block_data(self.address, BME280_REG_DATA, 8)

        # 原始数据
        raw_press = (data[0] << 12) | (data[1] << 4) | (data[2] >> 4)
        raw_temp = (data[3] << 12) | (data[4] << 4) | (data[5] >> 4)
        raw_hum = (data[6] << 8) | data[7]

        # 简化的温度补偿计算
        var1 = ((raw_temp >> 3) - (self.dig_T1 << 1)) * (self.dig_T2 >> 11)
        var2 = (((((raw_temp >> 4) - self.dig_T1) * ((raw_temp >> 4) - self.dig_T1)) >> 12)
                * self.dig_T3) >> 14
        t_fine = var1 + var2
        temperature = (t_fine * 5 + 128) >> 8
        temperature = temperature / 100.0

        # 简化处理，实际需要完整补偿算法
        humidity = raw_hum / 1024.0
        pressure = raw_press / 256.0

        return temperature, pressure, humidity

    def close(self):
        self.bus.close()

# 使用示例
sensor = BME280()
try:
    while True:
        temp, press, hum = sensor.read_data()
        print(f"Temperature: {temp:.1f}°C  |  "
              f"Pressure: {press:.1f} hPa  |  "
              f"Humidity: {hum:.1f}%")
        time.sleep(2)
except KeyboardInterrupt:
    sensor.close()
```

### 3.3 使用 I2C LCD1602

```python
import smbus2
from time import sleep

# PCF8574 I2C转并口（LCD1602）
LCD_ADDR = 0x27
I2C_BUS = 1

class I2CLCD:
    def __init__(self, bus=1, addr=0x27, cols=16, rows=2):
        self.bus = smbus2.SMBus(bus)
        self.addr = addr
        self.cols = cols
        self.rows = rows
        self.backlight = 0x08
        self._init_lcd()

    def _write_byte(self, data, mode=0):
        # mode: 0=command, 1=data
        high = (data & 0xF0) | self.backlight | mode
        low = ((data << 4) & 0xF0) | self.backlight | mode
        self.bus.write_byte(self.addr, high | 0x04)  # EN=1
        sleep(0.0002)
        self.bus.write_byte(self.addr, high & ~0x04)  # EN=0
        self.bus.write_byte(self.addr, low | 0x04)
        sleep(0.0002)
        self.bus.write_byte(self.addr, low & ~0x04)

    def _init_lcd(self):
        sleep(0.05)  # 等待LCD上电
        self._write_byte(0x33)
        self._write_byte(0x32)
        self._write_byte(0x28)  # 4-bit, 2行
        self._write_byte(0x0C)  # 显示开, 光标关
        self._write_byte(0x06)  # 自动递增
        self.clear()

    def clear(self):
        self._write_byte(0x01)
        sleep(0.002)

    def set_cursor(self, col, row):
        offsets = [0x00, 0x40, 0x14, 0x54]
        self._write_byte(0x80 | (offsets[row] + col))

    def print(self, text):
        for char in text:
            self._write_byte(ord(char), mode=1)

    def display(self, line1="", line2=""):
        self.set_cursor(0, 0)
        self.print(line1.ljust(self.cols)[:self.cols])
        self.set_cursor(0, 1)
        self.print(line2.ljust(self.cols)[:self.cols])

    def close(self):
        self.bus.close()

# 使用示例
lcd = I2CLCD()
lcd.display("Hello Pi!", "I2C LCD1602")
sleep(2)
lcd.display("Line 1 Text", "Line 2 Text")
lcd.close()
```

## 4. SPI 通信

### 4.1 SPI 配置与基础

```python
import spidev
import time

# 打开 SPI 设备
spi = spidev.SpiDev()
spi.open(0, 0)  # bus 0, CE0

# 配置 SPI 参数
spi.max_speed_hz = 1000000   # 1MHz
spi.mode = 0b00               # CPOL=0, CPHA=0
spi.bits_per_word = 8
spi.lsbfirst = False           # MSB first

# 简单的 SPI 读写
def spi_transfer(data):
    """发送并接收数据"""
    if isinstance(data, int):
        return spi.xfer2([data])[0]
    elif isinstance(data, (list, bytes)):
        return spi.xfer2(list(data))

# 读写一个字节
response = spi_transfer(0xFF)
print(f"Response: 0x{response:02X}")

spi.close()
```

### 4.2 MCP3008 ADC（SPI 读取模拟量）

```python
import spidev
import time

# MCP3008 是 8通道 10位 SPI ADC
class MCP3008:
    def __init__(self, bus=0, device=0, speed=1000000):
        self.spi = spidev.SpiDev()
        self.spi.open(bus, device)
        self.spi.max_speed_hz = speed
        self.spi.mode = 0

    def read_channel(self, channel):
        """读取指定通道的ADC值 (0-1023)"""
        if not 0 <= channel <= 7:
            raise ValueError("Channel must be 0-7")

        # MCP3008 通信协议
        # 发送: [1, (8+channel)<<4, 0]
        # 接收: [don't care, don't care, lower 10 bits in last 2 bytes]
        adc = self.spi.xfer2([1, (8 + channel) << 4, 0])
        data = ((adc[1] & 3) << 8) + adc[2]
        return data

    def read_voltage(self, channel, vref=3.3):
        """读取通道电压"""
        raw = self.read_channel(channel)
        voltage = (raw * vref) / 1023.0
        return voltage

    def close(self):
        self.spi.close()

# 使用示例：读取光敏电阻和温度传感器
adc = MCP3008()
try:
    while True:
        light_raw = adc.read_channel(0)    # CH0: 光敏电阻
        temp_raw = adc.read_channel(1)     # CH1: 温度传感器
        pot_raw = adc.read_channel(2)      # CH2: 电位器

        light_voltage = adc.read_voltage(0)
        temp_voltage = adc.read_voltage(1)

        # LM35: 10mV/°C
        temperature = temp_voltage * 100

        print(f"Light: {light_raw:4d} ({light_voltage:.2f}V) | "
              f"Temp: {temperature:.1f}°C | "
              f"Pot: {pot_raw:4d}")

        time.sleep(0.5)
except KeyboardInterrupt:
    adc.close()
```

### 4.3 MAX7219 点阵屏驱动

```python
import spidev
from time import sleep

# MAX7219 8x8 LED 点阵
class MAX7219:
    DECODE_MODE = 0x09
    INTENSITY = 0x0A
    SCAN_LIMIT = 0x0B
    SHUTDOWN = 0x0C
    DISPLAY_TEST = 0x0F

    # 笑脸图案
    SMILE = [
        0b00111100,
        0b01000010,
        0b10100101,
        0b10000001,
        0b10100101,
        0b10011001,
        0b01000010,
        0b00111100,
    ]

    def __init__(self, bus=0, device=0):
        self.spi = spidev.SpiDev()
        self.spi.open(bus, device)
        self.spi.max_speed_hz = 10000000
        self._init_display()

    def _send(self, register, data):
        self.spi.xfer2([register, data])

    def _init_display(self):
        self._send(self.SHUTDOWN, 0x01)      # 正常模式
        self._send(self.DISPLAY_TEST, 0x00)   # 关闭测试
        self._send(self.SCAN_LIMIT, 0x07)     # 扫描所有8位
        self._send(self.DECODE_MODE, 0x00)    # 无解码
        self._send(self.INTENSITY, 0x08)      # 中等亮度
        self.clear()

    def clear(self):
        for i in range(8):
            self._send(i + 1, 0)

    def display_matrix(self, matrix):
        for row in range(8):
            self._send(row + 1, matrix[row])

    def set_intensity(self, level):
        self._send(self.INTENSITY, level & 0x0F)

    def close(self):
        self.clear()
        self.spi.close()

# 使用示例
matrix = MAX7219()
matrix.display_matrix(MAX7219.SMILE)
sleep(2)
matrix.clear()
matrix.close()
```

### 5.2 Python 串口通信

```python
import serial
import serial.tools.list_ports
import time

# 列出所有可用串口
ports = serial.tools.list_ports.comports()
for port in ports:
    print(f"{port.device}: {port.description}")

# 打开串口
ser = serial.Serial(
    port='/dev/serial0',
    baudrate=9600,
    bytesize=serial.EIGHTBITS,
    parity=serial.PARITY_NONE,
    stopbits=serial.STOPBITS_ONE,
    timeout=1.0,        # 读超时
    write_timeout=1.0   # 写超时
)

# 检查串口是否打开
if ser.is_open:
    print(f"Serial port {ser.port} opened at {ser.baudrate} baud")

# 发送数据
def send_message(msg):
    if isinstance(msg, str):
        msg = msg.encode('utf-8')
    ser.write(msg)
    ser.flush()

# 接收数据
def receive_message():
    if ser.in_waiting > 0:
        data = ser.read(ser.in_waiting)
        return data.decode('utf-8', errors='replace')
    return None

# 与 Arduino/STM32 通信示例
try:
    while True:
        # 发送命令
        command = input("Enter command (temp/humidity/led_on/led_off): ")
        send_message(command + '\n')

        # 等待响应
        time.sleep(0.1)
        response = receive_message()
        if response:
            print(f"Response: {response}")
except KeyboardInterrupt:
    pass
finally:
    ser.close()
    print("Serial port closed")
```

### 5.3 串口 + 线程（异步接收）

```python
import serial
import threading
import queue
import time

class SerialCommunicator:
    def __init__(self, port='/dev/serial0', baudrate=115200):
        self.ser = serial.Serial(port, baudrate, timeout=0.1)
        self.rx_queue = queue.Queue()
        self.running = False
        self.thread = None

    def _rx_thread(self):
        """接收线程"""
        buffer = b''
        while self.running:
            if self.ser.in_waiting > 0:
                data = self.ser.read(self.ser.in_waiting)
                buffer += data

                # 按换行符分割消息
                while b'\n' in buffer:
                    line, buffer = buffer.split(b'\n', 1)
                    line = line.strip()
                    if line:
                        self.rx_queue.put(line)

            time.sleep(0.01)

    def start(self):
        self.running = True
        self.thread = threading.Thread(target=self._rx_thread, daemon=True)
        self.thread.start()

    def stop(self):
        self.running = False
        if self.thread:
            self.thread.join(timeout=2)
        self.ser.close()

    def send(self, message):
        if isinstance(message, str):
            message = message.encode('utf-8')
        self.ser.write(message + b'\n')

    def get_message(self, timeout=0.1):
        try:
            return self.rx_queue.get(timeout=timeout)
        except queue.Empty:
            return None

# 使用示例
comm = SerialCommunicator('/dev/serial0', 115200)
comm.start()

try:
    # 发送数据
    comm.send("GET_TEMP")

    # 处理接收的数据
    while True:
        msg = comm.get_message()
        if msg:
            print(f"Received: {msg.decode('utf-8', errors='replace')}")
        time.sleep(0.1)
except KeyboardInterrupt:
    comm.stop()
```

### 5.4 GPS 模块（NMEA 解析）

```python
import serial
import time
from datetime import datetime

class GPSReader:
    def __init__(self, port='/dev/serial0', baudrate=9600):
        self.ser = serial.Serial(port, baudrate, timeout=1.0)

    def parse_gga(self, sentence):
        """解析 $GPGGA 句子"""
        # $GPGGA,time,lat,N,lon,E,fix_quality,sats,hdop,alt,M,...
        parts = sentence.split(',')
        if len(parts) < 10:
            return None

        try:
            # 时间
            time_str = parts[1]
            hour = int(time_str[0:2])
            minute = int(time_str[2:4])
            second = int(time_str[4:6])

            # 纬度
            lat_raw = float(parts[2])
            lat_dir = parts[3]
            lat = lat_raw // 100 + (lat_raw % 100) / 60
            if lat_dir == 'S':
                lat = -lat

            # 经度
            lon_raw = float(parts[4])
            lon_dir = parts[5]
            lon = lon_raw // 100 + (lon_raw % 100) / 60
            if lon_dir == 'W':
                lon = -lon

            # 定位质量
            fix_quality = int(parts[6])
            satellites = int(parts[7])

            return {
                'time': f"{hour:02d}:{minute:02d}:{second:02d}",
                'latitude': lat,
                'longitude': lon,
                'fix': fix_quality,
                'satellites': satellites,
            }
        except (ValueError, IndexError):
            return None

    def read(self):
        while True:
            line = self.ser.readline().decode('ascii', errors='replace').strip()
            if line.startswith('$GPGGA') or line.startswith('$GNGGA'):
                return self.parse_gga(line)

    def close(self):
        self.ser.close()

# 使用示例
gps = GPSReader('/dev/serial0', 9600)
try:
    while True:
        data = gps.read()
        if data and data['fix'] > 0:
            print(f"Time: {data['time']} | "
                  f"Lat: {data['latitude']:.6f} | "
                  f"Lon: {data['longitude']:.6f} | "
                  f"Sats: {data['satellites']}")
        time.sleep(1)
except KeyboardInterrupt:
    gps.close()
```

### 7.2 超声波测距（HC-SR04）

```python
import RPi.GPIO as GPIO
import time

GPIO.setmode(GPIO.BCM)

TRIG = 23
ECHO = 24

GPIO.setup(TRIG, GPIO.OUT)
GPIO.setup(ECHO, GPIO.IN)

def get_distance():
    """返回距离（厘米）"""
    # 发送触发脉冲
    GPIO.output(TRIG, True)
    time.sleep(0.00001)  # 10微秒
    GPIO.output(TRIG, False)

    # 等待回波开始
    start_time = time.time()
    while GPIO.input(ECHO) == 0:
        start_time = time.time()
        if time.time() - start_time > 0.1:
            return -1  # 超时

    # 等待回波结束
    while GPIO.input(ECHO) == 1:
        end_time = time.time()
        if time.time() - start_time > 0.1:
            return -1

    # 计算距离
    duration = end_time - start_time
    distance = duration * 17150  # 声速 34300cm/s / 2
    return round(distance, 1)

try:
    while True:
        dist = get_distance()
        if dist > 0:
            bar = '█' * int(dist / 2)
            print(f"Distance: {dist:6.1f} cm  {bar}")
        time.sleep(0.5)
except KeyboardInterrupt:
    GPIO.cleanup()
```

## 8. 最佳实践

### 8.1 GPIO 安全使用

- **使用前清理**：程序结束时调用 `GPIO.cleanup()` 或 `gpiozero` 设备的 `close()`
- **限流电阻**：LED 串联 220Ω-330Ω 电阻
- **不要直接驱动大电流**：GPIO 最大 16mA，使用晶体管或继电器驱动大负载
- **电平转换**：5V 设备连接 3.3V GPIO 时需要电平转换

### 8.2 推荐使用 gpiozero
