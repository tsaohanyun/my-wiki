# 模型部署指南

> 将训练好的模型从实验环境推向生产环境，需要解决**推理性能、服务稳定性、资源效率**三大核心问题。本指南覆盖从模型序列化到高性能推理服务的全链路实践。

### 2.2 自定义Handler

```python
# src/handler.py
import torch
import json
from ts.torch_handler.base_handler import BaseHandler
from transformers import AutoTokenizer, AutoModelForSequenceClassification

class BertClassifierHandler(BaseHandler):
    """自定义推理Handler"""

    def initialize(self, context):
        """模型初始化"""
        self.manifest = context.manifest
        properties = context.system_properties
        model_dir = properties.get("model_dir")
        self.device = torch.device(
            "cuda:" + str(properties.get("gpu_id"))
            if torch.cuda.is_available() and properties.get("gpu_id") is not None
            else "cpu"
        )

        # 加载tokenizer和模型
        self.tokenizer = AutoTokenizer.from_pretrained(model_dir)
        self.model = AutoModelForSequenceClassification.from_pretrained(
            model_dir
        ).to(self.device)
        self.model.eval()

        # 预热
        self._warmup()

    def _warmup(self):
        """模型预热，避免首次请求延迟过高"""
        dummy = self.tokenizer("warmup", return_tensors="pt").to(self.device)
        with torch.no_grad():
            _ = self.model(**dummy)

    def preprocess(self, data):
        """预处理输入数据"""
        texts = []
        for row in data:
            text = row.get("data") or row.get("body")
            if isinstance(text, (bytes, bytearray)):
                text = text.decode("utf-8")
            texts.append(text)

        encoded = self.tokenizer(
            texts,
            padding=True,
            truncation=True,
            max_length=512,
            return_tensors="pt",
        ).to(self.device)
        return encoded

    def inference(self, inputs):
        """模型推理"""
        with torch.no_grad():
            outputs = self.model(**inputs)
            probabilities = torch.softmax(outputs.logits, dim=-1)
            predictions = torch.argmax(probabilities, dim=-1)
        return {"predictions": predictions, "probabilities": probabilities}

    def postprocess(self, inference_output):
        """后处理输出"""
        preds = inference_output["predictions"].cpu().tolist()
        probs = inference_output["probabilities"].cpu().tolist()

        results = []
        for pred, prob in zip(preds, probs):
            results.append({
                "label": self.model.config.id2label[pred],
                "confidence": round(prob[pred], 4),
            })
        return results
```

### 3.3 Python客户端调用

```python
import tritonclient.http as httpclient
import numpy as np
from transformers import AutoTokenizer

# 创建客户端
client = httpclient.InferenceServerClient(url="localhost:8000")

# 检查模型是否就绪
assert client.is_model_ready("bert_classifier"), "Model not ready"

# Tokenize输入
tokenizer = AutoTokenizer.from_pretrained("bert-base-chinese")
text = "这是一个测试句子"
encoded = tokenizer(text, padding="max_length", max_length=128, return_tensors="np")

# 构造输入
inputs = [
    httpclient.InferInput("input_ids", encoded["input_ids"].shape, "INT64"),
    httpclient.InferInput("attention_mask", encoded["attention_mask"].shape, "INT64"),
]
inputs[0].set_data_from_numpy(encoded["input_ids"])
inputs[1].set_data_from_numpy(encoded["attention_mask"])

# 构造输出
outputs = [
    httpclient.InferRequestedOutput("logits", binary_data=True),
]

# 推理
result = client.infer("bert_classifier", inputs=inputs, outputs=outputs)
logits = result.as_numpy("logits")
print(f"Logits: {logits}")
print(f"Prediction: {np.argmax(logits, axis=-1)}")
```

### 3.4 Triton 性能优化要点

| 配置项 | 说明 | 推荐值 |
|--------|------|--------|
| `max_batch_size` | 最大批大小 | 16~64（视GPU显存） |
| `preferred_batch_size` | 动态批处理优先大小 | [4, 8, 16, 32] |
| `max_queue_delay_microseconds` | 最大等待时间 | 50000~100000 |
| `instance_group.count` | 每GPU实例数 | 1~2 |
| `dynamic_batching.preserve_ordering` | 保序 | true（延迟略增，但保证响应顺序） |

## 4 vLLM — 大语言模型推理引擎

vLLM 是面向LLM的高吞吐推理引擎，核心创新是 **PagedAttention** 和 **Continuous Batching**。

### 4.1 离线批量推理

```python
from vllm import LLM, SamplingParams

# 加载模型
llm = LLM(
    model="meta-llama/Llama-2-7b-chat-hf",
    tensor_parallel_size=1,         # 张量并行GPU数
    gpu_memory_utilization=0.90,    # GPU显存利用率
    max_model_len=4096,             # 最大上下文长度
    trust_remote_code=True,
)

# 采样参数
sampling_params = SamplingParams(
    temperature=0.8,
    top_p=0.95,
    max_tokens=512,
    n=1,                            # 每个prompt生成几个回复
)

# 批量推理
prompts = [
    "请用三句话介绍量子计算的基本原理。",
    "写一首关于秋天的五言绝句。",
    "解释什么是梯度消失问题。",
]

outputs = llm.generate(prompts, sampling_params)
for output in outputs:
    prompt = output.prompt
    generated_text = output.outputs[0].text
```

## 5 模型量化

### 5.1 量化方法概览

| 方法 | 类型 | 精度损失 | 推理加速 | 适用场景 |
|------|------|---------|---------|---------|
| FP32→FP16 | 训练后量化 | 极小 | ~2x | 通用 |
| FP16→INT8 | 训练后量化 | 小 | ~3x | CPU/边缘 |
| GPTQ | 训练后量化 | 小 | ~3x | GPU LLM |
| AWQ | 训练后量化 | 极小 | ~3x | GPU LLM |
| BitsAndBytes | 训练后量化 | 中 | ~2x | 消费级GPU |
| QLoRA | 量化+微调 | 极小 | N/A | 低资源微调 |

### 5.2 GPTQ 量化

```python
# pip install auto-gptq optimum
from transformers import AutoModelForCausalLM, AutoTokenizer
from optimum.gptq import GPTQQuantizer

model_id = "meta-llama/Llama-2-7b-chat-hf"
tokenizer = AutoTokenizer.from_pretrained(model_id)

# 加载校准数据集
from datasets import load_dataset
calib_dataset = load_dataset("pile", split="train", streaming=True)
calib_data = [tokenizer.decode(tokenizer(list(calib_dataset.take(128))["text"],
                                         return_tensors="np")["input_ids"][0][:2048])]

# GPTQ量化
quantizer = GPTQQuantizer(
    bits=4,
    dataset="c4",
    group_size=128,
    desc_act=False,         # 无激活重排序，推理更快
    sym=True,               # 对称量化
)

model = AutoModelForCausalLM.from_pretrained(model_id, torch_dtype="auto")
quantized_model = quantizer.quantize_model(model, tokenizer)

# 保存量化模型
quantized_model.save_quantized("./models/llama2-7b-gptq-4bit")
```

### 6.4 推理性能基准测试脚本

```python
import time
import asyncio
import aiohttp
import statistics
from dataclasses import dataclass

@dataclass
class BenchmarkResult:
    total_requests: int
    success_count: int
    avg_latency_ms: float
    p50_latency_ms: float
    p95_latency_ms: float
    p99_latency_ms: float
    throughput_rps: float

async def benchmark_endpoint(
    url: str,
    payload: dict,
    num_requests: int = 100,
    concurrency: int = 10,
) -> BenchmarkResult:
    """异步压测推理端点"""
    semaphore = asyncio.Semaphore(concurrency)
    latencies = []
    success = 0

    async def single_request(session):
        nonlocal success
        async with semaphore:
            start = time.perf_counter()
            try:
                async with session.post(url, json=payload) as resp:
                    if resp.status == 200:
                        await resp.json()
                        success += 1
            except Exception as e:
                print(f"Request failed: {e}")
            finally:
                latencies.append((time.perf_counter() - start) * 1000)

    async with aiohttp.ClientSession() as session:
        tasks = [single_request(session) for _ in range(num_requests)]
        total_start = time.perf_counter()
        await asyncio.gather(*tasks)
        total_time = time.perf_counter() - total_start

    latencies.sort()
    return BenchmarkResult(
        total_requests=num_requests,
        success_count=success,
        avg_latency_ms=statistics.mean(latencies),
        p50_latency_ms=latencies[len(latencies) // 2],
        p95_latency_ms=latencies[int(len(latencies) * 0.95)],
        p99_latency_ms=latencies[int(len(latencies) * 0.99)],
        throughput_rps=num_requests / total_time,
    )

# 运行基准测试
import asyncio
result = asyncio.run(benchmark_endpoint(
    url="http://localhost:8000/v1/chat/completions",
    payload={
        "model": "llama2-7b",
        "messages": [{"role": "user", "content": "Hello"}],
        "max_tokens": 32,
    },
    num_requests=200,
    concurrency=20,
))
print(f"Result: {result}")
```
