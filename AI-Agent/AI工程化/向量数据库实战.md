# 向量数据库实战

> 向量数据库是RAG（检索增强生成）和语义搜索的核心基础设施。它将文本/图像/音频编码为高维向量，通过**近似最近邻（ANN）**算法实现毫秒级语义检索。

## 1 向量数据库核心概念

### 1.1 工作原理

```
原始数据                向量嵌入                   向量索引               查询
┌───────┐  Embedding   ┌───────────┐   Index    ┌───────────┐   ANN   ┌──────┐
│ 文本   │─────────────→│ [0.1,0.5, │───────────→│  HNSW /   │────────→│ Top-K │
│ 图像   │   Model      │  -0.3,...]│            │  IVF /    │        │ 结果   │
│ 音频   │             │ [128~1536 │            │  Flat     │         │       │
└───────┘             │   维]     │            └───────────┘         └──────┘
                      └───────────┘
```

### 1.2 索引算法对比

| 算法 | 全称 | 查询精度 | 查询速度 | 内存占用 | 构建速度 | 适用场景 |
|------|------|---------|---------|---------|---------|---------|
| **FLAT** | 暴力搜索 | 100% | 慢 | 低 | 快 | 小数据集（<10万） |
| **IVF** | 倒排文件 | 高 | 快 | 中 | 中 | 中等数据集 |
| **IVF_PQ** | 倒排+乘积量化 | 中 | 很快 | 很低 | 慢 | 超大规模、内存受限 |
| **HNSW** | 分层可导航小世界 | 很高 | 很快 | 高 | 中 | **推荐首选** |
| **ANNOY** | 基于树 | 高 | 快 | 中 | 慢 | 只读场景 |
| **SCANN** | Google扫描 | 很高 | 很快 | 中 | 中 | 高精度需求 |

### 1.3 相似度度量

```python
import numpy as np

# 余弦相似度 — 最常用于文本嵌入
def cosine_similarity(a, b):
    return np.dot(a, b) / (np.linalg.norm(a) * np.linalg.norm(b))

# 欧氏距离 (L2) — 常用于图像嵌入
def euclidean_distance(a, b):
    return np.sqrt(np.sum((a - b) ** 2))

# 内积 (Inner Product / Dot Product) — 用于已归一化的向量
def inner_product(a, b):
    return np.dot(a, b)

# 经验法则:
# - OpenAI embeddings → cosine
# - BERT/Sentence-BERT → cosine
# - CLIP图像 → cosine
# - 非归一化特征 → L2
```

```python
    },
)

# 创建Collection
client.create_collection(
    collection_name=collection_name,
    schema=schema,
    index_params=index_params,
```

```python
from pinecone import Pinecone

index = pc.Index("documents")

# 批量upsert（Pinecone建议每批100条）
def batch_upsert(data, batch_size=100, namespace="default"):
    for i in range(0, len(data), batch_size):
        batch = data[i:i + batch_size]
        index.upsert(vectors=batch, namespace=namespace)
        print(f"Upserted batch {i // batch_size + 1}")

# 使用生成器处理大规模数据
def vector_generator(documents, embedding_fn):
    for doc in documents:
        yield {
            "id": doc["id"],
            "values": embedding_fn(doc["text"]),
            "metadata": {
                "source": doc.get("source", ""),
                "category": doc.get("category", ""),
                "chunk_index": doc.get("chunk_index", 0),
            },
        }

# 删除特定namespace下的数据
index.delete(delete_all=True, namespace="staging")

# 获取索引统计
stats = index.describe_index_stats()
print(f"Total vectors: {stats['total_vector_count']}")
```

```python
from qdrant_client.models import PointStruct
import uuid

# 批量上传（使用generator节省内存）
def upload_in_batches(client, collection_name, documents, batch_size=512):
    batch = []
    for doc in documents:
        point = PointStruct(
            id=str(uuid.uuid4()),
            vector=doc["embedding"],
            payload=doc["metadata"],
        )
        batch.append(point)

        if len(batch) >= batch_size:
            client.upsert(
                collection_name=collection_name,
                points=batch,
                wait=True,
            )
            batch = []
            print(f"Uploaded batch of {batch_size}")

    # 上传剩余
    if batch:
        client.upsert(
            collection_name=collection_name,
            points=batch,
            wait=True,
        )

# 滚动查询所有数据（类似分页）
records, offset = client.scroll(
    collection_name="documents",
    limit=100,
    offset=None,
    with_payload=True,
    with_vectors=False,
)
print(f"Retrieved {len(records)} records, offset: {offset}")

# 继续获取下一页
records, offset = client.scroll(
    collection_name="documents",
    limit=100,
    offset=offset,
```

## 6 pgvector (PostgreSQL)

### 6.1 安装与配置

```sql
-- 在PostgreSQL中安装pgvector扩展
CREATE EXTENSION IF NOT EXISTS vector;

-- 创建表
CREATE TABLE documents (
    id BIGSERIAL PRIMARY KEY,
    content TEXT NOT NULL,
    source VARCHAR(256),
    category VARCHAR(64),
    embedding VECTOR(1536)  -- OpenAI text-embedding-3-small维度
);

-- 创建HNSW索引（推荐）
CREATE INDEX idx_documents_embedding_hnsw
ON documents USING hnsw (embedding vector_cosine_ops)
WITH (
    m = 16,              -- 每个节点最大连接数
    ef_construction = 64 -- 构建时搜索宽度
);

-- 或者使用IVFFlat索引（适合大批量导入后再建索引）
```

### 6.3 pgvector 运算符参考

```sql
-- 距离运算符
-- <=> : 余弦距离 (Cosine Distance) — 最常用
-- <-> : L2距离 (Euclidean Distance)
-- <#> : 负内积 (Negative Inner Product)

-- 查找最相似的5条记录（余弦相似度）
SELECT content, 1 - (embedding <=> '[0.1, 0.2, ...]'::vector) AS similarity
FROM documents
ORDER BY embedding <=> '[0.1, 0.2, ...]'::vector
LIMIT 5;

-- 范围搜索（相似度 > 0.8）
SELECT content
FROM documents
WHERE embedding <=> '[0.1, 0.2, ...]'::vector < 0.2
ORDER BY embedding <=> '[0.1, 0.2, ...]'::vector;

-- 聚合：找每类文档的聚类中心
SELECT category, avg(embedding) AS centroid
FROM documents
GROUP BY category;
```

## 7 RAG 系统集成示例

### 7.1 完整RAG Pipeline

```python
"""
完整的RAG（检索增强生成）Pipeline：
1. 文档分块 → 2. 向量化 → 3. 存入向量数据库 → 4. 检索 → 5. 生成
"""
from openai import OpenAI
from qdrant_client import QdrantClient
from qdrant_client.models import Distance, VectorParams, PointStruct
import uuid
from dataclasses import dataclass

@dataclass
class Document:
    content: str
    source: str
    metadata: dict

class RAGPipeline:
    def __init__(self, qdrant_host="localhost", qdrant_port=6333):
        self.llm = OpenAI()
        self.qdrant = QdrantClient(host=qdrant_host, port=qdrant_port)
        self.collection_name = "knowledge_base"
        self.embedding_model = "text-embedding-3-small"
        self.llm_model = "gpt-4o"
        self.chunk_size = 500
        self.chunk_overlap = 50

    def _get_embedding(self, text: str) -> list[float]:
        resp = self.llm.embeddings.create(
            input=text, model=self.embedding_model
        )
        return resp.data[0].embedding

    def chunk_text(self, text: str) -> list[str]:
        """固定窗口分块（带重叠）"""
        chunks = []
        start = 0
        while start < len(text):
            end = start + self.chunk_size
            chunks.append(text[start:end])
            start += self.chunk_size - self.chunk_overlap
        return chunks

    def ingest(self, text: str, source: str, metadata: dict = None):
        """文档入库：分块 → 向量化 → 存储"""
        chunks = self.chunk_text(text)

        points = []
        for i, chunk in enumerate(chunks):
            embedding = self._get_embedding(chunk)
            points.append(PointStruct(
                id=str(uuid.uuid4()),
                vector=embedding,
                payload={
                    "content": chunk,
                    "source": source,
                    "chunk_index": i,
                    **(metadata or {}),
                },
            ))

        self.qdrant.upsert(
            collection_name=self.collection_name,
            points=points,
            wait=True,
        )
        print(f"Ingested {len(points)} chunks from {source}")

    def retrieve(self, query: str, top_k: int = 5) -> list[dict]:
        """向量检索"""
        query_vector = self._get_embedding(query)
        results = self.qdrant.search(
            collection_name=self.collection_name,
            query_vector=query_vector,
            limit=top_k,
            with_payload=True,
        )
        return [
            {
                "content": hit.payload["content"],
                "source": hit.payload["source"],
                "score": hit.score,
            }
            for hit in results
        ]

    def generate(self, query: str, context_docs: list[dict]) -> str:
        """LLM生成"""
        context_text = "\n\n".join([
            f"[来源: {d['source']}] {d['content']}"
            for d in context_docs
        ])

        prompt = f"""你是一个专业的AI助手。请根据以下参考资料回答用户问题。
如果参考资料中没有相关信息，请说明你无法从提供的资料中找到答案。

参考资料：
{context_text}

用户问题：{query}

回答："""

        response = self.llm.chat.completions.create(
            model=self.llm_model,
            messages=[
                {"role": "system", "content": "你是一个专业的知识库问答助手。"},
                {"role": "user", "content": prompt},
            ],
            temperature=0.3,
            max_tokens=1024,
        )
        return response.choices[0].message.content

    def query(self, question: str) -> str:
        """端到端RAG查询"""
        # 1. 检索相关文档
        docs = self.retrieve(question, top_k=5)
        print(f"Retrieved {len(docs)} documents")

        # 2. 生成回答
        answer = self.generate(question, docs)

        # 3. 返回结果（附带引用来源）
        sources = [d["source"] for d in docs]
        return {"answer": answer, "sources": sources}


# 使用示例
rag = RAGPipeline()

# 初始化Collection
rag.qdrant.recreate_collection(
    collection_name="knowledge_base",
    vectors_config=VectorParams(size=1536, distance=Distance.COSINE),
)

# 入库文档
rag.ingest(
    text=open("docs/milvus_guide.md").read(),
    source="milvus_guide.md",
    metadata={"category": "database"},
)

# 查询
result = rag.query("Milvus支持哪些索引算法？")
print(f"Answer: {result['answer']}")
print(f"Sources: {result['sources']}")
```

## 8 最佳实践

### 8.1 索引选择指南

```
数据规模?
├── < 10万 → FLAT（暴力搜索，精度100%）
├── 10万~100万 → HNSW（内存足够时首选）
├── 100万~1000万 → HNSW / IVF_PQ
└── > 1000万 → DiskANN / IVF_PQ（内存优化）
```

### 8.2 嵌入维度优化

| 维度 | 模型 | 检索质量 | 存储/延迟 |
|------|------|---------|----------|
| 768 | BERT-base, MiniLM | 中 | 低 |
| 1024 | BGE-large, E5-large | 高 | 中 |
| 1536 | OpenAI ada-002 / 3-small | 高 | 高 |
| 3072 | OpenAI 3-large | 极高 | 很高 |

> **建议**：先从1536维开始，如果延迟或存储是瓶颈，考虑降维或使用Matryoshka嵌入。

### 8.3 性能调优清单

| 项目 | 建议 |
|------|------|
| 批量插入 | 每批100-500条，开启并行 |
| 向量归一化 | 使用余弦相似度时务必归一化 |
| 分块策略 | 语义分块 > 固定窗口分块 |
| 过滤优化 | 先标量过滤再向量搜索（pre-filter） |
| 缓存 | 缓存高频query的embedding结果 |
| 监控 | P99查询延迟、召回率、索引构建时间 |
