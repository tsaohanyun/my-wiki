## 1. 帧同步 vs 状态同步

这是网络游戏最根本的架构选择，决定了服务器与客户端的职责分工。

### 1.1 核心对比

| 维度 | 帧同步 (Lockstep) | 状态同步 (State Sync) |
|------|-------------------|----------------------|
| **传输内容** | 玩家输入操作 | 游戏对象状态（位置/血量等） |
| **服务器角色** | 转发输入，验证 | 模拟世界，权威状态 |
| **带宽消耗** | 极低（仅输入指令） | 较高（状态快照） |
| **客户端计算** | 本地模拟全部逻辑 | 渲染 + 预测/插值 |
| **断线重连** | 困难（需重放历史帧） | 容易（拉取当前状态） |
| **防作弊** | 困难（需确定性校验） | 容易（服务器权威） |
| **确定性要求** | 严格（浮点数需定点化） | 无要求 |
| **适用游戏** | RTS、格斗、MOBA | FPS、MMO、开放世界 |

### 1.2 帧同步架构详解

```
┌─────────┐     输入      ┌──────────┐     广播      ┌─────────┐
│ Player A │─────────────→│  Server  │─────────────→│ Player A │
│ (Client) │              │ (转发器)  │              │ (Client) │
└─────────┘              │          │              └─────────┘
                          │  帧锁定   │
┌─────────┐     输入      │          │     广播      ┌─────────┐
│ Player B │─────────────→│          │─────────────→│ Player B │
│ (Client) │              └──────────┘              │ (Client) │
└─────────┘                                         └─────────┘

关键：所有客户端在相同的输入下，运行相同的逻辑代码，得到相同的游戏状态。
```

#### 帧同步服务器实现（Go）

```go
package framesync

import (
	"encoding/json"
	"log"
	"sync"
	"time"
)

// PlayerInput 玩家单帧输入
type PlayerInput struct {
	PlayerID string          `json:"player_id"`
	Frame    int64           `json:"frame"`
	Input    json.RawMessage `json:"input"` // 具体操作指令
}

// FrameData 单帧所有玩家输入
type FrameData struct {
	Frame  int64                  `json:"frame"`
	Inputs map[string]json.RawMessage `json:"inputs"` // playerID -> input
}

// GameRoom 帧同步房间
type GameRoom struct {
	mu          sync.RWMutex
	RoomID      string
	MaxPlayers  int
	Players     map[string]bool       // playerID -> online
	InputBuffer map[int64]map[string]json.RawMessage // frame -> playerID -> input
	CurrentFrame int64
	TickRate     int // 每秒帧数（如 30 = 30fps logic）
	broadcastFn  func([]byte)
	stopCh       chan struct{}
}

func NewGameRoom(roomID string, maxPlayers, tickRate int) *GameRoom {
	return &GameRoom{
		RoomID:       roomID,
		MaxPlayers:   maxPlayers,
		Players:      make(map[string]bool),
		InputBuffer:  make(map[int64]map[string]json.RawMessage),
		CurrentFrame: 0,
		TickRate:     tickRate,
		stopCh:       make(chan struct{}),
	}
}

// AddPlayer 玩家加入房间
func (r *GameRoom) AddPlayer(playerID string) bool {
	r.mu.Lock()
	defer r.mu.Unlock()

	if len(r.Players) >= r.MaxPlayers {
		return false
	}
	r.Players[playerID] = true
	return true
}

// SubmitInput 玩家提交输入
func (r *GameRoom) SubmitInput(playerID string, frame int64, input json.RawMessage) {
	r.mu.Lock()
	defer r.mu.Unlock()

	if _, ok := r.InputBuffer[frame]; !ok {
		r.InputBuffer[frame] = make(map[string]json.RawMessage)
	}
	r.InputBuffer[frame][playerID] = input
}

// Start 主循环：固定频率广播帧数据
func (r *GameRoom) Start() {
	interval := time.Second / time.Duration(r.TickRate)
	ticker := time.NewTicker(interval)
	defer ticker.Stop()

	for {
		select {
		case <-ticker.C:
			r.tick()
		case <-r.stopCh:
			return
		}
	}
}

// tick 每个逻辑帧处理一次
func (r *GameRoom) tick() {
	r.mu.Lock()

	frame := r.CurrentFrame
	inputs := r.InputBuffer[frame]

	// 为离线玩家填充空输入（保证帧一致）
	for playerID := range r.Players {
		if inputs == nil {
			inputs = make(map[string]json.RawMessage)
			r.InputBuffer[frame] = inputs
		}
		if _, ok := inputs[playerID]; !ok {
			inputs[playerID] = json.RawMessage(`{}`)
		}
	}

	frameData := FrameData{
		Frame:  frame,
		Inputs: inputs,
	}

	// 清理旧帧
	delete(r.InputBuffer, frame)
	r.CurrentFrame++

	r.mu.Unlock()

	// 序列化并广播（不加锁）
	if data, err := json.Marshal(frameData); err == nil && r.broadcastFn != nil {
		r.broadcastFn(data)
	}
}

func (r *GameRoom) Stop() {
	close(r.stopCh)
}

func (r *GameRoom) SetBroadcast(fn func([]byte)) {
	r.broadcastFn = fn
}
```

### 1.3 状态同步架构详解

```
┌─────────┐    操作请求     ┌──────────────┐    状态快照    ┌─────────┐
│ Player A │───────────────→│              │──────────────→│ Player A │
│ (Client) │                │   Game       │  (位置/血量等) │ (Client) │
│  预测    │←───修正────────│   Server     │               │  插值    │
└─────────┘   (权威纠正)    │ (权威模拟)    │               └─────────┘
                            │              │
┌─────────┐    操作请求     │  世界状态    │    状态快照    ┌─────────┐
│ Player B │───────────────→│  物理引擎    │──────────────→│ Player B │
│ (Client) │                │  碰撞检测    │               │ (Client) │
└─────────┘                 └──────────────┘               └─────────┘

关键：服务器是游戏世界的唯一权威，客户端做预测展示并接受服务器修正。
```

#### 状态同步服务器实现（Go）

```go
package statesync

import (
	"encoding/json"
	"math"
	"sync"
	"time"
)

// EntityState 游戏对象状态
type EntityState struct {
	ID       string  `json:"id"`
	Type     string  `json:"type"` // "player", "npc", "projectile"
	X        float64 `json:"x"`
	Y        float64 `json:"y"`
	Z        float64 `json:"z"`
	RotY     float64 `json:"rot_y"`
	Health   int     `json:"health"`
	VelocityX float64 `json:"vx"`
	VelocityZ float64 `json:"vz"`
	LastUpdate int64 `json:"last_update"`
}

// PlayerAction 玩家操作
type PlayerAction struct {
	PlayerID string  `json:"player_id"`
	Type     string  `json:"type"` // "move", "attack", "jump"
	X        float64 `json:"x"`
	Z        float64 `json:"z"`
	TargetID string  `json:"target_id,omitempty"`
}

// GameServer 状态同步服务器
type GameServer struct {
	mu        sync.RWMutex
	Entities  map[string]*EntityState
	Players   map[string]*ClientConnection
	TickRate  int
	stopCh    chan struct{}
}

type ClientConnection struct {
	PlayerID  string
	SendCh    chan []byte
	ViewDistance float64 // 可见范围（AOI）
}

func NewGameServer(tickRate int) *GameServer {
	return &GameServer{
		Entities: make(map[string]*EntityState),
		Players:  make(map[string]*ClientConnection),
		TickRate: tickRate,
		stopCh:   make(chan struct{}),
	}
}

// ProcessAction 处理玩家操作（服务器权威）
func (s *GameServer) ProcessAction(action PlayerAction) {
	s.mu.Lock()
	defer s.mu.Unlock()

	entity, ok := s.Entities[action.PlayerID]
	if !ok {
		return
	}

	switch action.Type {
	case "move":
		// 服务器验证移动合法性（防穿墙/加速）
		newX := entity.X + action.X * 0.1
		newZ := entity.Z + action.Z * 0.1

		// 简单边界检查
		if math.Abs(newX) < 1000 && math.Abs(newZ) < 1000 {
			entity.X = newX
			entity.Z = newZ
		}

	case "attack":
		// 处理伤害逻辑
		if target, ok := s.Entities[action.TargetID]; ok {
			distance := math.Sqrt(
				math.Pow(entity.X-target.X, 2) +
				math.Pow(entity.Z-target.Z, 2),
			)
			if distance < 5.0 { // 攻击范围 5 米
				target.Health -= 10
				if target.Health <= 0 {
					delete(s.Entities, target.ID)
				}
			}
		}
	}

	entity.LastUpdate = time.Now().UnixMilli()
}

// Start 主循环：定时发送状态快照
func (s *GameServer) Start() {
	interval := time.Second / time.Duration(s.TickRate)
	ticker := time.NewTicker(interval)
	defer ticker.Stop()

	for {
		select {
		case <-ticker.C:
			s.broadcastSnapshot()
		case <-s.stopCh:
			return
		}
	}
}

// broadcastSnapshot 向每个玩家发送其视野内的实体状态
func (s *GameServer) broadcastSnapshot() {
	s.mu.RLock()
	defer s.mu.RUnlock()

	for playerID, conn := range s.Players {
		playerEntity := s.Entities[playerID]
		if playerEntity == nil {
			continue
		}

		// AOI（Area of Interest）：只发送附近的实体
		visibleEntities := make([]*EntityState, 0)
		for _, entity := range s.Entities {
			distance := math.Sqrt(
				math.Pow(playerEntity.X-entity.X, 2) +
				math.Pow(playerEntity.Z-entity.Z, 2),
			)
			if distance <= conn.ViewDistance {
				visibleEntities = append(visibleEntities, entity)
			}
		}

		snapshot := map[string]interface{}{
			"type":     "snapshot",
			"entities": visibleEntities,
			"frame":    time.Now().UnixMilli(),
		}

		data, _ := json.Marshal(snapshot)
		select {
		case conn.SendCh <- data:
		default:
			// 发送缓冲已满，跳过（防止慢客户端阻塞服务器）
		}
	}
}

func (s *GameServer) Stop() {
	close(s.stopCh)
}
```

### 1.4 客户端预测与插值

```csharp
// Unity 客户端状态同步示例
using UnityEngine;
using System.Collections.Generic;

public class NetworkEntityInterpolation : MonoBehaviour
{
    // 收到的服务器状态历史（用于插值）
    private struct Snapshot
    {
        public float timestamp;
        public Vector3 position;
        public Vector3 velocity;
    }

    private readonly List<Snapshot> snapshots = new();
    private float interpolationDelay = 0.1f; // 延迟 100ms 渲染，用于平滑插值

    // 收到服务器快照
    public void OnSnapshotReceived(float timestamp, Vector3 position, Vector3 velocity)
    {
        snapshots.Add(new Snapshot { timestamp = timestamp, position = position, velocity = velocity });

        // 保留最近 1 秒的数据
        float cutoff = Time.time - 1f;
        snapshots.RemoveAll(s => s.timestamp < cutoff);
    }

    private void Update()
    {
        if (snapshots.Count < 2) return;

        float renderTime = Time.time - interpolationDelay;

        // 找到插值的两个关键帧
        for (int i = 0; i < snapshots.Count - 1; i++)
        {
            if (snapshots[i].timestamp <= renderTime && renderTime <= snapshots[i + 1].timestamp)
            {
                float t = (renderTime - snapshots[i].timestamp) /
                          (snapshots[i + 1].timestamp - snapshots[i].timestamp);
                transform.position = Vector3.Lerp(snapshots[i].position, snapshots[i + 1].position, t);
                break;
            }
        }
    }
}
```

### 2.2 房间管理实现（Go + Redis）

```go
package room

import (
	"context"
	"encoding/json"
	"fmt"
	"time"

	"github.com/redis/go-redis/v9"
)

// Room 房间数据结构
type Room struct {
	RoomID      string          `json:"room_id"`
	GameMode    string          `json:"game_mode"`
	MapID       string          `json:"map_id"`
	HostID      string          `json:"host_id"`
	MaxPlayers  int             `json:"max_players"`
	Players     []PlayerInfo    `json:"players"`
	Status      string          `json:"status"` // "waiting", "playing", "finished"
	Password    string          `json:"-"`      // 不序列化
	CreatedAt   time.Time       `json:"created_at"`
	Settings    json.RawMessage `json:"settings"`
}

type PlayerInfo struct {
	PlayerID  string `json:"player_id"`
	Name      string `json:"name"`
	Team      int    `json:"team"`
	Ready     bool   `json:"ready"`
	JoinedAt  time.Time `json:"joined_at"`
}

// RoomManager 房间管理器
type RoomManager struct {
	rdb         *redis.Client
	ctx         context.Context
	maxRooms    int
}

func NewRoomManager(rdb *redis.Client) *RoomManager {
	return &RoomManager{
		rdb:      rdb,
		ctx:      context.Background(),
		maxRooms: 10000,
	}
}

// CreateRoom 创建房间
func (rm *RoomManager) CreateRoom(hostID, gameMode, mapID string, maxPlayers int, password string) (*Room, error) {
	// 生成唯一房间 ID
	roomID, err := rm.rdb.Incr(rm.ctx, "room:id_counter").Result()
	if err != nil {
		return nil, fmt.Errorf("failed to generate room ID: %w", err)
	}

	room := &Room{
		RoomID:     fmt.Sprintf("room_%d", roomID),
		GameMode:   gameMode,
		MapID:      mapID,
		HostID:     hostID,
		MaxPlayers: maxPlayers,
		Players: []PlayerInfo{
			{
				PlayerID: hostID,
				Team:     0,
				Ready:    true,
				JoinedAt: time.Now(),
			},
		},
		Status:    "waiting",
		Password:  password,
		CreatedAt: time.Now(),
	}

	// 序列化并存入 Redis
	data, err := json.Marshal(room)
	if err != nil {
		return nil, err
	}

	pipe := rm.rdb.Pipeline()
	pipe.Set(rm.ctx, fmt.Sprintf("room:%s", room.RoomID), data, 2*time.Hour)
	pipe.SAdd(rm.ctx, "rooms:active", room.RoomID)
	pipe.HSet(rm.ctx, "player:room", hostID, room.RoomID) // 玩家→房间映射
	_, err = pipe.Exec(rm.ctx)
	if err != nil {
		return nil, fmt.Errorf("failed to save room: %w", err)
	}

	return room, nil
}

// JoinRoom 玩家加入房间
func (rm *RoomManager) JoinRoom(playerID, roomID, password string) (*Room, error) {
	// 检查玩家是否已在其他房间
	existingRoom, _ := rm.rdb.HGet(rm.ctx, "player:room", playerID).Result()
	if existingRoom != "" {
		return nil, fmt.Errorf("player already in room %s", existingRoom)
	}

	// 获取房间数据
	data, err := rm.rdb.Get(rm.ctx, fmt.Sprintf("room:%s", roomID)).Result()
	if err != nil {
		return nil, fmt.Errorf("room not found")
	}

	var room Room
	if err := json.Unmarshal([]byte(data), &room); err != nil {
		return nil, err
	}

	// 检查密码
	if room.Password != "" && room.Password != password {
		return nil, fmt.Errorf("incorrect password")
	}

	// 检查人数
	if len(room.Players) >= room.MaxPlayers {
		return nil, fmt.Errorf("room is full")
	}

	// 检查状态
	if room.Status != "waiting" {
		return nil, fmt.Errorf("room is not joinable (status: %s)", room.Status)
	}

	// 添加玩家
	room.Players = append(room.Players, PlayerInfo{
		PlayerID: playerID,
		Team:     len(room.Players) % 2, // 简单分队
		Ready:    false,
		JoinedAt: time.Now(),
	})

	// 保存更新
	updatedData, _ := json.Marshal(room)
	pipe := rm.rdb.Pipeline()
	pipe.Set(rm.ctx, fmt.Sprintf("room:%s", roomID), updatedData, 2*time.Hour)
	pipe.HSet(rm.ctx, "player:room", playerID, roomID)
	_, err = pipe.Exec(rm.ctx)

	return &room, err
}

// LeaveRoom 玩家离开房间
func (rm *RoomManager) LeaveRoom(playerID string) error {
	roomID, err := rm.rdb.HGet(rm.ctx, "player:room", playerID).Result()
	if err != nil {
		return fmt.Errorf("player not in any room")
	}

	data, err := rm.rdb.Get(rm.ctx, fmt.Sprintf("room:%s", roomID)).Result()
	if err != nil {
		return fmt.Errorf("room not found")
	}

	var room Room
	json.Unmarshal([]byte(data), &room)

	// 移除玩家
	for i, p := range room.Players {
		if p.PlayerID == playerID {
			room.Players = append(room.Players[:i], room.Players[i+1:]...)
			break
		}
	}

	pipe := rm.rdb.Pipeline()
	pipe.HDel(rm.ctx, "player:room", playerID)

	if len(room.Players) == 0 {
		// 房间空了，销毁
		pipe.Del(rm.ctx, fmt.Sprintf("room:%s", roomID))
		pipe.SRem(rm.ctx, "rooms:active", roomID)
	} else {
		// 如果房主离开，转移房主
		if room.HostID == playerID {
			room.HostID = room.Players[0].PlayerID
		}
		updatedData, _ := json.Marshal(room)
		pipe.Set(rm.ctx, fmt.Sprintf("room:%s", roomID), updatedData, 2*time.Hour)
	}

	_, err = pipe.Exec(rm.ctx)
	return err
}

// ListWaitingRooms 列出等待中的房间
func (rm *RoomManager) ListWaitingRooms(gameMode string, offset, limit int64) ([]Room, error) {
	roomIDs, err := rm.rdb.SMembers(rm.ctx, "rooms:active").Result()
	if err != nil {
		return nil, err
	}

	var rooms []Room
	for _, roomID := range roomIDs {
		data, err := rm.rdb.Get(rm.ctx, fmt.Sprintf("room:%s", roomID)).Result()
		if err != nil {
			continue
		}

		var room Room
		if err := json.Unmarshal([]byte(data), &room); err != nil {
			continue
		}

		if room.Status == "waiting" && (gameMode == "" || room.GameMode == gameMode) {
			if len(room.Players) < room.MaxPlayers {
				rooms = append(rooms, room)
			}
		}
	}

	return rooms, nil
}
```

## 3. 匹配系统

### 3.1 匹配评分算法

```
匹配评分(Match Score) = |playerMMR - averageMMR| + 等待时间权重 × 等待秒数

策略：
- 初始搜索范围窄（±100 MMR）
- 随等待时间扩大范围（每 10 秒扩大 50 MMR）
- 超时阈值后强制匹配（降低标准）
```

### 3.2 匹配系统实现

```go
package matchmaking

import (
	"context"
	"fmt"
	"sort"
	"sync"
	"time"

	"github.com/redis/go-redis/v9"
)

// MatchRequest 匹配请求
type MatchRequest struct {
	PlayerID   string
	MMR        int           // Matchmaking Rating
	GameMode   string
	TeamSize   int           // 每队人数
	JoinedAt   time.Time
}

// Matchmaker 匹配系统
type Matchmaker struct {
	rdb       *redis.Client
	ctx       context.Context
	mu        sync.Mutex
	queues    map[string][]MatchRequest // gameMode -> 队列
	stopCh    chan struct{}

	// 匹配参数
	initialRange   int           // 初始 MMR 范围
	rangeExpansion int           // 每秒扩大范围
	maxWaitTime    time.Duration // 最长等待
	forceRange     int           // 强制匹配范围
	tickInterval   time.Duration
}

func NewMatchmaker(rdb *redis.Client) *Matchmaker {
	return &Matchmaker{
		rdb:            rdb,
		ctx:            context.Background(),
		queues:         make(map[string][]MatchRequest),
		stopCh:         make(chan struct{}),
		initialRange:   100,
		rangeExpansion: 5,
		maxWaitTime:    120 * time.Second,
		forceRange:     500,
		tickInterval:   1 * time.Second,
	}
}

// Enqueue 玩家加入匹配队列
func (m *Matchmaker) Enqueue(req MatchRequest) {
	m.mu.Lock()
	defer m.mu.Unlock()

	req.JoinedAt = time.Now()
	m.queues[req.GameMode] = append(m.queues[req.GameMode], req)
}

// Dequeue 玩家退出匹配
func (m *Matchmaker) Dequeue(playerID, gameMode string) {
	m.mu.Lock()
	defer m.mu.Unlock()

	queue := m.queues[gameMode]
	for i, req := range queue {
		if req.PlayerID == playerID {
			m.queues[gameMode] = append(queue[:i], queue[i+1:]...)
			break
		}
	}
}

// Run 匹配主循环
func (m *Matchmaker) Run() {
	ticker := time.NewTicker(m.tickInterval)
	defer ticker.Stop()

	for {
		select {
		case <-ticker.C:
			m.processQueues()
		case <-m.stopCh:
			return
		}
	}
}

// processQueues 处理所有队列
func (m *Matchmaker) processQueues() {
	m.mu.Lock()
	defer m.mu.Unlock()

	for gameMode, queue := range m.queues {
		if len(queue) == 0 {
			continue
		}

		matches := m.findMatches(queue)

		// 从队列中移除已匹配的玩家
		matchedPlayers := make(map[string]bool)
		for _, match := range matches {
			for _, req := range match {
				matchedPlayers[req.PlayerID] = true
			}
			// 创建房间并通知玩家
			go m.onMatchFound(match)
		}

		var remaining []MatchRequest
		for _, req := range queue {
			if !matchedPlayers[req.PlayerID] {
				remaining = append(remaining, req)
			}
		}
		m.queues[gameMode] = remaining
	}
}

// findMatches 核心匹配算法
func (m *Matchmaker) findMatches(queue []MatchRequest) [][]MatchRequest {
	now := time.Now()
	var matches [][]MatchRequest

	// 按 MMR 排序
	sortedQueue := make([]MatchRequest, len(queue))
	copy(sortedQueue, queue)
	sort.Slice(sortedQueue, func(i, j int) bool {
		return sortedQueue[i].MMR < sortedQueue[j].MMR
	})

	used := make([]bool, len(sortedQueue))

	for i := 0; i < len(sortedQueue); i++ {
		if used[i] {
			continue
		}

		waitTime := now.Sub(sortedQueue[i].JoinedAt)
		// 动态范围：等待越久，范围越大
		dynamicRange := m.initialRange + int(waitTime.Seconds())*m.rangeExpansion
		if dynamicRange > m.forceRange {
			dynamicRange = m.forceRange
		}

		teamSize := sortedQueue[i].TeamSize
		totalNeeded := teamSize * 2 // 两队

		// 贪心匹配：寻找 MMR 相近的玩家
		var team []MatchRequest
		team = append(team, sortedQueue[i])
		used[i] = true

		for j := i + 1; j < len(sortedQueue) && len(team) < totalNeeded; j++ {
			if used[j] {
				continue
			}

			avgMMR := m.averageMMR(team)
			if abs(sortedQueue[j].MMR-avgMMR) <= dynamicRange {
				team = append(team, sortedQueue[j])
				used[j] = true
			}
		}

		if len(team) == totalNeeded {
			matches = append(matches, team)
		} else {
			// 人数不够，回退标记
			for _, req := range team {
				for k := range sortedQueue {
					if sortedQueue[k].PlayerID == req.PlayerID {
						used[k] = false
						break
					}
				}
			}
		}
	}

	return matches
}

func (m *Matchmaker) averageMMR(team []MatchRequest) int {
	sum := 0
	for _, req := range team {
		sum += req.MMR
	}
	return sum / len(team)
}

func (m *Matchmaker) onMatchFound(team []MatchRequest) {
	playerIDs := make([]string, len(team))
	for i, req := range team {
		playerIDs[i] = req.PlayerID
	}
	// 通知房间服务创建房间，通知网关推送消息
	fmt.Printf("Match found! Players: %v\n", playerIDs)
}

func abs(x int) int {
	if x < 0 {
		return -x
	}
	return x
}

func (m *Matchmaker) Stop() {
	close(m.stopCh)
}
```

### 4.1 Redis Sorted Set 实现

Redis 的 `ZSET`（有序集合）是实现排行榜的最佳数据结构。

```go
package leaderboard

import (
	"context"
	"fmt"
	"strconv"
	"time"

	"github.com/redis/go-redis/v9"
)

// Leaderboard 排行榜系统
type Leaderboard struct {
	rdb    *redis.Client
	ctx    context.Context
	prefix string // Redis key 前缀
}

func NewLeaderboard(rdb *redis.Client, prefix string) *Leaderboard {
	return &Leaderboard{
		rdb:    rdb,
		ctx:    context.Background(),
		prefix: prefix,
	}
}

// SubmitScore 提交分数（取最高分）
func (lb *Leaderboard) SubmitScore(season, playerID string, score float64) error {
	key := lb.getKey(season)

	pipe := lb.rdb.Pipeline()
	// ZADD with GT flag: only update if new score is greater
	pipe.ZAddArgs(lb.ctx, key, redis.ZAddArgs{
		GT:      true,
		Members: []redis.Z{{Score: score, Member: playerID}},
	})
	pipe.HSet(lb.ctx, lb.getScoreDetailKey(season), playerID, score)
	pipe.Expire(lb.ctx, key, 90*24*time.Hour) // 90 天过期
	_, err := pipe.Exec(lb.ctx)
	return err
}

// GetTopN 获取前 N 名
func (lb *Leaderboard) GetTopN(season string, n int64) ([]RankEntry, error) {
	key := lb.getKey(season)

	results, err := lb.rdb.ZRevRangeWithScores(lb.ctx, key, 0, n-1).Result()
	if err != nil {
		return nil, err
	}

	entries := make([]RankEntry, 0, len(results))
	for i, z := range results {
		playerID, ok := z.Member.(string)
		if !ok {
			continue
		}
		entries = append(entries, RankEntry{
			Rank:     int64(i + 1),
			PlayerID: playerID,
			Score:    z.Score,
		})
	}
	return entries, nil
}

// GetPlayerRank 获取玩家排名和分数
func (lb *Leaderboard) GetPlayerRank(season, playerID string) (*RankEntry, error) {
	key := lb.getKey(season)

	pipe := lb.rdb.Pipeline()
	rankCmd := pipe.ZRevRank(lb.ctx, key, playerID)
	scoreCmd := pipe.ZScore(lb.ctx, key, playerID)
	_, err := pipe.Exec(lb.ctx)
	if err != nil {
		return nil, err
	}

	rank, err := rankCmd.Result()
	if err != nil {
		return nil, fmt.Errorf("player not ranked")
	}

	score, _ := scoreCmd.Result()

	return &RankEntry{
		Rank:     rank + 1,
		PlayerID: playerID,
		Score:    score,
	}, nil
}

// GetAroundPlayer 获取玩家附近排名（显示前后各 N 名）
func (lb *Leaderboard) GetAroundPlayer(season, playerID string, rangeSize int64) ([]RankEntry, error) {
	key := lb.getKey(season)

	// 先获取玩家排名
	rank, err := lb.rdb.ZRevRank(lb.ctx, key, playerID).Result()
	if err != nil {
		return nil, fmt.Errorf("player not ranked")
	}

	start := rank - rangeSize
	if start < 0 {
		start = 0
	}
	stop := rank + rangeSize

	results, err := lb.rdb.ZRevRangeWithScores(lb.ctx, key, start, stop).Result()
	if err != nil {
		return nil, err
	}

	entries := make([]RankEntry, 0, len(results))
	for i, z := range results {
		playerID, ok := z.Member.(string)
		if !ok {
			continue
		}
		entries = append(entries, RankEntry{
			Rank:     int64(start) + int64(i+1),
			PlayerID: playerID,
			Score:    z.Score,
		})
	}
	return entries, nil
}

// GetPercentile 获取玩家百分位（如 Top 5%）
func (lb *Leaderboard) GetPercentile(season, playerID string) (float64, error) {
	key := lb.getKey(season)

	pipe := lb.rdb.Pipeline()
	rankCmd := pipe.ZRevRank(lb.ctx, key, playerID)
	countCmd := pipe.ZCard(lb.ctx, key)
	_, err := pipe.Exec(lb.ctx)
	if err != nil {
		return 0, err
	}

	rank, _ := rankCmd.Result()
	total, _ := countCmd.Result()
	if total == 0 {
		return 0, fmt.Errorf("empty leaderboard")
	}

	percentile := float64(rank+1) / float64(total) * 100
	return percentile, nil
}

// BatchUpdate 批量更新赛季结算分数
func (lb *Leaderboard) BatchUpdate(season string, updates map[string]float64) error {
	key := lb.getKey(season)

	pipe := lb.rdb.Pipeline()
	for playerID, score := range updates {
		pipe.ZAddArgs(lb.ctx, key, redis.ZAddArgs{
			GT:      true,
			Members: []redis.Z{{Score: score, Member: playerID}},
		})
	}
	_, err := pipe.Exec(lb.ctx)
	return err
}

// Helper methods
func (lb *Leaderboard) getKey(season string) string {
	return fmt.Sprintf("%s:leaderboard:%s", lb.prefix, season)
}

func (lb *Leaderboard) getScoreDetailKey(season string) string {
	return fmt.Sprintf("%s:score_detail:%s", lb.prefix, season)
}

// RankEntry 排名条目
type RankEntry struct {
	Rank     int64   `json:"rank"`
	PlayerID string  `json:"player_id"`
	Score    float64 `json:"score"`
}

func (e RankEntry) String() string {
	return fmt.Sprintf("#%d %s: %.0f", e.Rank, e.PlayerID, e.Score)
}

// 快速整数转换辅助
func parseInt(s string) int {
	v, _ := strconv.Atoi(s)
	return v
}
```

### 5.1 协议选择

| 协议 | 可靠性 | 延迟 | 适用场景 |
|------|--------|------|----------|
| **TCP** | 可靠有序 | 较高（握手/重传） | 回合制、聊天、HTTP API |
| **UDP** | 不可靠 | 极低 | FPS/RTS 实时游戏 |
| **WebSocket** | 可靠 | 中等 | Web 游戏、H5 |
| **QUIC** | 可靠+低延迟 | 低 | 现代实时游戏 |
| **KCP** | 可靠(可选)+低延迟 | 极低 | 国内游戏常用 |

## 6. 最佳实践

### 6.1 架构设计原则

- ✅ **服务器权威**: 关键逻辑（伤害计算、碰撞、移动验证）在服务器执行
- ✅ **AOI（兴趣区域）**: 只同步玩家附近的实体状态，降低带宽
- ✅ **无状态网关**: 使用负载均衡 + 多台无状态游戏服务器
- ✅ **消息压缩**: Protobuf 比 JSON 节省 60%+ 带宽
- ✅ **增量同步**: 只发送变化的字段，而非完整状态
- ❌ 避免客户端发送位置/血量等关键状态（可被篡改）
- ❌ 避免单个 Tick 中处理过多逻辑（保持帧率稳定）

### 6.2 安全性

```go
// 服务器端速率限制
type RateLimiter struct {
    limits map[string]*tokenBucket
    mu     sync.Mutex
}

type tokenBucket struct {
    tokens   float64
    lastTime time.Time
    rate     float64 // tokens per second
    capacity float64
}

func (rl *RateLimiter) Allow(playerID string) bool {
    rl.mu.Lock()
    defer rl.mu.Unlock()

    bucket, ok := rl.limits[playerID]
    if !ok {
        bucket = &tokenBucket{
            tokens:   10,
            lastTime: time.Now(),
            rate:     10, // 10 msg/sec
            capacity: 10,
        }
        rl.limits[playerID] = bucket
    }

    now := time.Now()
    elapsed := now.Sub(bucket.lastTime).Seconds()
    bucket.tokens = min(bucket.capacity, bucket.tokens+elapsed*bucket.rate)
    bucket.lastTime = now

    if bucket.tokens >= 1 {
        bucket.tokens--
        return true
    }
    return false
}
```
