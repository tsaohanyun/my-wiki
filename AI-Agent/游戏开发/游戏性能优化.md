# 游戏性能优化

游戏性能优化的核心目标：在目标帧率（30/60/120 FPS）下，最大化画面质量和游戏体验。

---

## 一、渲染优化

### 1.1 Draw Call 合批（Batching）

```csharp
// Unity: 静态批处理 - 相同材质的静态物体自动合批
// 设置方式：Inspector → Static → Batching Static

// Unity: 动态批处理 - 小网格 (< 300 vertices) 自动合批
// 限制：不支持不同材质、不支持多Pass shader

// Unity: GPU Instancing - 大量相同网格不同材质属性
// Shader 中启用：
// Tags { "RenderType"="Opaque" }
// #pragma multi_compile_instancing

// 最佳方案：使用 SRP Batcher（URP/HDRP）
// 条件：相同 Shader Variant，不同材质属性

// 实际示例：使用 MaterialPropertyBlock 避免材质拷贝
using UnityEngine;

public class BatchedRenderer : MonoBehaviour
{
    [SerializeField] private MeshRenderer[] renderers;
    [SerializeField] private Color baseColor = Color.white;

    private MaterialPropertyBlock propBlock;

    void Awake()
    {
        propBlock = new MaterialPropertyBlock();
    }

    // 每个实例设置不同颜色，但不破坏合批
    public void SetColor(Color color)
    {
        foreach (var renderer in renderers)
        {
            renderer.GetPropertyBlock(propBlock);
            propBlock.SetColor("_BaseColor", color);
            renderer.SetPropertyBlock(propBlock);
        }
    }
}
```

### 1.2 Shader 优化

```hlsl
// 优化前：全精度计算
fixed4 frag(v2f i) : SV_Target
{
    fixed4 col = tex2D(_MainTex, i.uv);
    float3 worldNormal = normalize(i.worldNormal);
    float3 lightDir = normalize(_WorldSpaceLightPos0.xyz);
    float NdotL = saturate(dot(worldNormal, lightDir));
    return col * NdotL;
}

// 优化后：使用 half 精度（移动端重要）
half4 frag(v2f i) : SV_Target
{
    half4 col = tex2D(_MainTex, i.uv);
    half3 worldNormal = normalize(i.worldNormal);
    half3 lightDir = normalize(_WorldSpaceLightPos0.xyz);
    half NdotL = saturate(dot(worldNormal, lightDir));
    return col * NdotL;
}

// 避免动态分支
// ❌ if (NdotL > 0.5) return bright; else return dark;
// ✅ lerp(dark, bright, step(0.5, NdotL));

// 纹理采样优化
// 使用 texture2D 的 LOD 版本避免过度 mipmap
half4 col = tex2Dlod(_MainTex, half4(i.uv, 0, lod_level));
```

### 1.3 后处理优化

```csharp
// Unity URP 后处理性能调优
using UnityEngine;
using UnityEngine.Rendering;
using UnityEngine.Rendering.Universal;

public class PerformancePostProcess : MonoBehaviour
{
    [SerializeField] private Volume globalVolume;

    public void SetQualityLevel(int level)
    {
        var profile = globalVolume.profile;

        // 根据质量级别开关后处理
        if (profile.TryGet(out Bloom bloom))
        {
            bloom.active = level >= 1;
            bloom.intensity.overrideState = true;
            bloom.intensity.value = level >= 2 ? 1.5f : 0.8f;
            // 降低迭代次数
            bloom.maxIterations.value = level >= 2 ? 8 : 4;
        }

        if (profile.TryGet(out DepthOfField dof))
        {
            dof.active = level >= 2; // 只在高画质开景深
        }

        if (profile.TryGet(out MotionBlur motionBlur))
        {
            motionBlur.active = level >= 1;
        }
    }
}
```

### 2.2 资源加载与卸载

```csharp
// Unity Addressables 异步加载
using UnityEngine;
using UnityEngine.AddressableAssets;
using UnityEngine.ResourceManagement.AsyncOperations;
using System.Collections.Generic;

public class AssetManager : MonoBehaviour
{
    private Dictionary<string, AsyncOperationHandle> loadedAssets = new();

    // 异步加载
    public async System.Threading.Tasks.Task<T> LoadAssetAsync<T>(string key) where T : Object
    {
        if (loadedAssets.TryGetValue(key, out var existing))
            return existing.Result as T;

        var handle = Addressables.LoadAssetAsync<T>(key);
        await handle.Task;

        if (handle.Status == AsyncOperationStatus.Succeeded)
        {
            loadedAssets[key] = handle;
            return handle.Result;
        }

        Debug.LogError($"加载失败: {key}");
        return null;
    }

    // 卸载不需要的资源
    public void UnloadAsset(string key)
    {
        if (loadedAssets.TryGetValue(key, out var handle))
        {
            Addressables.Release(handle);
            loadedAssets.Remove(key);
        }
    }

    // 卸载所有
    public void UnloadAll()
    {
        foreach (var handle in loadedAssets.Values)
            Addressables.Release(handle);
        loadedAssets.Clear();
    }

    // 强制卸载未使用的资源
    public void UnloadUnused()
    {
        Resources.UnloadUnusedAssets();
        System.GC.Collect();
    }
}
```

## 三、LOD（Level of Detail）

### 3.1 LOD Group 配置

```csharp
using UnityEngine;

public class LODSetup : MonoBehaviour
{
    [SerializeField] private Mesh[] lodMeshes; // LOD0~LOD3
    [SerializeField] private float[] lodDistances = { 10f, 30f, 60f, 100f };

    private MeshFilter meshFilter;
    private MeshRenderer meshRenderer;
    private Camera mainCam;

    void Awake()
    {
        meshFilter = GetComponent<MeshFilter>();
        meshRenderer = GetComponent<MeshRenderer>();
        mainCam = Camera.main;
    }

    void Update()
    {
        float distance = Vector3.Distance(
            transform.position, mainCam.transform.position);

        // 根据距离切换 LOD
        int lodIndex = lodMeshes.Length - 1; // 默认最低 LOD
        for (int i = 0; i < lodDistances.Length; i++)
        {
            if (distance < lodDistances[i])
            {
                lodIndex = i;
                break;
            }
        }

        if (meshFilter.sharedMesh != lodMeshes[lodIndex])
        {
            meshFilter.sharedMesh = lodMeshes[lodIndex];
        }
    }
}
```

### 3.2 HLOD（Hierarchical LOD）

```csharp
// HLOD: 将多个小物体合并为一个大低面数网格
// 适用场景：城市建筑群、森林树木

// Unity 实现思路：
// 1. 预烘焙：将多个静态物体合并为简化 Mesh
// 2. 运行时：近距离显示原始物体，远距离显示合并 Mesh

public class HLODGroup : MonoBehaviour
{
    [SerializeField] private GameObject[] detailObjects;  // 近距离物体
    [SerializeField] private GameObject simplifiedObject;  // 远距离合并体
    [SerializeField] private float transitionDistance = 100f;

    private Camera mainCam;

    void Start()
    {
        mainCam = Camera.main;
    }

    void Update()
    {
        float dist = Vector3.Distance(transform.position, mainCam.transform.position);
        bool useDetail = dist < transitionDistance;

        foreach (var obj in detailObjects)
            obj.SetActive(useDetail);

        simplifiedObject.SetActive(!useDetail);
    }
}
```

```csharp
using UnityEngine;

public class VisibilityChecker : MonoBehaviour

    private Camera cam;
    private Plane[] frustumPlanes;

    {
        cam = Camera.main;

    }
}
```

### 4.2 遮挡剔除优化策略

```csharp
// 策略1: 手动区域管理（大世界场景）
public class WorldStreamer : MonoBehaviour
{
    [System.Serializable]
    public class WorldChunk
    {
        public string sceneName;
        public Bounds bounds;
        public bool isLoaded;
    }

    [SerializeField] private WorldChunk[] chunks;
    [SerializeField] private float loadDistance = 200f;
    [SerializeField] private float unloadDistance = 300f;

    private Transform player;

    void Update()
    {
        if (!player) return;

        foreach (var chunk in chunks)
        {
            float dist = DistanceToBounds(player.position, chunk.bounds);

            if (!chunk.isLoaded && dist < loadDistance)
            {
                LoadChunk(chunk);
            }
            else if (chunk.isLoaded && dist > unloadDistance)
            {
                UnloadChunk(chunk);
            }
        }
    }

    private float DistanceToBounds(Vector3 point, Bounds bounds)
    {
        return Vector3.Distance(point, bounds.ClosestPoint(point));
    }

    private async void LoadChunk(WorldChunk chunk)
    {
        chunk.isLoaded = true;
        await UnityEngine.SceneManagement.SceneManager.LoadSceneAsync(
            chunk.sceneName, UnityEngine.SceneManagement.LoadSceneMode.Additive).Task;
    }

    private async void UnloadChunk(WorldChunk chunk)
    {
        chunk.isLoaded = false;
        await UnityEngine.SceneManagement.SceneManager.UnloadSceneAsync(
            chunk.sceneName).Task;
    }
}

// 策略2: 层级裁剪 - 远处小物体直接不渲染
public class DistanceCuller : MonoBehaviour
{
    [SerializeField] private float cullDistance = 150f;
    private Renderer[] renderers;

    void Start()
    {
        renderers = GetComponentsInChildren<Renderer>();
    }

    void Update()
    {
        float dist = Vector3.Distance(
            transform.position, Camera.main.transform.position);

        bool shouldRender = dist < cullDistance;
        foreach (var r in renderers)
            r.enabled = shouldRender;
    }
}
```

### 5.2 性能检查清单

```
渲染：
  □ Draw Call 数量（移动 < 100，PC < 500）
  □ 三角面数（移动 < 100K，PC < 1M）
  □ Overdraw 检查（避免大面积半透明叠加）
  □ Shader 复杂度（避免移动端多 Pass）
  □ 纹理压缩格式正确

内存：
  □ GC Alloc 为零（每帧）
  □ 纹理总内存预算（移动 < 200MB）
  □ Mesh 内存优化（移除未用 UV/Color 通道）
  □ 音频压缩（移动端用 Vorbis 70%）

CPU：
  □ 主线程耗时 < 16ms（60fps）
  □ 物理模拟耗时 < 2ms
  □ 动画系统耗时 < 2ms
  □ 避免每帧 Find/GetComponent

GPU：
  □ 像素填充率（减少全屏后处理）
  □ 顶点处理（使用 LOD）
  □ 带宽（纹理压缩 + Mipmap）
```

## 六、最佳实践

| 实践 | 说明 |
|------|------|
| **Profile 先行** | 先用 Profiler 找瓶颈，再优化 |
| **优化 20% 代码获得 80% 性能** | 找到真正的热点循环 |
| **对象池管理频繁创建销毁** | 子弹、粒子、伤害数字 |
| **避免 Update 中做重型操作** | 分帧处理、协程延迟 |
| **使用 Jobs + Burst（Unity）** | 多线程 + SIMD 加速计算 |
| **纹理流式加载（Texture Streaming）** | 按需加载纹理 Mipmap 层级 |
| **遮挡剔除减少 GPU 负载** | 不渲染看不见的物体 |
| **内存预算 + 峰值监控** | 避免运行时 OOM 崩溃 |
