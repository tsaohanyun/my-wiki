# 监控告警体系设计

## 概述

一个完善的监控告警体系是保障系统可靠性的基石。本文档涵盖监控指标设计、告警规则、通知渠道和值班轮转的完整方案。

```python
from prometheus_client import Counter, Histogram, Gauge, start_http_server
import time

# HTTP请求计数器
REQUEST_COUNT = Counter(
    'http_requests_total',
    'Total HTTP requests',
    ['method', 'endpoint', 'status']
)

# 请求延迟直方图
REQUEST_LATENCY = Histogram(
    'http_request_duration_seconds',
    'HTTP request latency',
    ['method', 'endpoint'],
    buckets=[0.01, 0.05, 0.1, 0.25, 0.5, 1, 2.5, 5]
)

# 活跃连接数
ACTIVE_CONNECTIONS = Gauge(
    'active_connections',
    'Number of active connections'
)

# 指标使用示例
@app.middleware("http")
async def metrics_middleware(request, call_next):
    start = time.time()
    response = await call_next(request)
    duration = time.time() - start

    REQUEST_COUNT.labels(
        method=request.method,
        endpoint=request.url.path,
        status=response.status_code
    ).inc()

    REQUEST_LATENCY.labels(
        method=request.method,
        endpoint=request.url.path
    ).observe(duration)

    return response
```

## 2. Prometheus告警规则

### 2.1 基础设施告警

```yaml
# rules/infra-alerts.yml
groups:
  - name: infra-alerts
    interval: 30s
    rules:
      # CPU使用率过高
      - alert: HighCPUUsage
        expr: |
          100 - (avg by(instance) (rate(node_cpu_seconds_total{mode="idle"}[5m])) * 100) > 85
        for: 5m
        labels:
          severity: warning
          team: infra
        annotations:
          summary: "实例 {{ $labels.instance }} CPU使用率过高"
          description: "CPU使用率 {{ $value | printf \"%.1f\" }}%，持续超过5分钟"
          runbook: "https://wiki.example.com/runbooks/high-cpu"

      # 内存使用率过高
      - alert: HighMemoryUsage
        expr: |
          (1 - node_memory_MemAvailable_bytes / node_memory_MemTotal_bytes) * 100 > 90
        for: 3m
        labels:
          severity: critical
          team: infra
        annotations:
          summary: "实例 {{ $labels.instance }} 内存使用率超过90%"
          description: "当前内存使用率 {{ $value | printf \"%.1f\" }}%"

      # 磁盘空间不足
      - alert: DiskSpaceLow
        expr: |
          (1 - node_filesystem_avail_bytes{fstype!~"tmpfs|overlay"} / node_filesystem_size_bytes) * 100 > 85
        for: 5m
        labels:
          severity: warning
          team: infra
        annotations:
          summary: "实例 {{ $labels.instance }} 磁盘 {{ $labels.mountpoint }} 使用率超过85%"
          description: "当前使用率 {{ $value | printf \"%.1f\" }}%"

      # 磁盘将在4小时内满
      - alert: DiskWillFillIn4Hours
        expr: |
          predict_linear(node_filesystem_avail_bytes{fstype!~"tmpfs|overlay"}[1h], 4*3600) < 0
        for: 15m
        labels:
          severity: critical
          team: infra
        annotations:
          summary: "实例 {{ $labels.instance }} 磁盘预计4小时内将满"

      # 实例宕机
      - alert: InstanceDown
        expr: up == 0
        for: 1m
        labels:
          severity: critical
          team: infra
        annotations:
          summary: "实例 {{ $labels.instance }} 已宕机"
```

## 4. 通知渠道

### 4.1 多渠道通知网关

```python
# notification-gateway/app.py
from fastapi import FastAPI, Request
from enum import Enum
import httpx
import json

app = FastAPI()

class Channel(str, Enum):
    SLACK = "slack"
    DINGTALK = "dingtalk"
    WECHAT_WORK = "wechat_work"
    EMAIL = "email"
    PHONE = "phone"

async def send_slack(webhook_url: str, message: dict):
    """发送Slack通知"""
    async with httpx.AsyncClient() as client:
        await client.post(webhook_url, json={
            "text": message["text"],
            "blocks": [
                {
                    "type": "header",
                    "text": {"type": "plain_text", "text": message["title"]}
                },
                {
                    "type": "section",
                    "text": {"type": "mrkdwn", "text": message["body"]}
                }
            ]
        })

async def send_dingtalk(webhook_url: str, message: dict):
    """发送钉钉通知"""
    async with httpx.AsyncClient() as client:
        await client.post(webhook_url, json={
            "msgtype": "markdown",
            "markdown": {
                "title": message["title"],
                "text": f"## {message['title']}\n\n{message['body']}"
            }
        })

async def send_wechat_work(webhook_url: str, message: dict):
    """发送企业微信通知"""
    async with httpx.AsyncClient() as client:
        await client.post(webhook_url, json={
            "msgtype": "markdown",
            "markdown": {
                "content": f"## {message['title']}\n{message['body']}"
            }
        })

@app.post("/webhook/{channel}")
async def handle_webhook(channel: Channel, request: Request):
    payload = await request.json()

    for alert in payload.get("alerts", []):
        message = {
            "title": alert["annotations"].get("summary", alert["labels"].get("alertname")),
            "body": alert["annotations"].get("description", ""),
            "status": alert["status"],
        }

        if channel == Channel.SLACK:
            await send_slack(SLACK_WEBHOOK, message)
        elif channel == Channel.DINGTALK:
            await send_dingtalk(DINGTALK_WEBHOOK, message)
        elif channel == Channel.WECHAT_WORK:
            await send_wechat_work(WECHAT_WORK_WEBHOOK, message)

    return {"status": "ok"}
```

### 5.2 自建值班系统

```python
# oncall/rotation.py
from datetime import datetime, timedelta
from dataclasses import dataclass
from typing import List
import json

@dataclass
class OnCallPerson:
    name: str
    phone: str
    email: str
    wechat_id: str

@dataclass
class OnCallSchedule:
    primary: OnCallPerson
    secondary: OnCallPerson
    start: datetime
    end: datetime

class OnCallRotation:
    def __init__(self, team: List[OnCallPerson], rotation_days: int = 7):
        self.team = team
        self.rotation_days = rotation_days
        self.schedule_file = "/var/lib/oncall/schedule.json"

    def get_current_oncall(self) -> OnCallSchedule:
        """获取当前值班人员"""
        now = datetime.now()
        week_number = now.isocalendar()[1]
        primary_idx = week_number % len(self.team)
        secondary_idx = (week_number + 1) % len(self.team)

        # 计算本周值班开始/结束时间
        days_since_monday = now.weekday()
        week_start = now - timedelta(days=days_since_monday)
        week_start = week_start.replace(hour=0, minute=0, second=0)
        week_end = week_start + timedelta(days=self.rotation_days)

        return OnCallSchedule(
            primary=self.team[primary_idx],
            secondary=self.team[secondary_idx],
            start=week_start,
            end=week_end
        )

    def get_escalation_contacts(self, severity: str) -> List[OnCallPerson]:
        """根据严重级别获取升级联系人"""
        schedule = self.get_current_oncall()

        if severity == "critical":
            # P0: 通知主值班 + 次值班 + 团队负责人
            return [schedule.primary, schedule.secondary, self.team[0]]
        elif severity == "warning":
            # P1: 通知主值班
            return [schedule.primary]
        else:
            return [schedule.primary]

    def notify_oncall(self, alert: dict):
        """通知值班人员"""
        severity = alert["labels"].get("severity", "warning")
        contacts = self.get_escalation_contacts(severity)

        for person in contacts:
            # 多渠道通知
            self._send_wechat(person, alert)
            if severity == "critical":
                self._send_sms(person, alert)
                self._make_phone_call(person, alert)

    def _send_wechat(self, person: OnCallPerson, alert: dict):
        print(f"[微信] -> {person.name}: {alert['annotations']['summary']}")

    def _send_sms(self, person: OnCallPerson, alert: dict):
        print(f"[短信] -> {person.phone}: {alert['annotations']['summary']}")

    def _make_phone_call(self, person: OnCallPerson, alert: dict):
        print(f"[电话] -> {person.phone}: 告警通知")


# 使用示例
team = [
    OnCallPerson("张三", "13800138001", "zhangsan@example.com", "zhangsan"),
    OnCallPerson("李四", "13800138002", "lisi@example.com", "lisi"),
    OnCallPerson("王五", "13800138003", "wangwu@example.com", "wangwu"),
    OnCallPerson("赵六", "13800138004", "zhaoliu@example.com", "zhaoliu"),
]

rotation = OnCallRotation(team, rotation_days=7)
current = rotation.get_current_oncall()
print(f"当前主值班: {current.primary.name}")
print(f"当前副值班: {current.secondary.name}")
```

### 5.3 值班排班表生成脚本

```bash
#!/bin/bash
# generate_oncall_schedule.sh
# 生成未来12周的值班排班表

TEAM=("张三" "李四" "王五" "赵六" "周五" "吴七")
WEEKS=12
OUTPUT_FILE="oncall_schedule.csv"

echo "周次,开始日期,结束日期,主值班,副值班" > "$OUTPUT_FILE"

for ((w=0; w<WEEKS; w++)); do
    week_num=$(date -d "+${w} weeks" +%V)
    start_date=$(date -d "last monday +${w} weeks" +%Y-%m-%d)
    end_date=$(date -d "last monday +$((w+1)) weeks -1 day" +%Y-%m-%d)

    primary_idx=$((week_num % ${#TEAM[@]}))
    secondary_idx=$(( (week_num + 1) % ${#TEAM[@]} ))

    echo "第${week_num}周,${start_date},${end_date},${TEAM[$primary_idx]},${TEAM[$secondary_idx]}" >> "$OUTPUT_FILE"
done

echo "排班表已生成: $OUTPUT_FILE"
cat "$OUTPUT_FILE"
```

```yaml
# 告警质量度量
metrics:
  # 告警信噪比
  signal_to_noise_ratio:
    formula: "actionable_alerts / total_alerts"
    target: "> 0.8"

  # 平均响应时间
  mean_time_to_respond:
    formula: "sum(alert_ack_time - alert_fire_time) / alert_count"

  # 平均恢复时间
  mean_time_to_resolve:
    formula: "sum(alert_resolve_time - alert_fire_time) / alert_count"
    target: "< 1h"

  # 误报率
  false_positive_rate:
    formula: "false_alerts / total_alerts"
```

```bash
#!/bin/bash
# alert_review.sh - 每周告警Review

PROMETHEUS_URL="http://prometheus:9090"
WEEK_AGO="7d"

echo "=== 周告警Review报告 ==="

# Top 10 最频繁告警
echo ""
echo "Top 10 最频繁告警:"

    --data-urlencode 'query=topk(10, count by(alertname)(ALERTS{alertstate="firing"}))' \
    | jq -r '.data.result[] | "  \(.metric.alertname): \(.value[1])次"'
```

## 最佳实践

| 编号 | 实践 | 说明 |
|------|------|------|
